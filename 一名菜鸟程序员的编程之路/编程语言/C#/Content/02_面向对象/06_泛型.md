### 泛型类

```csharp
// 基本泛型类
public class Container<T>
{
    private T item;
    
    public void SetItem(T newItem)
    {
        item = newItem;
    }

    {
        return item;
    }

// 多类型参数
public class KeyValuePair<TKey, TValue>
{
    public TKey Key { get; set; }
    public TValue Value { get; set; }
    
    public KeyValuePair(TKey key, TValue value)
    {
        Key = key;
        Value = value;
    }
}
```

### 泛型方法
```csharp
public class Utilities
{
    // 泛型方法
    public static T FindMax<T>(T a, T b) where T : IComparable<T>
    {
        return a.CompareTo(b) > 0 ? a : b;
    }
    
    // 泛型数组方法
    public static void Swap<T>(T[] array, int i, int j)
    {
        T temp = array[i];
        array[i] = array[j];
        array[j] = temp;
    }
}
```

## Unity中的泛型应用

### 对象池
```csharp
public class ObjectPool<T> where T : MonoBehaviour
{
    private Queue<T> pool = new Queue<T>();
    private T prefab;
    
    public ObjectPool(T prefab, int initialSize)
    {
        this.prefab = prefab;
        for (int i = 0; i < initialSize; i++)
        {
            CreateNewInstance();
        }
    }
    
    public T Get()
    {
        if (pool.Count == 0)
        {
            CreateNewInstance();
        }
        
        T instance = pool.Dequeue();
        instance.gameObject.SetActive(true);
        return instance;
    }
    
    public void Return(T instance)
    {
        instance.gameObject.SetActive(false);
        pool.Enqueue(instance);
    }
    
    private void CreateNewInstance()
    {
        T instance = GameObject.Instantiate(prefab);
        instance.gameObject.SetActive(false);
        pool.Enqueue(instance);
    }
}
```

### 组件管理器
```csharp
public class ComponentManager<T> where T : Component
{
    private Dictionary<string, T> components = new Dictionary<string, T>();
    
    public void Register(string key, T component)
    {
        if (!components.ContainsKey(key))
        {
            components.Add(key, component);
        }
    }
    
    public T Get(string key)
    {
        return components.TryGetValue(key, out T component) ? component : null;
    }
}

// 使用示例
public class GameManager : MonoBehaviour
{
    private ComponentManager<Camera> cameraManager;
    private ComponentManager<AudioSource> audioManager;
    
    private void Start()
    {
        cameraManager = new ComponentManager<Camera>();
        audioManager = new ComponentManager<AudioSource>();
    }
}
```

### 自定义泛型集合
```csharp
public class CircularBuffer<T>
{
    private T[] buffer;
    private int start;
    private int end;
    
    public CircularBuffer(int capacity)
    {
        buffer = new T[capacity];
    }
    
    public void Add(T item)
    {
        buffer[end] = item;
        end = (end + 1) % buffer.Length;
        if (end == start)
        {
            start = (start + 1) % buffer.Length;
        }
    }
    
    public T Get(int index)
    {
        return buffer[(start + index) % buffer.Length];
    }
}
```

## 泛型接口

### 自定义泛型接口
```csharp
public interface IStorage<T>
{
    void Store(T item);
    T Retrieve();
    bool HasItem { get; }
}

public class Inventory<T> : IStorage<T>
{
    private T storedItem;
    private bool hasItem;
    
    public bool HasItem => hasItem;
    
    public void Store(T item)
    {
        storedItem = item;
        hasItem = true;
    }
    
    public T Retrieve()
    {
        if (!hasItem)
            throw new InvalidOperationException();
            
        hasItem = false;
        return storedItem;
    }
}
```
