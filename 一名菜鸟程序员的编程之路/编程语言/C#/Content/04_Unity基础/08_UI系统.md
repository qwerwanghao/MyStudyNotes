public class UIBasics : MonoBehaviour
{
    // UI组件引用
    public Text textComponent;
    public Image imageComponent;
    public Button buttonComponent;
    public Slider sliderComponent;
    public Toggle toggleComponent;
    
    void Start()
    {
        // 设置文本
        textComponent.text = "Hello World";
        textComponent.color = Color.white;
        
        // 设置图片
        imageComponent.sprite = Resources.Load<Sprite>("UI/Icon");
        imageComponent.color = new Color(1, 1, 1, 0.8f);
        
        // 按钮事件
        buttonComponent.onClick.AddListener(OnButtonClick);
        
        // 滑动条事件
        sliderComponent.onValueChanged.AddListener(OnSliderValueChanged);
        
        // 切换按钮事件
        toggleComponent.onValueChanged.AddListener(OnToggleValueChanged);
    }
    
    private void OnButtonClick()
    {
        Debug.Log("Button clicked!");
    }
    
    private void OnSliderValueChanged(float value)
    {
        Debug.Log($"Slider value: {value}");
    }
    
    private void OnToggleValueChanged(bool isOn)
    {
        Debug.Log($"Toggle is: {(isOn ? "On" : "Off")}");
    }
}

public class UILayout : MonoBehaviour
{
    private RectTransform rectTransform;
    public GridLayoutGroup gridLayout;
    public VerticalLayoutGroup verticalLayout;
    
    void Start()
    {
        rectTransform = GetComponent<RectTransform>();
        
        // 设置RectTransform
        rectTransform.anchoredPosition = Vector2.zero;
        rectTransform.sizeDelta = new Vector2(200, 100);
        
        // 配置网格布局
        gridLayout.cellSize = new Vector2(100, 100);
        gridLayout.spacing = new Vector2(10, 10);
        gridLayout.constraint = GridLayoutGroup.Constraint.FixedColumnCount;
        gridLayout.constraintCount = 3;
        
        // 配置垂直布局
        verticalLayout.spacing = 10;
        verticalLayout.childAlignment = TextAnchor.MiddleCenter;
        verticalLayout.childForceExpandHeight = false;
    }
}

public class UIEventSystem : MonoBehaviour, 
    IPointerClickHandler, 
    IPointerEnterHandler,
    IPointerExitHandler,
    IDragHandler
{
    public void OnPointerClick(PointerEventData eventData)
    {
        if (eventData.button == PointerEventData.InputButton.Left)
        {
            HandleLeftClick();
        }
        else if (eventData.button == PointerEventData.InputButton.Right)
        {
            HandleRightClick();
        }
    }
    
    public void OnPointerEnter(PointerEventData eventData)
    {
        // 鼠标进入
        transform.localScale = Vector3.one * 1.1f;
    }
    
    public void OnPointerExit(PointerEventData eventData)
    {
        // 鼠标离开
        transform.localScale = Vector3.one;
    }
    
    public void OnDrag(PointerEventData eventData)
    {
        // 拖拽处理
        transform.position = eventData.position;
    }
}

### UI动画系统
```csharp
public class UIAnimation : MonoBehaviour
{
    public CanvasGroup canvasGroup;
    
    // 淡入动画
    public IEnumerator FadeIn(float duration)
    {
        float elapsed = 0;
        while (elapsed < duration)
        {
            canvasGroup.alpha = Mathf.Lerp(0, 1, elapsed / duration);
            elapsed += Time.deltaTime;
            yield return null;
        }
        canvasGroup.alpha = 1;
    }
    
    // UI缩放动画
    public IEnumerator ScaleAnimation(Vector3 targetScale, float duration)
    {
        Vector3 startScale = transform.localScale;
        float elapsed = 0;
        
        while (elapsed < duration)
        {
            transform.localScale = Vector3.Lerp(
                startScale, 
                targetScale, 
                elapsed / duration
            );
            elapsed += Time.deltaTime;
            yield return null;
        }
        
        transform.localScale = targetScale;
    }
}

## UI管理系统

### UI窗口管理
```csharp
public class UIWindowManager : MonoBehaviour
{
    private Dictionary<string, GameObject> windows = 
        new Dictionary<string, GameObject>();
    private Stack<GameObject> windowStack = new Stack<GameObject>();
    
    // 注册窗口
    public void RegisterWindow(string name, GameObject window)
    {
        if (!windows.ContainsKey(name))
        {
            windows.Add(name, window);
            window.SetActive(false);
        }
    }
    
    // 显示窗口
    public void ShowWindow(string name)
    {
        if (windows.TryGetValue(name, out GameObject window))
        {
            if (windowStack.Count > 0)
            {
                windowStack.Peek().SetActive(false);
            }
            
            window.SetActive(true);
            windowStack.Push(window);
        }
    }
    
    // 关闭顶部窗口
    public void CloseTopWindow()
    {
        if (windowStack.Count > 0)
        {
            GameObject window = windowStack.Pop();
            window.SetActive(false);
            
            if (windowStack.Count > 0)
            {
                windowStack.Peek().SetActive(true);
            }
        }
    }
}
```

### UI状态管理
```csharp
public class UIStateManager : MonoBehaviour
{
    private Dictionary<string, UIState> uiStates = 
        new Dictionary<string, UIState>();
    private UIState currentState;
    
    [System.Serializable]
    public class UIState
    {
        public string name;
        public GameObject[] activeObjects;
        public GameObject[] inactiveObjects;
    }
    
    // 切换UI状态
    public void SwitchState(string stateName)
    {
        if (uiStates.TryGetValue(stateName, out UIState newState))
        {
            // 禁用当前状态的对象
            if (currentState != null)
            {
                foreach (var obj in currentState.activeObjects)
                {
                    obj.SetActive(false);
                }
            }
            
            // 启用新状态的对象
            foreach (var obj in newState.activeObjects)
            {
                obj.SetActive(true);
            }
            foreach (var obj in newState.inactiveObjects)
            {
                obj.SetActive(false);
            }
            
            currentState = newState;
        }
    }
}

## UI优化

### 性能优化
```csharp
public class UIOptimization : MonoBehaviour
{
    // UI对象池
    public class UIObjectPool
    {
        private GameObject prefab;
        private Queue<GameObject> pool;
        
        public UIObjectPool(GameObject prefab, int initialSize)
        {
            this.prefab = prefab;
            pool = new Queue<GameObject>();
            
            for (int i = 0; i < initialSize; i++)
            {
                CreateNewObject();
            }
        }
        
        private void CreateNewObject()
        {
            GameObject obj = Instantiate(prefab);
            obj.SetActive(false);
            pool.Enqueue(obj);
        }
        
        public GameObject Get()
        {
            if (pool.Count == 0)
            {
                CreateNewObject();
            }
            
            GameObject obj = pool.Dequeue();
            obj.SetActive(true);
            return obj;
        }
        
        public void Return(GameObject obj)
        {
            obj.SetActive(false);
            pool.Enqueue(obj);
        }
    }
    
    // Canvas重建优化
    public void OptimizeCanvasRebuild()
    {
        LayoutRebuilder.ForceRebuildLayoutImmediate(
            GetComponent<RectTransform>()
        );
    }
}
```

### 动态加载
```csharp
public class UIDynamicLoading : MonoBehaviour
{
    private Dictionary<string, AssetBundle> loadedBundles = 
        new Dictionary<string, AssetBundle>();
    
    // 异步加载UI资源
    public async Task<GameObject> LoadUIAssetAsync(string bundleName, 
        string assetName)
    {
        if (!loadedBundles.ContainsKey(bundleName))
        {
            var bundlePath = Path.Combine(Application.streamingAssetsPath, 
                bundleName);
            var bundle = await AssetBundle.LoadFromFileAsync(bundlePath);
            loadedBundles[bundleName] = bundle;
        }
        
        var assetBundle = loadedBundles[bundleName];
        var request = assetBundle.LoadAssetAsync<GameObject>(assetName);
        await Task.Yield();
        
        return request.asset as GameObject;
    }
    
    // 清理未使用的资源
    public void CleanupUnusedAssets()
    {
        foreach (var bundle in loadedBundles.Values)
        {
            bundle.Unload(false);
        }
        loadedBundles.Clear();
        Resources.UnloadUnusedAssets();
    }
}
