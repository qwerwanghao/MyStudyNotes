public class BasicInput : MonoBehaviour
{
    void Update()
    {
        // 键盘输入
        if (Input.GetKeyDown(KeyCode.Space))
        {
            Jump();
        }
        
        // 持续按键检测
        if (Input.GetKey(KeyCode.W))
        {
            MoveForward();
        }
        
        // 按键抬起检测
        if (Input.GetKeyUp(KeyCode.S))
        {
            StopMoving();
        }
        
        // 虚拟轴输入
        float horizontal = Input.GetAxis("Horizontal");
        float vertical = Input.GetAxis("Vertical");
        Vector3 movement = new Vector3(horizontal, 0, vertical);
    }
    
    // 鼠标输入
    void MouseInput()
    {
        // 鼠标位置
        Vector3 mousePosition = Input.mousePosition;
        
        // 鼠标按键
        if (Input.GetMouseButtonDown(0)) // 左键
        {
            Shoot();
        }
        
        if (Input.GetMouseButton(1)) // 右键
        {
            Aim();
        }
        
        // 鼠标滚轮
        float scrollDelta = Input.GetAxis("Mouse ScrollWheel");
        Zoom(scrollDelta);
    }
}

### 输入管理器
```csharp
public class InputManager : MonoBehaviour
{
    // 输入配置
    [System.Serializable]
    public class InputMapping
    {
        public string actionName;
        public KeyCode primaryKey;
        public KeyCode alternateKey;
    }
    
    public List<InputMapping> inputMappings = new List<InputMapping>();
    private Dictionary<string, bool> actionStates = 
        new Dictionary<string, bool>();
    
    void Update()
    {
        foreach (var mapping in inputMappings)
        {
            bool isPressed = Input.GetKey(mapping.primaryKey) || 
                           Input.GetKey(mapping.alternateKey);
            actionStates[mapping.actionName] = isPressed;
        }
    }
    
    public bool GetAction(string actionName)
    {
        return actionStates.ContainsKey(actionName) && 
               actionStates[actionName];
    }
}
```

## 新输入系统

### 输入动作配置
```csharp
using UnityEngine.InputSystem;

public class PlayerControls : MonoBehaviour
{
    private PlayerInput playerInput;
    private InputAction moveAction;
    private InputAction jumpAction;
    
    void Awake()
    {
        playerInput = GetComponent<PlayerInput>();
        
        // 获取输入动作引用
        moveAction = playerInput.actions["Move"];
        jumpAction = playerInput.actions["Jump"];
        
        // 注册回调
        moveAction.performed += OnMove;
        jumpAction.performed += OnJump;
    }
    
    void OnMove(InputAction.CallbackContext context)
    {
        Vector2 moveInput = context.ReadValue<Vector2>();
        Vector3 movement = new Vector3(moveInput.x, 0, moveInput.y);
        transform.Translate(movement * Time.deltaTime);
    }
    
    void OnJump(InputAction.CallbackContext context)
    {
        if (context.performed)
        {
            Jump();
        }
    }
    
    void OnDestroy()
    {
        // 取消注册
        moveAction.performed -= OnMove;
        jumpAction.performed -= OnJump;
    }
}
```

### 输入设备处理
```csharp
public class InputDeviceHandler : MonoBehaviour
{
    void OnEnable()
    {
        // 监听设备变化
        InputSystem.onDeviceChange += OnDeviceChange;
    }
    
    void OnDisable()
    {
        InputSystem.onDeviceChange -= OnDeviceChange;
    }
    
    void OnDeviceChange(InputDevice device, InputDeviceChange change)
    {
        switch (change)
        {
            case InputDeviceChange.Added:
                Debug.Log($"Device added: {device.name}");
                ConfigureDeviceInput(device);
                break;
                
            case InputDeviceChange.Removed:
                Debug.Log($"Device removed: {device.name}");
                HandleDeviceDisconnection(device);
                break;
        }
    }
    
    void ConfigureDeviceInput(InputDevice device)
    {
        if (device is Gamepad)
        {
            SetupGamepadControls(device as Gamepad);
        }
        else if (device is Keyboard)
        {
            SetupKeyboardControls(device as Keyboard);
        }
    }
}
```

## 触摸输入

### 基本触摸处理
```csharp
public class TouchInputHandler : MonoBehaviour
{
    private float minSwipeDistance = 50f;
    private Vector2 touchStart;
    
    void Update()
    {
        // 单点触摸
        if (Input.touchCount > 0)
        {
            Touch touch = Input.GetTouch(0);
            
            switch (touch.phase)
            {
                case TouchPhase.Began:
                    touchStart = touch.position;
                    break;
                    
                case TouchPhase.Moved:
                    ProcessSwipe(touch);
                    break;
                    
                case TouchPhase.Ended:
                    FinishTouch(touch);
                    break;
            }
        }
    }
    
    void ProcessSwipe(Touch touch)
    {
        Vector2 swipeDelta = touch.position - touchStart;
        
        if (swipeDelta.magnitude > minSwipeDistance)
        {
            if (Mathf.Abs(swipeDelta.x) > Mathf.Abs(swipeDelta.y))
            {
                // 水平滑动
                if (swipeDelta.x > 0)
                    SwipeRight();
                else
                    SwipeLeft();
            }
            else
            {
                // 垂直滑动
                if (swipeDelta.y > 0)
                    SwipeUp();
                else
                    SwipeDown();
            }
        }
    }
}
```

### 多点触摸
```csharp
public class MultiTouchHandler : MonoBehaviour
{
    private float initialDistance;
    private Vector3 initialScale;
    
    void Update()
    {
        // 处理多点触摸
        if (Input.touchCount == 2)
        {
            Touch touch0 = Input.GetTouch(0);
            Touch touch1 = Input.GetTouch(1);
            
            if (touch0.phase == TouchPhase.Began || 
                touch1.phase == TouchPhase.Began)
            {
                initialDistance = Vector2.Distance(
                    touch0.position, 
                    touch1.position
                );
                initialScale = transform.localScale;
            }
            else if (touch0.phase == TouchPhase.Moved || 
                     touch1.phase == TouchPhase.Moved)
            {
                float currentDistance = Vector2.Distance(
                    touch0.position, 
                    touch1.position
                );
                
                float scaleFactor = currentDistance / initialDistance;
                transform.localScale = initialScale * scaleFactor;
            }
        }
    }
}
```

## 输入系统优化

### 输入缓冲
```csharp
public class InputBuffer : MonoBehaviour
{
    private Queue<InputCommand> inputBuffer = new Queue<InputCommand>();
    private float bufferTime = 0.2f;
    
    public void BufferInput(InputCommand command)
    {
        command.timestamp = Time.time;
        inputBuffer.Enqueue(command);
        
        // 清理过期输入
        while (inputBuffer.Count > 0 && 
               Time.time - inputBuffer.Peek().timestamp > bufferTime)
        {
            inputBuffer.Dequeue();
        }
    }
    
    public bool CheckBufferedInput(string commandType)
    {
        foreach (var command in inputBuffer)
        {
            if (command.type == commandType)
            {
                return true;
            }
        }
        return false;
    }
}
```

### 输入预测
```csharp
public class InputPrediction : MonoBehaviour
{
    private Vector3 previousInput;
    private Vector3 predictedInput;
    
    void Update()
    {
        Vector3 currentInput = new Vector3(
            Input.GetAxis("Horizontal"),
            0,
            Input.GetAxis("Vertical")
        );
        
        // 预测下一帧输入
        predictedInput = currentInput + (currentInput - previousInput);
        previousInput = currentInput;
        
        // 使用预测值进行平滑移动
        Vector3 smoothedInput = Vector3.Lerp(
            currentInput, 
            predictedInput, 
            0.5f
        );
        Move(smoothedInput);
    }
}
