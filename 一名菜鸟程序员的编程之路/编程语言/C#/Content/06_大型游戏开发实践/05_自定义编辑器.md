```csharp
using UnityEditor;
using UnityEngine;

[CustomEditor(typeof(MyComponent))]
public class MyComponentEditor : Editor
{
    public override void OnInspectorGUI()
    {
        // 获取目标组件
        MyComponent myTarget = (MyComponent)target;
        
        // 开始检查是否有修改
        EditorGUI.BeginChangeCheck();
        
        // 绘制默认Inspector
        DrawDefaultInspector();
        
        // 自定义字段
        myTarget.customField = EditorGUILayout.TextField("自定义字段", myTarget.customField);
        
        // 如果有修改，标记为脏
        if (EditorGUI.EndChangeCheck())
        {
            EditorUtility.SetDirty(target);
        }
    }
}
```

### 2.2 自定义属性绘制器

```csharp
[CustomPropertyDrawer(typeof(MyCustomType))]
public class MyCustomPropertyDrawer : PropertyDrawer
{
    public override void OnGUI(Rect position, SerializedProperty property, GUIContent label)
    {
        // 开始属性
        EditorGUI.BeginProperty(position, label, property);
        
        // 创建缩进级别
        int indent = EditorGUI.indentLevel;
        EditorGUI.indentLevel = 0;
        
        // 计算每个字段的矩形
        Rect fieldRect = new Rect(position.x, position.y, position.width, EditorGUIUtility.singleLineHeight);
        
        // 绘制字段
        EditorGUI.PropertyField(fieldRect, property.FindPropertyRelative("fieldName"));
        
        // 恢复缩进
        EditorGUI.indentLevel = indent;
        
        // 结束属性
        EditorGUI.EndProperty();
    }
    
    public override float GetPropertyHeight(SerializedProperty property, GUIContent label)
    {
        return EditorGUIUtility.singleLineHeight;
    }
}
```

```csharp
[CustomEditor(typeof(MyComponent))]
public class MyComponentSceneEditor : Editor
{
    void OnSceneGUI()
    {
        MyComponent myTarget = (MyComponent)target;
        
        // 绘制手柄
        EditorGUI.BeginChangeCheck();
        Vector3 newPosition = Handles.PositionHandle(myTarget.transform.position, Quaternion.identity);

        {
            Undo.RecordObject(myTarget.transform, "Move Object");
            myTarget.transform.position = newPosition;
        }
        
        // 绘制标签
        Handles.Label(myTarget.transform.position + Vector3.up, "我的对象");
        
        // 绘制自定义控件
        Handles.color = Color.yellow;
        Handles.DrawWireDisc(myTarget.transform.position, Vector3.up, 1.0f);
    }
}
```

### 5.2 自定义资源类型

```csharp
// 自定义资源类型
public class CustomScriptableObject : ScriptableObject
{
    public string description;
    public float value;
    public GameObject prefab;
    
    // 创建资源实例的菜单项
    [MenuItem("Assets/Create/自定义资源")]
    public static void CreateAsset()
    {
        var asset = CreateInstance<CustomScriptableObject>();
        
        // 创建资源文件
        string path = AssetDatabase.GetAssetPath(Selection.activeObject);
        if (string.IsNullOrEmpty(path))
        {
            path = "Assets";
        }
        else if (Path.GetExtension(path) != "")
        {
            path = path.Replace(Path.GetFileName(path), "");
        }
        
        string assetPathAndName = AssetDatabase.GenerateUniqueAssetPath(path + "/New CustomAsset.asset");
        AssetDatabase.CreateAsset(asset, assetPathAndName);
        
        AssetDatabase.SaveAssets();
        EditorUtility.FocusProjectWindow();
        Selection.activeObject = asset;
    }
}
```

### 6.2 预制体工具

```csharp
public class PrefabTools : EditorWindow
{
    [MenuItem("Tools/预制体工具")]
    static void Init()
    {
        GetWindow<PrefabTools>("预制体工具");
    }
    
    void OnGUI()
    {
        if (GUILayout.Button("更新选中的预制体"))
        {
            UpdateSelectedPrefabs();
        }
        
        if (GUILayout.Button("查找缺失引用"))
        {
            FindMissingReferences();
        }
    }
    
    void UpdateSelectedPrefabs()
    {
        GameObject[] selection = Selection.gameObjects;
        
        foreach (GameObject go in selection)
        {
            if (PrefabUtility.IsPartOfPrefabInstance(go))
            {
                // 应用预制体修改
                PrefabUtility.ApplyPrefabInstance(go, InteractionMode.UserAction);
            }
        }
    }
    
    void FindMissingReferences()
    {
        GameObject[] selection = Selection.gameObjects;
        
        foreach (GameObject go in selection)
        {
            Component[] components = go.GetComponents<Component>();
            
            foreach (Component component in components)
            {
                if (component == null)
                {
                    Debug.LogError($"发现缺失组件在物体: {go.name}");
                    continue;
                }
                
                SerializedObject serializedObject = new SerializedObject(component);
                SerializedProperty property = serializedObject.GetIterator();
                
                while (property.Next(true))
                {
                    if (property.propertyType == SerializedPropertyType.ObjectReference &&
                        property.objectReferenceValue == null &&
                        property.objectReferenceInstanceIDValue != 0)
                    {
                        Debug.LogError($"发现缺失引用在组件: {component.GetType().Name} 物体: {go.name}");
                    }
                }
            }
        }
    }
}
```

### 7.2 编辑器协程

```csharp
public class EditorCoroutineExample : EditorWindow
{
    IEnumerator ProcessItemsCoroutine()
    {
        List<Item> items = GetItems();
        
        for (int i = 0; i < items.Count; i++)
        {
            // 处理当前项
            ProcessItem(items[i]);
            
            // 更新进度条
            EditorUtility.DisplayProgressBar("处理中", 
                $"正在处理项目 {i + 1}/{items.Count}",
                (float)i / items.Count);
            
            // 每处理10个项目暂停一帧
            if (i % 10 == 0)
            {
                yield return null;
            }
        }
        
        EditorUtility.ClearProgressBar();
    }
    
    void StartProcessing()
    {
        EditorCoroutineUtility.StartCoroutine(ProcessItemsCoroutine(), this);
    }
}
```

```csharp
        foreach (GameObject obj in allObjects)
        {
            Component[] components = obj.GetComponents<Component>();
            
            foreach (Component component in components)
            {
                if (component == null)
                {
                    Debug.LogError($"空组件在物体: {obj.name}", obj);
                }
            }
        }
    }
}
```

### 8.2 编辑器测试

```csharp
public class EditorTests
{
    [MenuItem("Tools/测试/运行所有测试")]
    static void RunAllTests()
    {
        // 运行编辑器测试
        var testAssembly = Assembly.GetExecutingAssembly();
        var testTypes = testAssembly.GetTypes()
            .Where(t => t.GetCustomAttributes(typeof(TestFixtureAttribute), true).Length > 0);
            
        foreach (var type in testTypes)
        {
            RunTestsForType(type);
        }
    }
    
    static void RunTestsForType(Type type)
    {
        var methods = type.GetMethods()
            .Where(m => m.GetCustomAttributes(typeof(TestAttribute), true).Length > 0);
            
        object instance = Activator.CreateInstance(type);
        
        foreach (var method in methods)
        {
            try
            {
                method.Invoke(instance, null);
                Debug.Log($"测试通过: {type.Name}.{method.Name}");
            }
            catch (Exception e)
            {
                Debug.LogError($"测试失败: {type.Name}.{method.Name}\n{e.Message}");
            }
        }
    }
}
```
