# 学习笔记：位运算

```c
#include <stdio.h>

// 学习笔记：位运算

int main() {
    // 位运算操作简介
    // 位运算可以对二进制数的每一位进行操作，包括按位与、按位或、按位异或、按位取反、左移和右移。
    // 这些操作常用于代码优化、加密算法、图像处理等领域。

    // a. 按位与（&）
    // 对两个二进制数的每一位进行与运算，只有当两个数的对应位都为1时，结果的对应位才为1，否则为0。
    // 示例：1010 & 1100 = 1000
    unsigned char a = 0b1010;
    unsigned char b = 0b1100;
    unsigned char result_and = a & b;
    printf("按位与操作:\n");
    printf("0b1010 & 0b1100 = 0b%04b\n", result_and);

    // b. 按位或（|）
    // 对两个二进制数的每一位进行或运算，只有当两个数的对应位都为0时，结果的对应位才为0，否则为1。
    // 示例：1010 | 1100 = 1110
    unsigned char result_or = a | b;
    printf("\n按位或操作:\n");
    printf("0b1010 | 0b1100 = 0b%04b\n", result_or);

    // c. 按位异或（^）
    // 对两个二进制数的每一位进行异或运算，只有当两个数的对应位不同时，结果的对应位才为1，否则为0。
    // 示例：1010 ^ 1100 = 0110
    unsigned char result_xor = a ^ b;
    printf("\n按位异或操作:\n");
    printf("0b1010 ^ 0b1100 = 0b%04b\n", result_xor);

    // d. 按位取反（~）
    // 对一个二进制数的每一位进行取反运算，即0变成1，1变成0。
    // 示例：~1010 = 0101（假设8位数）
    unsigned char result_not = ~a;
    printf("\n按位取反操作:\n");
    printf("~0b1010 = 0b%04b\n", result_not & 0x0F); // 只显示低4位

    // e. 左移（<<）
    // 将一个二进制数的所有位向左移动指定的位数，右侧空出的位用0填充。
    // 示例：1010 << 2 = 101000
    unsigned char result_left_shift = a << 2;
    printf("\n左移操作:\n");
    printf("0b1010 << 2 = 0b%06b\n", result_left_shift); // 结果需要6位显示

    // f. 右移（>>）
    // 将一个二进制数的所有位向右移动指定的位数，左侧空出的位用符号位填充（对于有符号数）或0填充（对于无符号数）。
    // 示例：1010 >> 2 = 0010
    unsigned char result_right_shift = a >> 2;
    printf("\n右移操作:\n");
    printf("0b1010 >> 2 = 0b%04b\n", result_right_shift);

    // g. 注意事项
    // - 左移和右移的位数不能超过数据类型的位数，否则会导致数据溢出或未定义行为。
    // - 例如，对于8位二进制数，左移9位或右移9位的结果是未定义的。
    // - 在进行位移运算时，要注意数据类型的位数和移动的位数，以避免数据溢出和错误结果。

    return 0;
}
