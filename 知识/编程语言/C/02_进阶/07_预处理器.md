# C语言预处理器

## 预处理器指令

### 常用指令
```c
#include    // 包含头文件
#define     // 定义宏
#undef      // 取消宏定义
#ifdef      // 条件编译
#ifndef     // 条件编译
#if         // 条件编译
#else       // 条件编译
#elif       // 条件编译
#endif      // 条件编译
#pragma     // 编译器指令
```

## 宏定义

### 常量宏
```c
#define PI 3.14159
#define MAX_SIZE 100
#define BUFFER_SIZE (1024 * 4)

// 使用示例
double area = PI * radius * radius;
```

### 函数宏
```c
#define MAX(a, b) ((a) > (b) ? (a) : (b))
#define MIN(a, b) ((a) < (b) ? (a) : (b))

// 使用括号避免运算符优先级问题
#define SQUARE(x) ((x) * (x))
```

## 条件编译

### 基本用法
```c
#ifdef DEBUG
    printf("Debug: value = %d\n", x);
#endif

#ifndef BUFFER_SIZE
    #define BUFFER_SIZE 1024
#endif

#if defined(WIN32)
    // Windows特定代码
#elif defined(LINUX)
    // Linux特定代码
#else
    // 其他平台代码
#endif
```

### 防止重复包含
```c
#ifndef HEADER_H
#define HEADER_H

// 头文件内容

#endif /* HEADER_H */
```

## 预定义宏

### 标准预定义宏
```c
__FILE__    // 当前文件名
__LINE__    // 当前行号
__DATE__    // 编译日期
__TIME__    // 编译时间
__STDC__    // 是否符合ANSI C
__func__    // 当前函数名（C99）
```

### 使用示例
```c
#define LOG_ERROR(msg) \
    fprintf(stderr, "[Error] %s:%d: %s\n", \
            __FILE__, __LINE__, msg)

// 使用
LOG_ERROR("内存分配失败");
```

## 高级预处理技术

### 字符串化运算符 #
```c
#define PRINT_VAR(var) \
    printf(#var " = %d\n", var)

// 使用
int x = 42;
PRINT_VAR(x);  // 输出：x = 42
```

### 标记连接运算符 ##
```c
#define CONCAT(a, b) a##b

// 使用
int var12 = 100;
printf("%d\n", CONCAT(var, 12));  // 访问var12
```

### 可变参数宏
```c
#define DEBUG_LOG(fmt, ...) \
    fprintf(stderr, "[Debug] "fmt"\n", ##__VA_ARGS__)

// 使用
DEBUG_LOG("Count: %d", count);
DEBUG_LOG("Ready");
```

## 条件编译技巧

### 1. 调试控制
```c
#define DEBUG_LEVEL 2

#if DEBUG_LEVEL >= 1
    #define DEBUG1(fmt, ...) printf(fmt, ##__VA_ARGS__)
#else
    #define DEBUG1(fmt, ...) ((void)0)
#endif

#if DEBUG_LEVEL >= 2
    #define DEBUG2(fmt, ...) printf(fmt, ##__VA_ARGS__)
#else
    #define DEBUG2(fmt, ...) ((void)0)
#endif
```

### 2. 平台特定代码
```c
#if defined(_WIN32)
    #define PATH_SEPARATOR '\\'
#else
    #define PATH_SEPARATOR '/'
#endif

// 使用
char path[256];
sprintf(path, "folder%cfile.txt", PATH_SEPARATOR);
```

## 最佳实践

### 1. 宏命名约定
```c
// 常量宏全大写
#define MAX_BUFFER_SIZE 1024

// 函数宏使用大写或前缀
#define Abs(x) ((x) < 0 ? -(x) : (x))
#define SAFE_FREE(p) do { free(p); p = NULL; } while(0)
```

### 2. 安全的函数宏
```c
// 使用do-while(0)包装多语句宏
#define SAFE_DELETE(p) do { \
    if (p) { \
        delete p; \
        p = NULL; \
    } \
} while(0)
```

### 3. 条件编译的管理
```c
// 特性开关
#define FEATURE_ENABLED 1

#if FEATURE_ENABLED
    // 特性实现
#else
    // 替代实现
#endif
```
