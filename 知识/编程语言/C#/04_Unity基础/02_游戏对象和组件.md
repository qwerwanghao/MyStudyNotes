# 游戏对象和组件

## 游戏对象基础

### 创建和管理
```csharp
public class GameObjectManager : MonoBehaviour
{
    // 创建游戏对象
    void CreateGameObject()
    {
        // 创建空对象
        GameObject emptyObject = new GameObject("EmptyObject");
        
        // 创建原始体
        GameObject cube = GameObject.CreatePrimitive(PrimitiveType.Cube);
        GameObject sphere = GameObject.CreatePrimitive(PrimitiveType.Sphere);
        
        // 设置父子关系
        sphere.transform.SetParent(cube.transform);
        
        // 查找对象
        GameObject foundObject = GameObject.Find("ObjectName");
        GameObject taggedObject = GameObject.FindWithTag("Player");
    }
    
    // 销毁游戏对象
    void DestroyGameObject(GameObject obj)
    {
        // 立即销毁
        DestroyImmediate(obj);
        
        // 延迟销毁
        Destroy(obj, 2f);
    }
}
```

### 对象层级
```csharp
public class HierarchyManager : MonoBehaviour
{
    // 遍历子对象
    void TraverseChildren(Transform parent)
    {
        foreach (Transform child in parent)
        {
            Debug.Log($"Found child: {child.name}");
            TraverseChildren(child);
        }
    }
    
    // 查找特定层级对象
    Transform FindInChildren(Transform parent, string name)
    {
        if (parent.name == name)
            return parent;
            
        foreach (Transform child in parent)
        {
            Transform found = FindInChildren(child, name);
            if (found != null)
                return found;
        }
        
        return null;
    }
}
```

## 组件系统

### 基本组件
1. **Transform组件**
```csharp
public class TransformExample : MonoBehaviour
{
    void ManipulateTransform()
    {
        // 位置
        transform.position = new Vector3(1, 2, 3);
        transform.localPosition = Vector3.zero;
        
        // 旋转
        transform.rotation = Quaternion.Euler(0, 90, 0);
        transform.Rotate(Vector3.up * 45f);
        
        // 缩放
        transform.localScale = new Vector3(2, 2, 2);
    }
}
```

2. **Renderer组件**
```csharp
public class RendererExample : MonoBehaviour

```

3. **Collider组件**
```csharp
public class ColliderExample : MonoBehaviour
{
    private BoxCollider boxCollider;
    
    void Start()
    {
        boxCollider = gameObject.AddComponent<BoxCollider>();
        
        // 设置碰撞体大小
        boxCollider.size = new Vector3(2, 1, 1);
        boxCollider.center = Vector3.zero;
        
        // 设置触发器
        boxCollider.isTrigger = true;
    }
    
    void OnTriggerEnter(Collider other)
    {
        Debug.Log($"Triggered with: {other.gameObject.name}");
    }
}
```

## 组件通信

### 直接引用
```csharp
public class ComponentCommunication : MonoBehaviour
{
    // 直接引用
    public OtherComponent otherComponent;
    
    // 通过Inspector拖拽赋值
    [SerializeField]
    private GameObject targetObject;
    
    void Start()
    {
        // 获取组件
        otherComponent = GetComponent<OtherComponent>();

        OtherComponent[] childComponents = 
            GetComponentsInChildren<OtherComponent>();
    }
}
```

### 事件系统
```csharp
public class EventBasedComponent : MonoBehaviour
{
    // 定义事件
    public event System.Action onValueChanged;
    public event System.Action<float> onHealthChanged;
    
    private float health;
    
    public void TakeDamage(float damage)
    {
        health -= damage;
        
        // 触发事件
        onValueChanged?.Invoke();
        onHealthChanged?.Invoke(health);
    }
}

public class EventListener : MonoBehaviour
{
    private EventBasedComponent eventComponent;
    
    void Start()
    {
        eventComponent = GetComponent<EventBasedComponent>();
        
        // 订阅事件
        eventComponent.onValueChanged += OnValueChanged;
        eventComponent.onHealthChanged += OnHealthChanged;
    }
    
    void OnValueChanged()
    {
        Debug.Log("Value changed!");
    }
    
    void OnHealthChanged(float newHealth)
    {
        Debug.Log($"Health changed to: {newHealth}");
    }
    
    void OnDestroy()
    {
        // 取消订阅
        if (eventComponent != null)
        {
            eventComponent.onValueChanged -= OnValueChanged;
            eventComponent.onHealthChanged -= OnHealthChanged;
        }
    }
}
```

## 组件优化

### 性能考虑
```csharp
public class OptimizedComponent : MonoBehaviour
{
    // 缓存组件引用
    private Transform cachedTransform;
    private Rigidbody cachedRigidbody;
    
    void Awake()
    {
        // 在Awake中获取组件引用
        cachedTransform = transform;
        cachedRigidbody = GetComponent<Rigidbody>();
    }
    
    // 使用缓存的引用
    void Update()
    {
        cachedTransform.Translate(Vector3.forward * Time.deltaTime);
        
        if (cachedRigidbody != null)
        {
            cachedRigidbody.AddForce(Vector3.up);
        }
    }
}
```
