# 场景管理

## 基本场景操作

### 场景加载
```csharp
public class SceneLoader : MonoBehaviour
{
    // 同步加载场景
    public void LoadScene(string sceneName)
    {
        SceneManager.LoadScene(sceneName);
    }
    
    // 异步加载场景
    public async Task LoadSceneAsync(string sceneName)
    {
        var operation = SceneManager.LoadSceneAsync(sceneName);
        operation.allowSceneActivation = false;
        
        while (operation.progress < 0.9f)
        {
            // 更新加载进度
            float progress = Mathf.Clamp01(operation.progress / 0.9f);
            UpdateLoadingProgress(progress);
            await Task.Yield();
        }
        
        // 完成加载
        operation.allowSceneActivation = true;
        while (!operation.isDone)
        {
            await Task.Yield();
        }
    }
    
    // 加载多个场景
    public IEnumerator LoadMultipleScenes(string[] sceneNames)
    {
        foreach (string sceneName in sceneNames)
        {
            yield return SceneManager.LoadSceneAsync(
                sceneName, 
                LoadSceneMode.Additive
            );
        }
    }
}
```

### 场景切换
```csharp
public class SceneTransition : MonoBehaviour
{
    // 带淡入淡出效果的场景切换
    public IEnumerator TransitionToScene(string sceneName)
    {
        // 淡出
        yield return FadeOut();
        
        // 加载新场景
        var operation = SceneManager.LoadSceneAsync(sceneName);
        while (!operation.isDone)
        {
            yield return null;
        }
        
        // 淡入
        yield return FadeIn();
    }
    
    // 带进度条的场景切换
    public IEnumerator LoadSceneWithProgress(string sceneName)
    {
        // 显示加载界面
        ShowLoadingUI();
        
        // 开始异步加载
        var operation = SceneManager.LoadSceneAsync(sceneName);
        operation.allowSceneActivation = false;
        
        while (operation.progress < 0.9f)
        {
            UpdateProgressBar(operation.progress);
            yield return null;
        }
        
        // 等待用户输入
        while (!Input.GetKeyDown(KeyCode.Space))
        {
            yield return null;
        }
        
        operation.allowSceneActivation = true;
        HideLoadingUI();
    }
}
```

## 场景管理系统

### 场景状态管理
```csharp
public class SceneManager : MonoBehaviour
{
    private Dictionary<string, Scene> loadedScenes = 
        new Dictionary<string, Scene>();
    
    // 场景状态
    [System.Serializable]
    public class SceneState
    {
        public string sceneName;
        public bool isLoaded;
        public bool isActive;
        public float loadProgress;
    }
    
    // 加载和管理场景
    public async Task<bool> LoadAndManageScene(string sceneName)
    {
        try
        {
            var operation = UnityEngine.SceneManagement.SceneManager
                .LoadSceneAsync(sceneName, LoadSceneMode.Additive);
                
            while (!operation.isDone)
            {
                UpdateSceneState(sceneName, operation.progress);
                await Task.Yield();
            }
            
            var scene = UnityEngine.SceneManagement.SceneManager
                .GetSceneByName(sceneName);
            loadedScenes[sceneName] = scene;
            
            return true;
        }
        catch (Exception e)
        {
            Debug.LogError($"Failed to load scene {sceneName}: {e}");
            return false;
        }
    }
    
    // 卸载场景
    public async Task UnloadScene(string sceneName)
    {
        if (loadedScenes.ContainsKey(sceneName))
        {
            var operation = UnityEngine.SceneManagement.SceneManager
                .UnloadSceneAsync(sceneName);
                
            while (!operation.isDone)
            {
                await Task.Yield();
            }
            
            loadedScenes.Remove(sceneName);
        }
    }
}
```

### 场景依赖管理
```csharp
public class SceneDependencyManager : MonoBehaviour
{
    [System.Serializable]
    public class SceneDependency
    {
        public string mainScene;
        public string[] dependentScenes;
    }
    
    public List<SceneDependency> sceneDependencies;
    
    // 加载场景及其依赖
    public async Task LoadSceneWithDependencies(string mainScene)
    {
        var dependency = sceneDependencies.Find(x => x.mainScene == mainScene);
        if (dependency != null)
        {
            // 先加载依赖场景
            foreach (var depScene in dependency.dependentScenes)
            {
                await LoadSceneAsync(depScene, LoadSceneMode.Additive);
            }
            
            // 加载主场景
            await LoadSceneAsync(mainScene, LoadSceneMode.Additive);
        }
    }
    
    // 卸载场景及其依赖
    public async Task UnloadSceneWithDependencies(string mainScene)
    {
        var dependency = sceneDependencies.Find(x => x.mainScene == mainScene);
        if (dependency != null)
        {
            await UnloadSceneAsync(mainScene);
            
            foreach (var depScene in dependency.dependentScenes)
            {
                await UnloadSceneAsync(depScene);
            }
        }
    }
}
```

## 场景优化

### 场景加载优化
```csharp
public class SceneLoadOptimizer : MonoBehaviour
{
    // 场景预加载
    private Dictionary<string, AsyncOperation> preloadOperations = 
        new Dictionary<string, AsyncOperation>();
        
    public void PreloadScene(string sceneName)
    {
        if (!preloadOperations.ContainsKey(sceneName))
        {
            var operation = SceneManager.LoadSceneAsync(
                sceneName, 
                LoadSceneMode.Additive
            );
            operation.allowSceneActivation = false;
            preloadOperations[sceneName] = operation;
        }
    }
    
    // 激活预加载的场景
    public void ActivatePreloadedScene(string sceneName)
    {
        if (preloadOperations.TryGetValue(sceneName, out var operation))
        {
            operation.allowSceneActivation = true;
            preloadOperations.Remove(sceneName);
        }
    }
    
    // 分帧加载场景对象
    public IEnumerator LoadSceneObjectsProgressively(Scene scene)
    {
        var rootObjects = scene.GetRootGameObjects();
        foreach (var obj in rootObjects)
        {
            obj.SetActive(true);
            if (Time.frameCount % 2 == 0)
            {
                yield return null;
            }
        }
    }
}
```

### 场景内存管理
```csharp
public class SceneMemoryManager : MonoBehaviour
{
    // 场景资源清理
    public void CleanupScene()
    {
        Resources.UnloadUnusedAssets();
        GC.Collect();
    }
    
    // 场景对象池化
    private Dictionary<string, Queue<GameObject>> sceneObjectPools = 
        new Dictionary<string, Queue<GameObject>>();
        
    public void PoolSceneObject(GameObject obj)
    {
        string key = obj.name;
        if (!sceneObjectPools.ContainsKey(key))
        {
            sceneObjectPools[key] = new Queue<GameObject>();
        }
        
        obj.SetActive(false);
        sceneObjectPools[key].Enqueue(obj);
    }
    
    public GameObject GetPooledSceneObject(string key)
    {
        if (sceneObjectPools.TryGetValue(key, out var pool) && pool.Count > 0)
        {
            var obj = pool.Dequeue();
            obj.SetActive(true);
            return obj;
        }
        return null;
    }
}
```

### 场景引用管理
```csharp
public class SceneReferenceManager : MonoBehaviour
{
    // 场景引用缓存
    private Dictionary<string, WeakReference> sceneReferences = 
        new Dictionary<string, WeakReference>();
        
    // 缓存场景引用
    public void CacheSceneReference(string key, UnityEngine.Object obj)
    {
        sceneReferences[key] = new WeakReference(obj);
    }
    
    // 获取场景引用
    public UnityEngine.Object GetSceneReference(string key)
    {
        if (sceneReferences.TryGetValue(key, out var weakRef))
        {
            if (weakRef.IsAlive)
            {
                return weakRef.Target as UnityEngine.Object;
            }
            sceneReferences.Remove(key);
        }
        return null;
    }
    
    // 清理失效引用
    public void CleanupDeadReferences()
    {
        var deadKeys = sceneReferences.Keys
            .Where(key => !sceneReferences[key].IsAlive)
            .ToList();
            
        foreach (var key in deadKeys)
        {
            sceneReferences.Remove(key);
        }
    }
}
