# 接口和抽象类

## 接口基础

### 接口定义
```csharp
public interface IDestroyable
{
    // 属性
    float Health { get; set; }
    bool IsDestroyed { get; }
    
    // 方法
    void TakeDamage(float damage);
    void Repair(float amount);
    
    // C# 8.0后可以有默认实现
    void OnDestroyed() 
    {
        Console.WriteLine("Object destroyed");
    }
}
```

### 接口实现
```csharp
public class Crate : MonoBehaviour, IDestroyable
{
    private float currentHealth;
    
    public float Health 
    { 
        get => currentHealth;
        set => currentHealth = Mathf.Clamp(value, 0, 100);
    }
    
    public bool IsDestroyed => currentHealth <= 0;
    
    public void TakeDamage(float damage)
    {
        Health -= damage;
        if (IsDestroyed)
        {
            OnDestroyed();
        }
    }
    
    public void Repair(float amount)
    {
        Health += amount;
    }
}
```

## 抽象类

### 抽象类定义
```csharp
public abstract class Weapon
{
    // 字段
    protected float damage;
    protected float range;
    
    // 普通方法
    public virtual void Initialize(float damage, float range)
    {
        this.damage = damage;
        this.range = range;
    }
    
    // 抽象方法
    public abstract void Attack();
    
    // 抽象属性
    public abstract float DamagePerSecond { get; }
}
```

### 抽象类实现
```csharp
public class Sword : Weapon
{
    private float attackSpeed;
    
    public override float DamagePerSecond => damage * attackSpeed;
    
    public override void Attack()
    {
        // 近战攻击实现
        PerformMeleeAttack();
    }
    
    private void PerformMeleeAttack()
    {
        // 具体的攻击逻辑
    }
}
```

## 接口vs抽象类

### 主要区别
1. **结构区别**
   ```csharp
   // 接口 - 只定义契约
   public interface IMoveable
   {
       void Move(Vector3 direction);
   }
   
   // 抽象类 - 可以包含实现
   public abstract class MovableEntity
   {
       protected float speed;
       public abstract void Move(Vector3 direction);
       protected virtual void UpdatePosition(Vector3 position)
       {
           transform.position = position;
       }
   }
   ```

2. **使用场景**
   ```csharp
   // 接口用于定义能力
   public interface IInteractable
   {
       void Interact();
   }
   
   // 抽象类用于提供基础实现
   public abstract class InteractableObject : MonoBehaviour
   {
       protected bool canInteract = true;
       
       protected virtual bool CanInteract()
       {
           return canInteract;
       }
   }
   ```

## Unity中的应用

### 组件接口
```csharp

public interface IDamageable
{
    void TakeDamage(float damage);
    float CurrentHealth { get; }
}

public interface IHealable
{
    void Heal(float amount);
}

// 实现多个接口
public class Player : MonoBehaviour, IDamageable, IHealable

        health -= damage;

}

## 设计模式实践

### 策略模式
```csharp
// 定义策略接口
public interface IAttackStrategy
{
    void Execute(Transform source, Transform target);
}

// 具体策略实现
public class MeleeAttackStrategy : IAttackStrategy
{
    public void Execute(Transform source, Transform target)
    {
        // 近战攻击实现
    }
}

public class RangedAttackStrategy : IAttackStrategy
{
    public void Execute(Transform source, Transform target)
    {
        // 远程攻击实现
    }
}

// 使用策略
public class Enemy : MonoBehaviour
{
    private IAttackStrategy attackStrategy;
    
    public void SetStrategy(IAttackStrategy strategy)
    {
        attackStrategy = strategy;
    }
    
    public void Attack(Transform target)
    {
        attackStrategy?.Execute(transform, target);
    }
}
```

### 观察者模式
```csharp
// 事件接口
public interface IObserver
{
    void OnNotify(object data);
}

// 被观察者接口
public interface ISubject
{
    void AddObserver(IObserver observer);
    void RemoveObserver(IObserver observer);
    void NotifyObservers(object data);
}

// 具体实现
public class HealthSystem : MonoBehaviour, ISubject
{
    private List<IObserver> observers = new List<IObserver>();
    
    public void AddObserver(IObserver observer)
    {
        observers.Add(observer);
    }
    
    public void RemoveObserver(IObserver observer)
    {
        observers.Remove(observer);
    }
    
    public void NotifyObservers(object data)
    {
        foreach (var observer in observers)
        {
            observer.OnNotify(data);
        }
    }
}
```

## 最佳实践

### 接口设计原则
1. **接口隔离原则**
   ```csharp
   // 好的设计 - 小而专注的接口
   public interface IMoveable
   {
       void Move(Vector3 direction);
   }
   
   public interface IJumpable
   {
       void Jump();
   }
   
   // 避免 - 大而全的接口
   public interface ICharacterActions
   {
       void Move(Vector3 direction);
       void Jump();
       void Attack();
       void Defend();
   }
   ```

2. **依赖注入**
   ```csharp
   public class WeaponController
   {
       private readonly IWeapon weapon;
       
       public WeaponController(IWeapon weapon)
       {
           this.weapon = weapon;
       }
       
       public void Attack()
       {
           weapon.Fire();
       }
   }
   ```

### 抽象类设计原则
```csharp
// 抽象程度要适中
public abstract class AIController : MonoBehaviour
{
    // 提供通用实现
    protected virtual void Initialize()
    {
        SetupPathing();
        InitializeBehavior();
    }
    
    // 必须由派生类实现的核心行为
    protected abstract void UpdateAI();
    
    // 可选的行为重写
    protected virtual void OnTargetDetected()
    {
        // 默认实现
    }
}
