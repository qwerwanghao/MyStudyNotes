# 属性和索引器

## 属性基础

### 自动属性
```csharp
public class Player
{
    // 基本自动属性
    public string Name { get; set; }
    
    // 只读自动属性
    public string Id { get; }
    
    // 带默认值的自动属性
    public int Level { get; set; } = 1;
    
    // 私有set的自动属性
    public float Experience { get; private set; }
}
```

### 完整属性
```csharp
public class Character
{
    private float health;
    private float maxHealth = 100f;
    
    public float Health
    {
        get { return health; }
        set 
        {
            health = Mathf.Clamp(value, 0f, maxHealth);
            OnHealthChanged?.Invoke(health);
        }
    }
    
    public float HealthPercentage
    {
        get { return health / maxHealth * 100f; }
    }
}
```

## 属性高级特性

### 计算属性
```csharp
public class Weapon
{
    public float BaseDamage { get; set; }
    public float DamageMultiplier { get; set; }
    
    // 计算属性
    public float TotalDamage
    {
        get => BaseDamage * DamageMultiplier;
    }
    
    // 带缓存的计算属性
    private float? cachedDPS;
    public float DPS
    {
        get
        {
            if (!cachedDPS.HasValue)
            {
                cachedDPS = CalculateDPS();
            }
            return cachedDPS.Value;
        }
    }
}
```

### 属性表达式体
```csharp
public class GameState
{
    private bool isGameOver;
    private float gameTime;
    
    // 表达式体属性
    public bool IsComplete => isGameOver;
    public bool IsRunning => !isGameOver;
    public float GameTimeInMinutes => gameTime / 60f;
    
    // 带set的表达式体属性
    public float GameTime
    {
        get => gameTime;
        set => gameTime = value >= 0 ? value : 0;
    }
}
```

## 索引器

### 基本索引器
```csharp
public class Inventory
{
    private Item[] items = new Item[10];
    
    // 基本索引器
    public Item this[int index]
    {
        get
        {
            if (index < 0 || index >= items.Length)
                throw new IndexOutOfRangeException();
            return items[index];
        }
        set
        {
            if (index < 0 || index >= items.Length)
                throw new IndexOutOfRangeException();
            items[index] = value;
        }
    }
}
```

### 多参数索引器
```csharp
public class GameGrid
{
    private GameObject[,] grid;
    
    public GameGrid(int width, int height)
    {
        grid = new GameObject[width, height];
    }
    
    // 二维索引器
    public GameObject this[int x, int y]
    {
        get => grid[x, y];
        set => grid[x, y] = value;
    }
}
```

## Unity中的应用

### 序列化属性
```csharp
public class PlayerController : MonoBehaviour
{
    // Unity序列化的属性
    [SerializeField]
    private float moveSpeed = 5f;
    
    // 在Inspector中显示但不可编辑
    [SerializeField]
    private float health;
    public float Health 
    { 
        get => health;
        private set
        {
            health = value;
            OnHealthChanged?.Invoke(health);
        }
    }
    
    // 自定义Range属性
    [Range(0f, 100f)]
    public float stamina;
}
```

### 属性绑定
```csharp
public class UIController : MonoBehaviour
{
    private Text healthText;
    private Slider healthSlider;
    
    // 属性变化通知
    public float CurrentHealth
    {
        get => currentHealth;
        set
        {
            if (currentHealth != value)
            {
                currentHealth = value;
                UpdateUI();
            }
        }
    }
    
    private void UpdateUI()
    {
        healthText.text = $"Health: {CurrentHealth}";
        healthSlider.value = CurrentHealth / maxHealth;
    }
}
```

## 最佳实践

### 属性设计
```csharp
public class GameManager
{
    // 1. 简单属性使用自动属性
    public string GameVersion { get; }
    
    // 2. 需要验证的使用完整属性
    private int score;
    public int Score
    {
        get => score;
        set
        {
            if (value < 0) return;
            score = value;
            OnScoreChanged?.Invoke(score);
        }
    }
    
    // 3. 只读属性使用表达式体
    public bool IsGameActive => CurrentState == GameState.Playing;
    
    // 4. 计算属性使用缓存
    private float? cachedAverageScore;
    public float AverageScore
    {
        get
        {
            cachedAverageScore ??= CalculateAverageScore();
            return cachedAverageScore.Value;
        }
    }
}
```

### 索引器设计
```csharp
public class ObjectPool<T> where T : Component
{
    private List<T> activeObjects = new List<T>();
    
    // 1. 提供边界检查
    public T this[int index]
    {
        get
        {
            if (index < 0 || index >= activeObjects.Count)
                throw new ArgumentOutOfRangeException(nameof(index));
            return activeObjects[index];
        }
    }
    
    // 2. 提供多种访问方式
    public T this[string key]
    {
        get => objectDictionary.TryGetValue(key, out var obj) ? obj : null;
    }
}
```

### 性能考虑
```csharp
public class PerformanceCriticalComponent
{
    // 1. 避免在属性中进行耗时操作
    public Vector3 Position
    {
        get => transform.position;  // 简单获取
        set
        {
            transform.position = value;  // 简单设置
            // 不要在这里进行复杂计算
        }
    }
    
    // 2. 使用方法代替复杂属性
    public Vector3 CalculateNextPosition()
    {
        // 复杂的位置计算
        return Vector3.zero;
    }
    
    // 3. 缓存频繁访问的计算结果
    private float? cachedDistance;
    public float DistanceToTarget
    {
        get
        {
            if (!cachedDistance.HasValue)
                cachedDistance = CalculateDistance();
            return cachedDistance.Value;
        }
    }
}
