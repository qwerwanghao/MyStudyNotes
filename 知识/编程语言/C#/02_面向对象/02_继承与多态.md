# 继承与多态

## 继承基础

### 基类和派生类
```csharp
// 基类
public class Character
{
    protected string name;
    protected int health;
    
    public Character(string name)
    {
        this.name = name;
        health = 100;
    }
    
    public virtual void TakeDamage(int damage)
    {
        health -= damage;
        if (health < 0) health = 0;
    }
}

// 派生类
public class Player : Character
{
    private int experience;
    
    public Player(string name) : base(name)
    {
        experience = 0;
    }
    
    public override void TakeDamage(int damage)
    {
        base.TakeDamage(damage);
        // 玩家特有的受伤逻辑
        OnPlayerDamaged();
    }
}
```

### 继承规则
1. **单一继承**
   ```csharp
   // C#只支持单一继承
   public class Monster : Character  // 允许
   public class Boss : Monster      // 允许
   // public class Hybrid : Monster, Character  // 不允许多重继承
   ```

2. **继承访问级别**
   ```csharp
   public class Base
   {
       private int privateField;    // 派生类不可访问
       protected int protectedField;// 派生类可访问
       public int publicField;      // 派生类可访问
   }
   ```

## 多态实现

### 方法重写
1. **虚方法**
   ```csharp
   public class Enemy : Character
   {
       // 重写基类虚方法
       public override void TakeDamage(int damage)
       {
           // 敌人特有的伤害计算
           int actualDamage = CalculateDamage(damage);
           base.TakeDamage(actualDamage);
       }
   }
   ```

2. **抽象方法**
   ```csharp
   public abstract class Weapon
   {
       // 抽象方法必须被派生类实现
       public abstract void Fire();
       
       // 普通方法可以有实现
       public virtual void Reload()
       {
           // 默认装弹逻辑
       }
   }
   
   public class Pistol : Weapon
   {
       public override void Fire()
       {
           // 手枪射击实现
       }
   }
   ```

### 多态应用
```csharp
public class GameManager : MonoBehaviour
{
    // 使用基类引用指向派生类对象
    private List<Character> characters = new List<Character>();
    
    public void DamageAll(int damage)
    {
        foreach (Character character in characters)
        {
            // 多态调用 - 根据实际类型调用相应的方法
            character.TakeDamage(damage);
        }
    }
}
```

## Unity中的继承

### MonoBehaviour继承
```csharp
public class EntityBase : MonoBehaviour
{
    protected virtual void Initialize()
    {
        // 基础初始化
    }
    
    protected virtual void OnDeath()
    {
        // 死亡处理
    }
}

public class PlayerEntity : EntityBase
{
    protected override void Initialize()
    {
        base.Initialize();
        // 玩家特有初始化
    }
    
    protected override void OnDeath()
    {
        base.OnDeath();
        GameManager.Instance.GameOver();
    }
}
```

### 组件继承
```csharp
public class WeaponBase : MonoBehaviour
{
    [SerializeField] protected float damage;
    [SerializeField] protected float range;
    
    public virtual void Attack()
    {
        // 基础攻击逻辑
    }
}

public class RangedWeapon : WeaponBase
{
    [SerializeField] private GameObject projectilePrefab;
    
    public override void Attack()
    {
        // 远程武器攻击实现
        SpawnProjectile();
    }
}
```

## 继承设计模式

### 模板方法模式
```csharp
public abstract class GameState
{
    public void Execute()
    {
        OnEnter();
        UpdateState();
        OnExit();
    }
    
    protected abstract void OnEnter();
    protected abstract void UpdateState();
    protected abstract void OnExit();
}

public class PlayingState : GameState
{
    protected override void OnEnter()
    {
        // 进入游戏状态
    }
    
    protected override void UpdateState()
    {
        // 更新游戏逻辑
    }
    
    protected override void OnExit()
    {
        // 退出游戏状态
    }
}
```

### 策略模式
```csharp
public interface IMovementStrategy
{
    void Move(Transform transform);
}

public class WalkStrategy : IMovementStrategy
{
    public void Move(Transform transform)
    {
        // 步行移动实现
    }
}

public class FlyStrategy : IMovementStrategy
{
    public void Move(Transform transform)
    {
        // 飞行移动实现
    }
}

public class Character : MonoBehaviour
{
    private IMovementStrategy movementStrategy;
    
    public void SetMovementStrategy(IMovementStrategy strategy)
    {
        movementStrategy = strategy;
    }
    
    private void Update()
    {
        movementStrategy?.Move(transform);
    }
}
```

## 最佳实践

### 继承原则
1. **合理使用继承**
   ```csharp
   // 好的继承关系
   public class Vehicle { }
   public class Car : Vehicle { }
   
   // 避免过深的继承层次
   public class SportsCar : Car { }
   public class RaceCar : SportsCar { }  // 考虑使用组合替代
   ```

2. **组合优于继承**
   ```csharp
   // 使用组合
   public class Player
   {
       private HealthSystem healthSystem;
       private MovementSystem movementSystem;
       private InventorySystem inventorySystem;
       
       private void Initialize()
       {
           healthSystem = new HealthSystem();
           movementSystem = new MovementSystem();
           inventorySystem = new InventorySystem();
       }
   }
   ```

### 多态设计
1. **面向接口编程**
   ```csharp
   public interface IDamageable
   {
       void TakeDamage(int damage);
   }
   
   public class Player : MonoBehaviour, IDamageable
   {
       public void TakeDamage(int damage)
       {
           // 实现接口方法
       }
   }
   
   // 使用接口类型
   public void DealDamage(IDamageable target, int damage)
   {
       target.TakeDamage(damage);
   }
   ```

2. **依赖倒置**
   ```csharp
   // 依赖抽象而不是具体实现
   public class WeaponSystem
   {
       private IWeapon currentWeapon;
       
       public void SetWeapon(IWeapon weapon)
       {
           currentWeapon = weapon;
       }
       
       public void Attack()
       {
           currentWeapon?.Fire();
       }
   }
