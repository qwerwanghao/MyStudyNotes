# 委托和事件

## 委托基础

### 委托定义
```csharp
// 定义委托类型
public delegate void GameEventHandler();
public delegate void PlayerEventHandler(Player player);
public delegate float CalculateHandler(float a, float b);

// 泛型委托
public delegate void GenericHandler<T>(T item);
public delegate TResult GenericCalculator<T, TResult>(T input);
```

### 委托使用
```csharp
public class GameManager
{
    private GameEventHandler gameStartHandler;
    private CalculateHandler damageCalculator;
    
    public void Initialize()
    {
        // 添加方法到委托
        gameStartHandler += OnGameStart;
        gameStartHandler += LogGameStart;
        
        // 设置伤害计算器
        damageCalculator = CalculateBasicDamage;
    }
    
    private void OnGameStart()
    {
        Debug.Log("Game Started!");
    }
    
    private void LogGameStart()
    {
        Debug.Log("Logging game start...");
    }
    
    private float CalculateBasicDamage(float baseDamage, float multiplier)
    {
        return baseDamage * multiplier;
    }
}
```

## 事件

### 事件声明
```csharp
public class Player : MonoBehaviour
{
    // 基本事件
    public event GameEventHandler OnSpawn;
    public event PlayerEventHandler OnDeath;
    
    // 使用Action和Func
    public event Action OnJump;
    public event Action<float> OnHealthChanged;
    public event Func<float, float> OnDamageCalculate;
}
```

### 事件触发和订阅
```csharp
public class HealthSystem : MonoBehaviour
{
    public event Action<float> OnHealthChanged;
    public event Action OnDeath;
    
    private float currentHealth;
    
    public void TakeDamage(float damage)
    {
        currentHealth -= damage;
        
        // 触发事件
        OnHealthChanged?.Invoke(currentHealth);
        
        if (currentHealth <= 0)
        {
            OnDeath?.Invoke();
        }
    }
}

public class HealthUI : MonoBehaviour
{
    private HealthSystem healthSystem;
    
    private void Start()
    {
        // 订阅事件
        healthSystem.OnHealthChanged += UpdateHealthBar;
        healthSystem.OnDeath += ShowGameOver;
    }
    
    private void OnDestroy()
    {
        // 取消订阅
        healthSystem.OnHealthChanged -= UpdateHealthBar;
        healthSystem.OnDeath -= ShowGameOver;
    }
}
```

## Unity事件系统

### Unity内置事件
```csharp
public class GameController : MonoBehaviour
{
    // Unity序列化事件
    [SerializeField]
    private UnityEvent onGameStart;
    
    [SerializeField]
    private UnityEvent<int> onScoreChanged;
    
    private void StartGame()
    {
        // 触发Unity事件
        onGameStart.Invoke();
    }
    
    private void UpdateScore(int newScore)
    {
        onScoreChanged.Invoke(newScore);
    }
}
```

## 事件模式

### 观察者模式
```csharp
public class Subject : MonoBehaviour
{
    private event Action onStateChanged;
    
    public void AddObserver(Action observer)
    {
        onStateChanged += observer;
    }
    
    public void RemoveObserver(Action observer)
    {
        onStateChanged -= observer;
    }
    
    protected void NotifyObservers()
    {
        onStateChanged?.Invoke();
    }
}

### 消息系统
```csharp
public class EventManager
{
    private static Dictionary<string, Action<object>> eventDictionary = 
        new Dictionary<string, Action<object>>();
    
    public static void Subscribe(string eventName, Action<object> listener)
    {
        if (!eventDictionary.ContainsKey(eventName))
        {
            eventDictionary[eventName] = null;
        }
        eventDictionary[eventName] += listener;
    }
    
    public static void Unsubscribe(string eventName, Action<object> listener)
    {
        if (eventDictionary.ContainsKey(eventName))
        {
            eventDictionary[eventName] -= listener;
        }
    }
    
    public static void TriggerEvent(string eventName, object data)
    {
        if (eventDictionary.ContainsKey(eventName))
        {
            eventDictionary[eventName]?.Invoke(data);
        }
    }
}
```

### 安全性考虑
```csharp
public class SafeEventSystem
{
    private event Action<object> onEvent;
    
    public void RaiseEvent(object data)
    {
        // 复制事件处理器列表防止在回调中修改
        var handlers = onEvent?.GetInvocationList();
        if (handlers == null) return;
        
        foreach (Action<object> handler in handlers)
        {
            try
            {
                handler(data);
            }
            catch (Exception e)
            {
                Debug.LogError($"Error in event handler: {e}");
            }
        }
    }
}
