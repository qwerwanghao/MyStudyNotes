# Unity性能优化

## 1. 性能优化概述

Unity游戏性能优化是一个系统工程，主要涉及以下几个方面：

- CPU性能优化
- GPU性能优化
- 内存优化
- 资源加载优化
- 网络性能优化

### 1.1 性能指标

- **帧率（FPS）**：每秒渲染的帧数
- **CPU使用率**：处理游戏逻辑的开销
- **内存占用**：游戏运行时的内存使用情况
- **Draw Call**：渲染批次
- **顶点数和三角面数**：模型复杂度
- **纹理内存**：贴图占用的内存

### 1.2 性能分析工具

- Unity Profiler
- Frame Debugger
- Memory Profiler
- GPU Profiler
- Build Report

### 2.1 代码优化

```csharp
public class PerformanceOptimizer : MonoBehaviour
{
    // 缓存组件引用
    private Transform cachedTransform;
    private Rigidbody cachedRigidbody;
    
    void Awake()
    {
        // 在Awake中缓存组件引用
        cachedTransform = transform;
        cachedRigidbody = GetComponent<Rigidbody>();
    }
    
    // 使用对象池而不是实时创建销毁
    private Queue<GameObject> objectPool = new Queue<GameObject>();
    
    public GameObject GetObjectFromPool()
    {
        if (objectPool.Count > 0)
        {
            return objectPool.Dequeue();
        }
        return Instantiate(prefab);
    }
    
    // 避免在Update中进行复杂计算
    private float updateInterval = 0.5f;
    private float nextUpdateTime = 0;
    
    void Update()
    {
        if (Time.time >= nextUpdateTime)
        {
            PerformHeavyCalculation();
            nextUpdateTime = Time.time + updateInterval;
        }
    }
    
    // 使用协程处理分帧任务
    IEnumerator ProcessLargeDataSet(List<Data> dataSet)
    {
        const int itemsPerFrame = 100;
        for (int i = 0; i < dataSet.Count; i += itemsPerFrame)
        {
            for (int j = i; j < Mathf.Min(i + itemsPerFrame, dataSet.Count); j++)
            {
                ProcessData(dataSet[j]);
            }
            yield return null;
        }
    }
}
```

```csharp
public class PhysicsOptimizer : MonoBehaviour

    }
```

### 2.3 协程优化

```csharp
public class CoroutineOptimizer : MonoBehaviour
{
    // 使用对象池管理协程
    private Queue<IEnumerator> coroutinePool = new Queue<IEnumerator>();
    private List<Coroutine> activeCoroutines = new List<Coroutine>();
    
    public void StartPooledCoroutine(IEnumerator routine)
    {
        if (coroutinePool.Count > 0)
        {
            var pooledRoutine = coroutinePool.Dequeue();
            var coroutine = StartCoroutine(pooledRoutine);
            activeCoroutines.Add(coroutine);
        }
        else
        {
            var coroutine = StartCoroutine(routine);
            activeCoroutines.Add(coroutine);
        }
    }
    
    // 分帧处理大量数据
    IEnumerator ProcessDataInChunks<T>(List<T> items, int chunkSize, System.Action<T> process)
    {
        for (int i = 0; i < items.Count; i += chunkSize)
        {
            int endIndex = Mathf.Min(i + chunkSize, items.Count);
            for (int j = i; j < endIndex; j++)
            {
                process(items[j]);
            }
            yield return null;
        }
    }
}
```

## 3. GPU性能优化

### 3.1 Draw Call优化

```csharp
public class DrawCallOptimizer : MonoBehaviour
{
    // 静态批处理
    void Start()
    {
        // 标记对象为静态
        gameObject.isStatic = true;
        
        // 合并网格
        MeshFilter[] meshFilters = GetComponentsInChildren<MeshFilter>();
        CombineInstance[] combine = new CombineInstance[meshFilters.Length];
        
        for (int i = 0; i < meshFilters.Length; i++)
        {
            combine[i].mesh = meshFilters[i].sharedMesh;
            combine[i].transform = meshFilters[i].transform.localToWorldMatrix;
        }
        
        Mesh combinedMesh = new Mesh();
        combinedMesh.CombineMeshes(combine);
        GetComponent<MeshFilter>().mesh = combinedMesh;
    }
}
```

```csharp
// 优化的简单着色器示例
Shader "Custom/OptimizedShader"
{
    Properties
    {
        _MainTex ("Texture", 2D) = "white" {}
    }
    SubShader
    {
        Tags { "RenderType"="Opaque" }
        LOD 100
        
        Pass
        {
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #pragma multi_compile_fog
            
            #include "UnityCG.cginc"
            
            struct appdata
            {
                float4 vertex : POSITION;
                float2 uv : TEXCOORD0;
            };
            
            struct v2f
            {
                float2 uv : TEXCOORD0;
                UNITY_FOG_COORDS(1)
                float4 vertex : SV_POSITION;
            };
            
            sampler2D _MainTex;
            float4 _MainTex_ST;
            
            v2f vert (appdata v)
            {
                v2f o;
                o.vertex = UnityObjectToClipPos(v.vertex);
                o.uv = TRANSFORM_TEX(v.uv, _MainTex);
                UNITY_TRANSFER_FOG(o,o.vertex);
                return o;
            }
            
            fixed4 frag (v2f i) : SV_Target
            {
                fixed4 col = tex2D(_MainTex, i.uv);
                UNITY_APPLY_FOG(i.fogCoord, col);
                return col;
            }
            ENDCG
        }
    }
}
```

### 3.3 LOD系统

```csharp
public class LODManager : MonoBehaviour
{
    [System.Serializable]
    public class LODLevel
    {
        public float distance;
        public Mesh mesh;
        public Material material;
    }
    
    public LODLevel[] lodLevels;
    private MeshFilter meshFilter;
    private MeshRenderer meshRenderer;
    private Transform playerCamera;
    
    void Start()
    {
        meshFilter = GetComponent<MeshFilter>();
        meshRenderer = GetComponent<MeshRenderer>();
        playerCamera = Camera.main.transform;
        
        // 初始化为最高级别LOD
        UpdateLOD();
    }
    
    void Update()
    {
        UpdateLOD();
    }
    
    void UpdateLOD()
    {
        float distance = Vector3.Distance(transform.position, playerCamera.position);
        
        for (int i = 0; i < lodLevels.Length; i++)
        {
            if (distance <= lodLevels[i].distance)
            {
                meshFilter.mesh = lodLevels[i].mesh;
                meshRenderer.material = lodLevels[i].material;
                break;
            }
        }
    }
}
```

### 4.1 内存泄漏检测

```csharp
public class MemoryLeakDetector : MonoBehaviour
{
    private Dictionary<string, WeakReference> objectReferences = new Dictionary<string, WeakReference>();
    
    public void TrackObject(string key, object obj)
    {
        objectReferences[key] = new WeakReference(obj);
    }
    
    public void CheckForLeaks()
    {
        foreach (var kvp in objectReferences)
        {
            if (kvp.Value.IsAlive)
            {
                Debug.LogWarning($"Potential memory leak detected: {kvp.Key}");
            }
        }
    }
    
    void OnDestroy()
    {
        // 检查是否有未释放的资源
        CheckForLeaks();
    }
}
```

### 4.2 资源释放

```csharp
public class ResourceCleaner : MonoBehaviour
{
    private List<UnityEngine.Object> managedResources = new List<UnityEngine.Object>();
    
    public void RegisterResource(UnityEngine.Object resource)
    {
        managedResources.Add(resource);
    }
    
    void OnDestroy()
    {
        // 释放所有注册的资源
        foreach (var resource in managedResources)
        {
            if (resource != null)
            {
                if (resource is Material || resource is Texture)
                {
                    Resources.UnloadAsset(resource);
                }
                else
                {
                    Destroy(resource);
                }
            }
        }
        
        managedResources.Clear();
        Resources.UnloadUnusedAssets();
        System.GC.Collect();
    }
}
```

## 5. UI优化

### 5.1 UGUI优化

```csharp
public class UIOptimizer : MonoBehaviour
{
    // 使用对象池管理UI元素
    private Dictionary<string, Queue<GameObject>> uiPool = new Dictionary<string, Queue<GameObject>>();
    
    // 动态加载UI
    public void LoadUIPanel(string panelName)
    {
        if (uiPool.ContainsKey(panelName) && uiPool[panelName].Count > 0)
        {
            var panel = uiPool[panelName].Dequeue();
            panel.SetActive(true);
        }
        else
        {
            // 异步加载UI预制体
            StartCoroutine(LoadUIPanelAsync(panelName));
        }
    }
    
    // 设置Canvas的更新模式
    public void OptimizeCanvas(Canvas canvas)
    {
        if (canvas.renderMode == RenderMode.ScreenSpaceOverlay)
        {
            canvas.renderMode = RenderMode.ScreenSpaceCamera;
            canvas.worldCamera = Camera.main;
        }
    }
    
    // 优化布局组件
    public void OptimizeLayoutGroup(LayoutGroup layoutGroup)
    {
        layoutGroup.childControlWidth = false;
        layoutGroup.childControlHeight = false;
    }
}
```

### 5.2 图集优化

```csharp
public class AtlasManager : MonoBehaviour
{
    private Dictionary<string, Sprite> spriteAtlas = new Dictionary<string, Sprite>();
    
    public void LoadAtlas(string atlasName)
    {
        // 加载图集
        var atlas = Resources.Load<SpriteAtlas>(atlasName);
        if (atlas != null)
        {
            Sprite[] sprites = new Sprite[atlas.spriteCount];
            atlas.GetSprites(sprites);
            
            foreach (var sprite in sprites)
            {
                spriteAtlas[sprite.name] = sprite;
            }
        }
    }
    
    public Sprite GetSprite(string spriteName)
    {
        if (spriteAtlas.TryGetValue(spriteName, out Sprite sprite))
        {
            return sprite;
        }
        return null;
    }
}
```

## 6. 网络优化

### 6.1 数据压缩

```csharp
public class NetworkOptimizer
{
    // 使用Protocol Buffers序列化数据
    public byte[] SerializeData<T>(T data) where T : IMessage
    {
        using (MemoryStream stream = new MemoryStream())
        {
            // 序列化数据
            data.WriteTo(stream);
            return stream.ToArray();
        }
    }
    
    // 压缩数据
    public byte[] CompressData(byte[] data)
    {
        using (MemoryStream output = new MemoryStream())
        {
            using (GZipStream gzip = new GZipStream(output, CompressionLevel.Optimal))
            {
                gzip.Write(data, 0, data.Length);
            }
            return output.ToArray();
        }
    }
}
```

### 6.2 网络请求优化

```csharp
public class NetworkRequestOptimizer : MonoBehaviour
{
    private Queue<UnityWebRequest> requestPool = new Queue<UnityWebRequest>();
    
    // 重用WebRequest对象
    public UnityWebRequest GetWebRequest(string url)
    {
        if (requestPool.Count > 0)
        {
            var request = requestPool.Dequeue();
            request.url = url;
            request.downloadHandler = new DownloadHandlerBuffer();
            return request;
        }
        return new UnityWebRequest(url);
    }
    
    // 批量请求处理
    public IEnumerator BatchRequests(List<string> urls, int batchSize)
    {
        for (int i = 0; i < urls.Count; i += batchSize)
        {
            List<UnityWebRequest> requests = new List<UnityWebRequest>();
            
            for (int j = i; j < Mathf.Min(i + batchSize, urls.Count); j++)
            {
                var request = GetWebRequest(urls[j]);
                requests.Add(request);
                request.SendWebRequest();
            }
            
            bool allDone;
            do
            {
                allDone = true;
                foreach (var request in requests)
                {
                    if (!request.isDone)
                    {
                        allDone = false;
                        break;
                    }
                }
                yield return null;
            } while (!allDone);
            
            // 处理响应
            foreach (var request in requests)
            {
                ProcessResponse(request);
                requestPool.Enqueue(request);
            }
        }
    }
}
```

## 7. 性能监控

### 7.1 性能监控器

```csharp
public class PerformanceMonitor : MonoBehaviour
{
    private float[] fpsBuffer = new float[60];
    private int fpsBufferIndex = 0;
    
    private void Update()
    {
        // 记录FPS
        fpsBuffer[fpsBufferIndex] = 1.0f / Time.deltaTime;
        fpsBufferIndex = (fpsBufferIndex + 1) % fpsBuffer.Length;
        
        // 计算平均FPS
        float averageFps = 0;
        foreach (float fps in fpsBuffer)
        {
            averageFps += fps;
        }
        averageFps /= fpsBuffer.Length;
        
        // 监控内存使用
        float totalMemory = Profiler.GetTotalAllocatedMemoryLong() / (1024f * 1024f); // MB
        
        // 监控GC
        if (Profiler.GetMonoHeapSizeLong() > 256 * 1024 * 1024) // 256MB
        {
            System.GC.Collect();
        }
    }
}
```

### 7.2 性能报告生成

```csharp
public class PerformanceReporter : MonoBehaviour
{
    [System.Serializable]
    public class PerformanceData
    {
        public float averageFps;
        public float minFps;
        public float maxFps;
        public float memoryUsage;
        public int drawCalls;
        public int triangles;
        public float loadTime;
    }
    
    private List<PerformanceData> performanceHistory = new List<PerformanceData>();
    
    public void GenerateReport()
    {
        var report = new PerformanceData
        {
            averageFps = Time.frameCount / Time.time,
            memoryUsage = Profiler.GetTotalAllocatedMemoryLong() / (1024f * 1024f),
            drawCalls = UnityStats.drawCalls,
            triangles = UnityStats.triangles
        };
        
        performanceHistory.Add(report);
        
        // 导出报告
        string json = JsonUtility.ToJson(performanceHistory);
        File.WriteAllText("PerformanceReport.json", json);
    }
}
```
