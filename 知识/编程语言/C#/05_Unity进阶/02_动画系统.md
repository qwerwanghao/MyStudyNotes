# 动画系统

## Animation组件

### 基础动画设置
```csharp
public class AnimationController : MonoBehaviour
{
    private Animation anim;
    
    void Start()
    {
        anim = GetComponent<Animation>();
        
        // 播放动画
        anim.Play("Idle");
        
        // 混合动画
        anim.Blend("Walk");
        
        // 交叉淡入动画
        anim.CrossFade("Run", 0.3f);
    }
    
    // 动画回调
    void OnAnimationComplete()
    {
        Debug.Log("Animation completed");
    }
    
    // 动画控制
    public void PlayAnimation(string animName, float fadeLength = 0.3f)
    {
        if (anim.IsPlaying(animName))
            return;
            
        anim.CrossFade(animName, fadeLength);
    }
}
```

## Animator组件

### 状态机控制
```csharp
public class AnimatorController : MonoBehaviour
{
    private Animator animator;
    
    void Start()
    {
        animator = GetComponent<Animator>();
        
        // 设置参数
        animator.SetBool("IsRunning", true);
        animator.SetFloat("Speed", 1.0f);
        animator.SetTrigger("Jump");
        
        // 获取参数
        bool isRunning = animator.GetBool("IsRunning");
        float speed = animator.GetFloat("Speed");
    }
    
    // 状态检查
    public bool IsInState(string stateName)
    {
        return animator.GetCurrentAnimatorStateInfo(0).IsName(stateName);
    }
    
    // 状态切换
    public void TransitionToState(string stateName)
    {
        animator.CrossFade(stateName, 0.25f);
    }
}
```

### 混合树设置
```csharp
public class BlendTreeController : MonoBehaviour
{
    private Animator animator;
    
    void Start()
    {
        animator = GetComponent<Animator>();
    }
    
    void Update()
    {
        // 设置混合参数
        float horizontal = Input.GetAxis("Horizontal");
        float vertical = Input.GetAxis("Vertical");
        
        // 更新混合树参数
        animator.SetFloat("X", horizontal);
        animator.SetFloat("Y", vertical);
        
        // 计算混合权重
        float speed = new Vector2(horizontal, vertical).magnitude;
        animator.SetFloat("Speed", speed);
    }
}
```

## 高级动画功能

### IK (Inverse Kinematics)
```csharp
public class IKController : MonoBehaviour
{
    private Animator animator;
    public Transform leftHandTarget;
    public Transform rightHandTarget;
    public Transform lookAtTarget;
    
    void Start()
    {
        animator = GetComponent<Animator>();
    }
    
    void OnAnimatorIK(int layerIndex)
    {
        if (animator)
        {
            // 设置视线IK
            if (lookAtTarget != null)
            {
                animator.SetLookAtWeight(1);
                animator.SetLookAtPosition(lookAtTarget.position);
            }
            
            // 设置手部IK
            if (leftHandTarget != null)
            {
                animator.SetIKPositionWeight(AvatarIKGoal.LeftHand, 1);
                animator.SetIKRotationWeight(AvatarIKGoal.LeftHand, 1);
                animator.SetIKPosition(AvatarIKGoal.LeftHand, 
                    leftHandTarget.position);
                animator.SetIKRotation(AvatarIKGoal.LeftHand, 
                    leftHandTarget.rotation);
            }
            
            if (rightHandTarget != null)
            {
                animator.SetIKPositionWeight(AvatarIKGoal.RightHand, 1);
                animator.SetIKRotationWeight(AvatarIKGoal.RightHand, 1);
                animator.SetIKPosition(AvatarIKGoal.RightHand, 
                    rightHandTarget.position);
                animator.SetIKRotation(AvatarIKGoal.RightHand, 
                    rightHandTarget.rotation);
            }
        }
    }
}
```

### 动画层管理
```csharp
public class AnimationLayerManager : MonoBehaviour
{
    private Animator animator;
    
    void Start()
    {
        animator = GetComponent<Animator>();
    }
    
    // 设置层权重
    public void SetLayerWeight(int layerIndex, float weight)
    {
        animator.SetLayerWeight(layerIndex, weight);
    }
    
    // 渐变层权重
    public IEnumerator FadeLayerWeight(int layerIndex, float targetWeight, 
        float duration)
    {
        float startWeight = animator.GetLayerWeight(layerIndex);
        float elapsed = 0f;
        
        while (elapsed < duration)
        {
            elapsed += Time.deltaTime;
            float weight = Mathf.Lerp(startWeight, targetWeight, 
                elapsed / duration);
            animator.SetLayerWeight(layerIndex, weight);
            yield return null;
        }
        
        animator.SetLayerWeight(layerIndex, targetWeight);
    }
}
```

## 动画优化

### 性能优化
```csharp
public class AnimationOptimization : MonoBehaviour
{
    private Animator animator;
    
    void Start()
    {
        animator = GetComponent<Animator>();
        
        // 设置Culling模式
        animator.cullingMode = AnimatorCullingMode.CullCompletely;
        
        // 优化更新模式
        animator.updateMode = AnimatorUpdateMode.Normal;
        
        // 关闭不需要的功能
        animator.applyRootMotion = false;
        
        // 设置动画质量
        QualitySettings.skinWeights = SkinWeights.TwoBones;
    }
    
    // 距离裁剪
    void Update()
    {
        float distanceToCamera = Vector3.Distance(
            transform.position, 
            Camera.main.transform.position
        );
        
        if (distanceToCamera > 50f)
        {
            animator.enabled = false;
        }
        else
        {
            animator.enabled = true;
        }
    }
}
```

### 动画复用
```csharp
public class AnimationReuse : MonoBehaviour
{
    // 动画覆盖控制器
    private AnimatorOverrideController overrideController;
    
    void Start()
    {
        var animator = GetComponent<Animator>();
        overrideController = new AnimatorOverrideController(
            animator.runtimeAnimatorController);
        animator.runtimeAnimatorController = overrideController;
    }
    
    // 替换动画片段
    public void OverrideAnimation(string animationName, 
        AnimationClip newClip)
    {
        overrideController[animationName] = newClip;
    }
    
    // 共享动画状态
    public void ShareAnimationState(Animator sourceAnimator)
    {
        var animator = GetComponent<Animator>();
        foreach (var parameter in sourceAnimator.parameters)
        {
            animator.SetFloat(parameter.name, 
                sourceAnimator.GetFloat(parameter.name));
        }
    }
}
