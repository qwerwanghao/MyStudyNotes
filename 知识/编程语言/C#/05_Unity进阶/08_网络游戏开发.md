# Unity网络游戏开发

## 1. 网络基础

### 1.1 网络协议

- **TCP协议**：可靠的、面向连接的协议
- **UDP协议**：快速的、无连接的协议
- **WebSocket**：基于TCP的全双工通信协议
- **HTTP/HTTPS**：用于RESTful API通信

### 1.2 网络架构

- **客户端-服务器架构（C/S）**
- **点对点架构（P2P）**
- **混合架构**

### 1.3 网络同步模型

- **状态同步**：适用于即时性要求高的游戏
- **帧同步**：适用于策略类游戏
- **事件同步**：适用于回合制游戏

```csharp
    private Quaternion networkRotation;

        if (stream.IsWriting)
        {
            // 发送数据
            stream.SendNext(transform.position);
            stream.SendNext(transform.rotation);
        }
        else
        {
            // 接收数据
            networkPosition = (Vector3)stream.ReceiveNext();
            networkRotation = (Quaternion)stream.ReceiveNext();
        }
    }
    
    void Update()
    {
        if (!photonView.IsMine)
        {
            // 平滑同步位置和旋转
            transform.position = Vector3.Lerp(transform.position, networkPosition, Time.deltaTime * 10);
            transform.rotation = Quaternion.Lerp(transform.rotation, networkRotation, Time.deltaTime * 10);
        }
    }
}
```

## 3. 状态同步

### 3.1 位置同步

```csharp
public class PositionSync : NetworkBehaviour
{
    [SerializeField]
    private float syncInterval = 0.1f;
    private float lastSyncTime;
    
    [SyncVar(hook = nameof(OnPositionChanged))]
    private Vector3 syncPosition;
    
    void Update()
    {
        if (isLocalPlayer)
        {
            if (Time.time - lastSyncTime >= syncInterval)
            {
                CmdSyncPosition(transform.position);
                lastSyncTime = Time.time;
            }
        }
    }
    
    [Command]
    void CmdSyncPosition(Vector3 position)
    {
        syncPosition = position;
    }
    
    void OnPositionChanged(Vector3 oldPos, Vector3 newPos)
    {
        if (!isLocalPlayer)
        {
            transform.position = Vector3.Lerp(transform.position, newPos, Time.deltaTime * 10f);
        }
    }
}
```

## 4. 帧同步

### 4.1 帧同步管理器

```csharp
public class FrameSyncManager : NetworkBehaviour
{
    private const int BUFFER_FRAMES = 6;
    private Queue<FrameData> frameBuffer = new Queue<FrameData>();
    private int currentFrame = 0;
    
    [System.Serializable]
    public struct FrameData
    {
        public int frameNumber;
        public Dictionary<int, PlayerInput> inputs;
    }
    
    [System.Serializable]
    public struct PlayerInput
    {
        public Vector2 movement;
        public bool[] actions;
    }
    
    void FixedUpdate()
    {
        if (isServer)
        {
            ProcessFrame();
        }
    }
    
    [Command]
    void CmdSendInput(PlayerInput input)
    {
        if (frameBuffer.Count < BUFFER_FRAMES)
        {
            var frameData = new FrameData
            {
                frameNumber = currentFrame,
                inputs = new Dictionary<int, PlayerInput>()
            };
            frameData.inputs.Add(connectionToClient.connectionId, input);
            frameBuffer.Enqueue(frameData);
        }
    }
    
    void ProcessFrame()
    {
        if (frameBuffer.Count > 0)
        {
            var frame = frameBuffer.Dequeue();
            RpcExecuteFrame(frame);
            currentFrame++;
        }
    }
    
    [ClientRpc]
    void RpcExecuteFrame(FrameData frame)
    {
        foreach (var input in frame.inputs)
        {
            // 执行输入
            ExecutePlayerInput(input.Key, input.Value);
        }
    }
}
```

### 4.2 帧同步校验

```csharp
public class FrameValidator : NetworkBehaviour
{
    private Dictionary<int, uint> frameChecksums = new Dictionary<int, uint>();
    
    public uint CalculateChecksum()
    {
        uint checksum = 0;
        // 计算游戏状态的校验和
        var gameObjects = FindObjectsOfType<NetworkIdentity>();
        foreach (var obj in gameObjects)
        {
            checksum ^= (uint)obj.transform.position.GetHashCode();
            checksum ^= (uint)obj.transform.rotation.GetHashCode();
        }
        return checksum;
    }
    
    [Command]
    void CmdSubmitChecksum(int frame, uint checksum)
    {
        if (!frameChecksums.ContainsKey(frame))
        {
            frameChecksums[frame] = checksum;
        }
        else if (frameChecksums[frame] != checksum)
        {
            // 检测到不同步，触发重同步
            RpcResync(frame);
        }
    }
    
    [ClientRpc]
    void RpcResync(int frame)
    {
        // 回滚到指定帧
        RollbackToFrame(frame);
    }
}
```

```csharp
public class RoomManager : NetworkBehaviour
{
    private Dictionary<string, Room> rooms = new Dictionary<string, Room>();
    
    [System.Serializable]
    public class Room
    {
        public string name;
        public int maxPlayers;
        public List<NetworkConnection> players = new List<NetworkConnection>();
        public GameState gameState;
    }
    
    [Command]
    public void CmdCreateRoom(string roomName, int maxPlayers)
    {
        if (!rooms.ContainsKey(roomName))
        {
            var room = new Room
            {
                name = roomName,
                maxPlayers = maxPlayers
            };
            rooms.Add(roomName, room);
            RpcUpdateRoomList(GetRoomList());
        }
    }
    
    [Command]
    public void CmdJoinRoom(string roomName)
    {
        if (rooms.TryGetValue(roomName, out Room room))
        {
            if (room.players.Count < room.maxPlayers)
            {
                room.players.Add(connectionToClient);
                RpcJoinedRoom(roomName);
            }
        }
    }
    
    [ClientRpc]
    void RpcUpdateRoomList(string[] roomList)
    {
        // 更新房间列表UI
        UpdateRoomListUI(roomList);
    }
    
    [ClientRpc]
    void RpcJoinedRoom(string roomName)
```

### 6.2 匹配系统

```csharp
public class MatchmakingSystem : NetworkBehaviour
{
    private Queue<NetworkConnection> waitingPlayers = new Queue<NetworkConnection>();
    private const int PLAYERS_PER_MATCH = 2;
    
    [Command]
    public void CmdEnterMatchmaking()
    {
        waitingPlayers.Enqueue(connectionToClient);
        CheckForMatch();
    }
    
    void CheckForMatch()
    {
        if (waitingPlayers.Count >= PLAYERS_PER_MATCH)
        {
            var players = new List<NetworkConnection>();
            for (int i = 0; i < PLAYERS_PER_MATCH; i++)
            {
                players.Add(waitingPlayers.Dequeue());
            }
            CreateMatch(players);
        }
    }
    
    void CreateMatch(List<NetworkConnection> players)
    {
        // 创建新的比赛实例
        var match = new Match(players);
        
        // 通知所有玩家
        foreach (var player in players)
        {
            TargetStartMatch(player, match.matchId);
        }
    }
    
    [TargetRpc]
    void TargetStartMatch(NetworkConnection target, string matchId)
    {
        // 客户端开始比赛
        StartMatch(matchId);
    }
}
```

## 7. 网络安全

### 7.1 反作弊系统

```csharp
public class AntiCheatSystem : NetworkBehaviour
{
    private Dictionary<NetworkConnection, PlayerStats> playerStats = new Dictionary<NetworkConnection, PlayerStats>();
    
    [System.Serializable]
    public class PlayerStats
    {
        public Vector3 lastPosition;
        public float lastUpdateTime;
        public int suspiciousActions;
    }
    
    [Server]
    void ValidatePlayerAction(NetworkConnection conn, Vector3 position, float speed)
    {
        if (playerStats.TryGetValue(conn, out PlayerStats stats))
        {
            float timeDelta = Time.time - stats.lastUpdateTime;
            float distance = Vector3.Distance(position, stats.lastPosition);
            float actualSpeed = distance / timeDelta;
            
            if (actualSpeed > speed * 1.5f) // 速度容差
            {
                stats.suspiciousActions++;
                if (stats.suspiciousActions > 5)
                {
                    // 触发惩罚机制
                    PunishPlayer(conn);
                }
            }
            
            stats.lastPosition = position;
            stats.lastUpdateTime = Time.time;
        }
    }
    
    void PunishPlayer(NetworkConnection conn)
    {
        // 实现惩罚机制
        TargetShowWarning(conn, "检测到异常行为");
    }
    
    [TargetRpc]
    void TargetShowWarning(NetworkConnection target, string message)
    {
        Debug.LogWarning($"反作弊警告: {message}");
    }
}
```

### 7.2 数据加密

```csharp
public class NetworkEncryption
{
    private static readonly byte[] Key = new byte[32]; // AES-256
    private static readonly byte[] IV = new byte[16];
    
    public static byte[] EncryptData(byte[] data)
    {
        using (Aes aes = Aes.Create())
        {
            aes.Key = Key;
            aes.IV = IV;
            
            using (MemoryStream ms = new MemoryStream())
            {
                using (CryptoStream cs = new CryptoStream(ms, aes.CreateEncryptor(), CryptoStreamMode.Write))
                {
                    cs.Write(data, 0, data.Length);
                    cs.FlushFinalBlock();
                    return ms.ToArray();
                }
            }
        }
    }
    
    public static byte[] DecryptData(byte[] encryptedData)
    {
        using (Aes aes = Aes.Create())
        {
            aes.Key = Key;
            aes.IV = IV;
            
            using (MemoryStream ms = new MemoryStream())
            {
                using (CryptoStream cs = new CryptoStream(ms, aes.CreateDecryptor(), CryptoStreamMode.Write))
                {
                    cs.Write(encryptedData, 0, encryptedData.Length);
                    cs.FlushFinalBlock();
                    return ms.ToArray();
                }
            }
        }
    }
}
```

## 8. 网络优化

### 8.1 带宽优化

```csharp
public class NetworkOptimizer : NetworkBehaviour
{
    [SerializeField]
    private float minSendInterval = 0.1f; // 最小发送间隔
    private float lastSendTime;
    
    [SerializeField]
    private float positionThreshold = 0.1f; // 位置变化阈值
    private Vector3 lastSentPosition;
    
    void Update()
    {
        if (isLocalPlayer && Time.time - lastSendTime >= minSendInterval)
        {
            Vector3 currentPosition = transform.position;
            if (Vector3.Distance(currentPosition, lastSentPosition) >= positionThreshold)
            {
                CmdSendPosition(currentPosition);
                lastSentPosition = currentPosition;
                lastSendTime = Time.time;
            }
        }
    }
    
    // 数据压缩
    Vector3 CompressPosition(Vector3 position)
    {
        return new Vector3(
            Mathf.Round(position.x * 100f) / 100f,
            Mathf.Round(position.y * 100f) / 100f,
            Mathf.Round(position.z * 100f) / 100f
        );
    }
}
```

### 8.2 延迟补偿

```csharp
public class LagCompensation : NetworkBehaviour
{
    private const int HISTORY_SIZE = 20;
    private Dictionary<int, Queue<StateHistory>> playerHistory = new Dictionary<int, Queue<StateHistory>>();
    
    [System.Serializable]
    public struct StateHistory
    {
        public Vector3 position;
        public Quaternion rotation;
        public float timestamp;
    }
    
    void RecordState(int playerId, Vector3 position, Quaternion rotation)
    {
        if (!playerHistory.ContainsKey(playerId))
        {
            playerHistory[playerId] = new Queue<StateHistory>();
        }
        
        var history = playerHistory[playerId];
        history.Enqueue(new StateHistory
        {
            position = position,
            rotation = rotation,
            timestamp = Time.time
        });
        
        while (history.Count > HISTORY_SIZE)
        {
            history.Dequeue();
        }
    }
    
    StateHistory? GetPastState(int playerId, float timestamp)
    {
        if (playerHistory.TryGetValue(playerId, out Queue<StateHistory> history))
        {
            foreach (var state in history)
            {
                if (Mathf.Abs(state.timestamp - timestamp) < 0.1f)
                {
                    return state;
                }
            }
        }
        return null;
    }
}
```
