# 特性和反射

## 特性（Attributes）

### 内置特性
```csharp
// 常用特性示例
public class GameComponent : MonoBehaviour
{
    [SerializeField]  // Unity序列化字段
    private float speed = 5f;
    
    [Range(0, 100)]  // 限制数值范围
    public float health;
    
    [Header("武器设置")]  // 在Inspector中添加标题
    [Tooltip("武器伤害值")]  // 添加提示信息
    public int weaponDamage;
    
    [Obsolete("请使用新版本的Move方法")]  // 标记过时
    public void OldMove() { }
}
```

### 自定义特性
```csharp
// 自定义特性定义
[AttributeUsage(AttributeTargets.Method | AttributeTargets.Field)]
public class DevOnlyAttribute : Attribute
{
    public string Reason { get; }
    
    public DevOnlyAttribute(string reason)
    {
        Reason = reason;
    }
}

// 使用自定义特性
public class DebugTools
{
    [DevOnly("仅用于调试")]
    public static void ClearAllData()
    {
        // 清除所有数据
    }
}
```

## 反射基础

### 类型信息
```csharp
public class ReflectionExample
{
    public void InspectType<T>(T obj)
    {
        Type type = typeof(T);
        // 或者
        Type type2 = obj.GetType();
        
        // 获取类型信息
        Console.WriteLine($"类型名称: {type.Name}");
        Console.WriteLine($"命名空间: {type.Namespace}");
        Console.WriteLine($"是否为类: {type.IsClass}");
        Console.WriteLine($"基类: {type.BaseType?.Name}");
        
        // 获取成员信息
        var fields = type.GetFields(BindingFlags.Public | 
                                  BindingFlags.NonPublic | 
                                  BindingFlags.Instance);
                                  
        var properties = type.GetProperties();
        var methods = type.GetMethods();
    }
}
```

### 动态创建实例
```csharp
public class Factory
{
    public T CreateInstance<T>() where T : class
    {
        // 使用反射创建实例
        return Activator.CreateInstance<T>();
    }
    
    public object CreateInstanceByName(string typeName)
    {
        // 通过类型名创建实例
        Type type = Type.GetType(typeName);
        if (type != null)
        {
            return Activator.CreateInstance(type);
        }
        return null;
    }
}
```

## Unity中的反射应用

### 组件管理
```csharp
public class ComponentManager : MonoBehaviour
{
    // 通过反射获取组件
    public T GetOrAddComponent<T>() where T : Component
    {
        T component = GetComponent<T>();
        if (component == null)
        {
            component = gameObject.AddComponent<T>();
        }
        return component;
    }
    
    // 通过反射查找所有特定特性的方法
    public void InvokeMethodsWithAttribute<T>() where T : Attribute
    {
        var methods = GetType().GetMethods()
            .Where(m => m.GetCustomAttributes(typeof(T), true).Any());
            
        foreach (var method in methods)
        {
            method.Invoke(this, null);
        }
    }
}
```

### 序列化系统
```csharp
public class SerializationSystem
{
    public string SerializeObject(object obj)
    {
        var data = new Dictionary<string, object>();
        Type type = obj.GetType();
        
        // 获取所有需要序列化的字段
        var fields = type.GetFields(BindingFlags.Public | 
                                  BindingFlags.NonPublic | 
                                  BindingFlags.Instance)
            .Where(f => f.GetCustomAttribute<SerializeField>() != null);
            
        foreach (var field in fields)
        {
            data[field.Name] = field.GetValue(obj);
        }
        
        return JsonUtility.ToJson(data);
    }
}
```

## 动态代码执行

### 方法调用
```csharp
public class MethodInvoker
{
    public object InvokeMethod(object instance, string methodName, params object[] parameters)
    {
        Type type = instance.GetType();
        MethodInfo method = type.GetMethod(methodName);
        
        if (method != null)
        {
            return method.Invoke(instance, parameters);
        }
        
        throw new MissingMethodException(type.Name, methodName);
    }
}
```

### 属性访问
```csharp
public class PropertyAccessor
{
    public object GetPropertyValue(object instance, string propertyName)
    {
        PropertyInfo property = instance.GetType().GetProperty(propertyName);
        return property?.GetValue(instance);
    }
    
    public void SetPropertyValue(object instance, string propertyName, object value)
    {
        PropertyInfo property = instance.GetType().GetProperty(propertyName);
        property?.SetValue(instance, value);
    }
}
```

## 性能优化

### 缓存反射信息
```csharp
public class ReflectionCache
{
    private static readonly Dictionary<Type, PropertyInfo[]> PropertyCache 
        = new Dictionary<Type, PropertyInfo[]>();
        
    public static PropertyInfo[] GetTypeProperties(Type type)
    {
        if (!PropertyCache.TryGetValue(type, out var properties))
        {
            properties = type.GetProperties();
            PropertyCache[type] = properties;
        }
        return properties;
    }
}
```

### 委托优化
```csharp
public class DelegateOptimization
{
    // 创建委托缓存
    private static readonly Dictionary<MethodInfo, Delegate> DelegateCache 
        = new Dictionary<MethodInfo, Delegate>();
        
    public static Delegate CreateDelegate(MethodInfo method, Type delegateType)
    {
        if (!DelegateCache.TryGetValue(method, out var delegate))
        {
            delegate = Delegate.CreateDelegate(delegateType, method);
            DelegateCache[method] = delegate;
        }
        return delegate;
    }
}
```

## 安全考虑

### 反射安全检查
```csharp
public class SecurityCheck
{
    public static bool IsSafeToAccess(MemberInfo member)
    {
        // 检查是否具有特定特性
        if (member.GetCustomAttribute<DevOnlyAttribute>() != null)
        {
            #if !UNITY_EDITOR
            return false;
            #endif
        }
        
        // 检查访问级别
        if (member is MethodInfo method)
        {
            return !method.IsPrivate && !method.IsFamily;
        }
        
        return true;
    }
}
```

### 异常处理
```csharp
public class SafeReflection
{
    public static object SafeInvoke(MethodInfo method, object instance, 
        params object[] parameters)
    {
        try
        {
            if (!SecurityCheck.IsSafeToAccess(method))
            {
                throw new SecurityException("无权访问此方法");
            }
            
            return method.Invoke(instance, parameters);
        }
        catch (TargetInvocationException ex)
        {
            // unwrap真实异常
            throw ex.InnerException;
        }
        catch (Exception ex)
        {
            Debug.LogError($"反射调用失败: {ex.Message}");
            return null;
        }
    }
}
