### 异常类层次结构

// 常见异常类型
public class ExceptionExample
{
    public void DemonstrateExceptions()
    {
        try
        {
            // System.NullReferenceException
            object obj = null;
            obj.ToString();
            
            // System.IndexOutOfRangeException
            int[] arr = new int[5];
            int item = arr[10];
            
            // System.DivideByZeroException
            int result = 10 / 0;
            
            // System.InvalidCastException
            object str = "hello";
            int number = (int)str;
        }
        catch (Exception ex)
        {
            Debug.LogError($"捕获到异常：{ex.Message}");
        }
    }
}

### try-catch-finally结构
```csharp
public class ResourceManager
{
    private IDisposable resource;
    
    public void ProcessResource()
    {
        try
        {
            // 可能抛出异常的代码
            resource = AcquireResource();
            ProcessData();
        }
        catch (ResourceException ex)
        {
            // 处理特定异常
            Debug.LogError($"资源错误：{ex.Message}");
        }
        catch (Exception ex)
        {
            // 处理其他异常
            Debug.LogError($"未知错误：{ex.Message}");
        }
        finally
        {
            // 清理代码，总是执行
            resource?.Dispose();
        }
    }
}
```

## Unity中的异常处理

### MonoBehaviour异常处理
```csharp
public class GameController : MonoBehaviour
{
    private void OnEnable()
    {
        // 注册全局异常处理器
        Application.logMessageReceived += HandleException;
    }
    
    private void OnDisable()
    {
        Application.logMessageReceived -= HandleException;
    }
    
    private void HandleException(string condition, string stackTrace, LogType type)
    {
        if (type == LogType.Exception)
        {
            // 处理未捕获的异常
            SaveErrorLog(condition, stackTrace);
            ShowErrorUI();
        }
    }
    
    private void ShowErrorUI()
    {
        // 显示错误提示界面
    }
}
```

### 协程异常处理
```csharp
public class CoroutineManager : MonoBehaviour
{
    public IEnumerator SafeCoroutine(IEnumerator routine)
    {
        while (true)
        {
            try
            {
                if (!routine.MoveNext())
                    break;
            }
            catch (Exception ex)
            {
                Debug.LogError($"协程执行错误：{ex.Message}");
                break;
            }
            yield return routine.Current;
        }
    }
    
    // 使用示例
    private void Start()
    {
        StartCoroutine(SafeCoroutine(LoadAssets()));
    }
}
```

## 自定义异常

### 创建自定义异常
```csharp
public class GameException : Exception
{
    public GameException() { }
    
    public GameException(string message) 
        : base(message) { }
    
    public GameException(string message, Exception inner) 
        : base(message, inner) { }
}

public class ResourceNotFoundException : GameException
{
    public string ResourcePath { get; }
    
    public ResourceNotFoundException(string path)
        : base($"找不到资源：{path}")
    {
        ResourcePath = path;
    }
}
```

### 使用自定义异常
```csharp
public class AssetLoader
{
    public T LoadAsset<T>(string path) where T : UnityEngine.Object
    {
        try
        {
            var asset = Resources.Load<T>(path);
            if (asset == null)
                throw new ResourceNotFoundException(path);
                
            return asset;
        }
        catch (ResourceNotFoundException ex)
        {
            Debug.LogError($"资源加载失败：{ex.Message}");
            return null;
        }
    }
}
```

## 异常处理最佳实践

### 异常处理原则
```csharp
public class BestPractices
{
    // 1. 只捕获可以处理的异常
    public void HandleSpecificExceptions()
    {
        try
        {
            ProcessData();
        }
        catch (InvalidDataException ex)
        {
            // 处理数据错误
            LogError(ex);
        }
        catch (IOException ex)
        {
            // 处理IO错误
            LogError(ex);
        }
    }
    
    // 2. 避免空catch块
    public void BadPractice()
    {
        try
        {
            ProcessData();
        }
        catch
        {
            // 不要这样做！
        }
    }
    
    // 3. 正确使用finally
    public void ProcessFile(string path)
    {
        FileStream file = null;
        try
        {
            file = File.OpenRead(path);
            // 处理文件
        }
        finally
        {
            file?.Dispose();
        }
    }
}
```

### 性能考虑
```csharp
public class PerformanceAwareExceptions
{
    // 1. 避免用异常控制程序流程
    public bool TryGetValue(Dictionary<string, int> dict, string key, out int value)
    {
        // 好的做法
        return dict.TryGetValue(key, out value);
        
        // 不好的做法
        /*try
        {
            value = dict[key];
            return true;
        }
        catch
        {
            value = 0;
            return false;
        }*/
    }
    
    // 2. 异常过滤
    public void HandleExceptions()
    {
        try
        {
            ProcessData();
        }
        catch (Exception ex) when (ShouldHandle(ex))
        {
            // 处理异常
        }
    }
    
    private bool ShouldHandle(Exception ex)
    {
        return ex is InvalidOperationException || 
               ex is ArgumentException;
    }
}
```

### 日志记录
```csharp
public class LoggingSystem
{
    public void LogException(Exception ex, string context = "")
    {
        var errorInfo = new StringBuilder();
        errorInfo.AppendLine($"时间：{DateTime.Now}");
        errorInfo.AppendLine($"上下文：{context}");
        errorInfo.AppendLine($"异常类型：{ex.GetType().Name}");
        errorInfo.AppendLine($"异常信息：{ex.Message}");
        errorInfo.AppendLine($"堆栈跟踪：{ex.StackTrace}");
        
        // 写入日志文件
        string logPath = Path.Combine(Application.persistentDataPath, "error.log");
        File.AppendAllText(logPath, errorInfo.ToString());
        
        // 开发模式下显示详细信息
        #if UNITY_EDITOR
        Debug.LogError(errorInfo.ToString());
        #endif
    }
}
