# 异步编程

## async/await基础

### 基本用法
```csharp
public class AsyncBasics : MonoBehaviour
{
    // 异步方法声明
    public async Task LoadGameDataAsync()
    {
        Debug.Log("开始加载数据...");
        await Task.Delay(1000); // 模拟耗时操作
        Debug.Log("数据加载完成");
    }
    
    // 带返回值的异步方法
    public async Task<PlayerData> LoadPlayerDataAsync()
    {
        await Task.Delay(1000);
        return new PlayerData { Level = 1, Score = 0 };
    }
    
    // 在Unity中使用异步方法
    private async void Start()
    {
        try
        {
            PlayerData data = await LoadPlayerDataAsync();
            Debug.Log($"玩家等级: {data.Level}");
        }
        catch (Exception ex)
        {
            Debug.LogError($"加载失败: {ex.Message}");
        }
    }
}
```

### 异步流程控制
```csharp
public class AsyncFlow
{
    // 并行执行多个异步操作
    public async Task LoadAllDataAsync()
    {
        var audioTask = LoadAudioAsync();
        var textureTask = LoadTexturesAsync();
        var configTask = LoadConfigAsync();
        
        await Task.WhenAll(audioTask, textureTask, configTask);
        Debug.Log("所有资源加载完成");
    }
    
    // 异步操作超时处理
    public async Task<bool> LoadWithTimeoutAsync()
    {
        using var cts = new CancellationTokenSource(TimeSpan.FromSeconds(5));
        try
        {
            await LoadGameDataAsync(cts.Token);
            return true;
        }
        catch (OperationCanceledException)
        {
            Debug.Log("加载超时");
            return false;
        }
    }
}
```

## Unity中的异步操作

### 协程和异步
```csharp
public class UnityAsync : MonoBehaviour
{
    // 将协程转换为异步操作
    public async Task WaitForSecondsAsync(float seconds)
    {
        float startTime = Time.time;
        while (Time.time - startTime < seconds)
        {
            await Task.Yield();
        }
    }
    
    // 异步加载资源
    public async Task<GameObject> LoadPrefabAsync(string path)
    {
        var request = Resources.LoadAsync<GameObject>(path);
        while (!request.isDone)
        {
            await Task.Yield();
        }
        return request.asset as GameObject;
    }
    
    // 异步场景加载
    public async Task LoadSceneAsync(string sceneName)
    {
        var operation = SceneManager.LoadSceneAsync(sceneName);
        operation.allowSceneActivation = false;
        
        while (operation.progress < 0.9f)
        {
            await Task.Yield();
            UpdateLoadingProgress(operation.progress);
        }
        
        // 等待用户输入
        await WaitForUserInput();
        operation.allowSceneActivation = true;
    }
}
```

### 异步资源管理
```csharp
public class AssetManager
{
    private Dictionary<string, WeakReference> assetCache = 
        new Dictionary<string, WeakReference>();
        
    public async Task<T> LoadAssetAsync<T>(string path) where T : UnityEngine.Object
    {
        // 检查缓存
        if (assetCache.TryGetValue(path, out var weakRef))
        {
            if (weakRef.IsAlive)
            {
                return weakRef.Target as T;
            }
            assetCache.Remove(path);
        }
        
        // 异步加载
        var request = Resources.LoadAsync<T>(path);
        while (!request.isDone)
        {
            await Task.Yield();
        }
        
        // 缓存并返回
        var asset = request.asset as T;
        assetCache[path] = new WeakReference(asset);
        return asset;
    }
}
```

## 异步模式

### TAP (Task-based Asynchronous Pattern)
```csharp
public class NetworkManager
{
    // 异步网络请求
    public async Task<string> FetchDataAsync(string url)
    {
        using var client = new HttpClient();
        return await client.GetStringAsync(url);
    }
    
    // 异步操作进度报告
    public async Task DownloadFileAsync(string url, 
        IProgress<float> progress = null)
    {
        using var client = new HttpClient();
        using var response = await client.GetAsync(url, 
            HttpCompletionOption.ResponseHeadersRead);
        
        var totalBytes = response.Content.Headers.ContentLength ?? -1L;
        var readBytes = 0L;
        
        using var stream = await response.Content.ReadAsStreamAsync();
        var buffer = new byte[8192];
        
        while (true)
        {
            var read = await stream.ReadAsync(buffer, 0, buffer.Length);
            if (read == 0) break;
            
            readBytes += read;
            if (totalBytes != -1L)
            {
                progress?.Report((float)readBytes / totalBytes);
            }
        }
    }
}
```

### 异步事件处理
```csharp
public class AsyncEventHandler
{
    private TaskCompletionSource<bool> tcs;
    
    // 异步等待事件
    public async Task WaitForEventAsync()
    {
        tcs = new TaskCompletionSource<bool>();
        try
        {
            await tcs.Task;
        }
        finally
        {
            tcs = null;
        }
    }
    
    // 触发事件完成
    public void OnEventCompleted()
    {
        tcs?.SetResult(true);
    }
}
```

### 异常处理
```csharp
public class AsyncExceptionHandling
{
    // 全局异常处理
    public static async Task RunWithErrorHandlingAsync(Func<Task> action)
    {
        try
        {
            await action();
        }
        catch (Exception ex)
        {
            Debug.LogError($"异步操作异常: {ex}");
            // 可以在这里添加错误报告逻辑
        }
    }
    
    // 重试机制
    public static async Task<T> RetryAsync<T>(
        Func<Task<T>> operation, 
        int maxAttempts = 3)
    {
        for (int attempt = 1; attempt <= maxAttempts; attempt++)
        {
            try
            {
                return await operation();
            }
            catch (Exception) when (attempt < maxAttempts)
            {
                await Task.Delay(attempt * 1000); // 递增延迟
            }
        }
        return await operation(); // 最后一次尝试
    }
}
```

```csharp
public class AsyncMemoryManagement
{
    // 使用ValueTask减少分配
    public ValueTask<int> GetValueAsync()
    {
        if (HasCachedValue)
        {
            return new ValueTask<int>(cachedValue);
        }
        return new ValueTask<int>(LoadValueAsync());
    }
    
    // 异步资源释放
    public async Task ProcessDataAsync()
    {
        await using var resource = new AsyncResource();
        await resource.ProcessAsync();
    }
}

public class AsyncResource : IAsyncDisposable
{
    public async ValueTask DisposeAsync()
    {
        await CleanupAsync();
    }
