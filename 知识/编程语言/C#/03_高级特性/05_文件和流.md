# 文件和流

## 文件操作

### 基本文件操作
```csharp
public class FileOperations
{
    // 读取文件
    public string ReadTextFile(string path)
    {
        if (!File.Exists(path))
            throw new FileNotFoundException("文件不存在", path);
            
        return File.ReadAllText(path);
    }
    
    // 写入文件
    public void WriteTextFile(string path, string content)
    {
        File.WriteAllText(path, content);
    }
    
    // 异步文件操作
    public async Task<string> ReadTextFileAsync(string path)
    {
        using var reader = new StreamReader(path);
        return await reader.ReadToEndAsync();
    }
    
    // 文件信息
    public void GetFileInfo(string path)
    {
        var fileInfo = new FileInfo(path);
        Debug.Log($"文件大小: {fileInfo.Length}");
        Debug.Log($"创建时间: {fileInfo.CreationTime}");
        Debug.Log($"最后修改时间: {fileInfo.LastWriteTime}");
    }
}
```

### 目录操作
```csharp
public class DirectoryOperations
{
    // 创建目录
    public void CreateDirectory(string path)
    {
        if (!Directory.Exists(path))
        {
            Directory.CreateDirectory(path);
        }
    }
    
    // 遍历目录
    public IEnumerable<string> GetFiles(string path, string searchPattern = "*.*")
    {
        return Directory.GetFiles(path, searchPattern, 
            SearchOption.AllDirectories);
    }
    
    // 复制目录
    public void CopyDirectory(string sourcePath, string targetPath)
    {
        if (!Directory.Exists(targetPath))
        {
            Directory.CreateDirectory(targetPath);
        }
        
        foreach (string file in Directory.GetFiles(sourcePath))
        {
            string fileName = Path.GetFileName(file);
            string destFile = Path.Combine(targetPath, fileName);
            File.Copy(file, destFile, true);
        }
        
        foreach (string dir in Directory.GetDirectories(sourcePath))
        {
            string dirName = Path.GetFileName(dir);
            string destDir = Path.Combine(targetPath, dirName);
            CopyDirectory(dir, destDir);
        }
    }
}
```

## 流操作

### 内存流
```csharp
public class MemoryStreamExample
{
    // 使用MemoryStream处理数据
    public byte[] ProcessData(byte[] input)
    {
        using var memoryStream = new MemoryStream();
        using (var writer = new BinaryWriter(memoryStream))
        {
            writer.Write(input.Length);
            writer.Write(input);
        }
        
        return memoryStream.ToArray();
    }
    
    // 序列化对象到内存流
    public byte[] SerializeObject<T>(T obj)
    {
        using var memoryStream = new MemoryStream();
        var formatter = new BinaryFormatter();
        formatter.Serialize(memoryStream, obj);
        return memoryStream.ToArray();
    }
}
```

```csharp
public class FileStreamOperations
{
    // 大文件读取
    public async Task ProcessLargeFileAsync(string path)
    {
        using var fileStream = new FileStream(path, FileMode.Open, 
            FileAccess.Read);
        var buffer = new byte[8192];
        int bytesRead;
        
        while ((bytesRead = await fileStream.ReadAsync(buffer, 0, 
            buffer.Length)) > 0)
        {
            await ProcessDataChunkAsync(buffer, bytesRead);
        }
    }
    
    // 文件复制带进度
    public async Task CopyFileWithProgressAsync(string sourcePath, 
        string targetPath, IProgress<float> progress)
    {
        using var sourceStream = new FileStream(sourcePath, FileMode.Open);
        using var targetStream = new FileStream(targetPath, FileMode.Create);
        
        var buffer = new byte[8192];
        long totalBytes = sourceStream.Length;
        long copiedBytes = 0;
        int bytesRead;
        
        while ((bytesRead = await sourceStream.ReadAsync(buffer, 0, 
            buffer.Length)) > 0)
        {
            await targetStream.WriteAsync(buffer, 0, bytesRead);
            copiedBytes += bytesRead;
            progress?.Report((float)copiedBytes / totalBytes);
        }
    }
}
```

## Unity文件操作

### 持久化数据
```csharp
public class GameDataManager : MonoBehaviour
{
    private string SavePath => 
        Path.Combine(Application.persistentDataPath, "save.dat");
    
    // 保存游戏数据
    public async Task SaveGameDataAsync(GameData data)
    {
        var json = JsonUtility.ToJson(data);
        await File.WriteAllTextAsync(SavePath, json);
    }
    
    // 加载游戏数据
    public async Task<GameData> LoadGameDataAsync()
    {
        if (!File.Exists(SavePath))
            return new GameData();
            
        var json = await File.ReadAllTextAsync(SavePath);
        return JsonUtility.FromJson<GameData>(json);
    }
    
    // 自动保存系统
    private async void AutoSave()
    {
        while (true)
        {
            await Task.Delay(TimeSpan.FromMinutes(5));
            await SaveGameDataAsync(GetCurrentGameData());
        }
    }
}
```

### 资源管理
```csharp
public class ResourceManager
{
    // 异步加载资源
    public async Task<T> LoadAssetAsync<T>(string path) 
        where T : UnityEngine.Object
    {
        var request = Resources.LoadAsync<T>(path);
        while (!request.isDone)
        {
            await Task.Yield();
        }
        return request.asset as T;
    }
    
    // 资源包管理
    public async Task<AssetBundle> LoadAssetBundleAsync(string path)
    {
        using var stream = new FileStream(path, FileMode.Open);
        var request = AssetBundle.LoadFromStreamAsync(stream);
        while (!request.isDone)
        {
            await Task.Yield();
        }
        return request.assetBundle;
    }
}
```

## 安全性和性能

### 安全文件操作
```csharp
public class SecureFileOperations
{
    // 安全写入文件
    public async Task SaveSecurelyAsync(string path, string content)
    {
        var tempPath = path + ".tmp";
        var backupPath = path + ".bak";
        
        try
        {
            // 写入临时文件
            await File.WriteAllTextAsync(tempPath, content);
            
            // 创建备份
            if (File.Exists(path))
            {
                File.Copy(path, backupPath, true);
            }
            
            // 替换原文件
            File.Move(tempPath, path, true);
            
            // 清理备份
            if (File.Exists(backupPath))
            {
                File.Delete(backupPath);
            }
        }
        catch
        {
            // 恢复备份
            if (File.Exists(backupPath))
            {
                File.Copy(backupPath, path, true);
            }
            throw;
        }
        finally
        {
            // 清理临时文件
            if (File.Exists(tempPath))
                File.Delete(tempPath);
            if (File.Exists(backupPath))
                File.Delete(backupPath);
        }
    }
}
```
