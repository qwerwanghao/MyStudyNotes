# 多线程编程

## 线程基础

### 创建和启动线程
```csharp
public class ThreadExample
{
    // 基本线程创建
    public void BasicThread()
    {
        Thread thread = new Thread(DoWork);
        thread.Start();
    }
    
    // 带参数的线程
    public void ParameterizedThread()
    {
        Thread thread = new Thread(param => 
        {
            string message = param as string;
            Debug.Log(message);
        });
        thread.Start("Hello from thread!");
    }
    
    private void DoWork()
    {
        Debug.Log("工作线程执行中...");
        Thread.Sleep(1000);
        Debug.Log("工作完成");
    }
}
```

### 线程生命周期
```csharp
public class ThreadLifecycle
{
    private Thread workerThread;
    
    public void StartThread()
    {
        workerThread = new Thread(() =>
        {
            try
            {
                while (!workerThread.IsBackground)
                {
                    // 执行工作
                    Thread.Sleep(100);
                }
            }
            catch (ThreadInterruptedException)
            {
                Debug.Log("线程被中断");
            }
        });
        
        workerThread.Start();
    }
    
    public void StopThread()
    {
        if (workerThread != null && workerThread.IsAlive)
        {
            workerThread.Interrupt();
            workerThread.Join(1000); // 等待线程结束
        }
    }
}
```

## 线程同步

### 锁定机制
```csharp
public class SynchronizationExample
{
    private readonly object lockObject = new object();
    private int sharedResource;
    
    // 使用lock关键字
    public void SafeIncrement()
    {
        lock (lockObject)
        {
            sharedResource++;
        }
    }
    
    // 使用Monitor类
    public void SafeDecrement()
    {
        Monitor.Enter(lockObject);
        try
        {
            sharedResource--;
        }
        finally
        {
            Monitor.Exit(lockObject);
        }
    }
}
```

## Unity中的多线程

### 线程安全考虑
```csharp
public class UnityThreading : MonoBehaviour
{
    private Thread backgroundThread;
    private readonly object dataLock = new object();
    private List<Vector3> processedData = new List<Vector3>();
    
    private void ProcessDataInBackground()
    {
        backgroundThread = new Thread(() =>
        {
            while (true)
            {
                Vector3[] data = GenerateData();
                lock (dataLock)
                {
                    processedData.Clear();
                    processedData.AddRange(data);
                }
                Thread.Sleep(100);
            }
        });
        
        backgroundThread.IsBackground = true;
        backgroundThread.Start();
    }
    
    private void Update()
    {
        lock (dataLock)
        {
            // 在主线程中安全访问数据
            foreach (var point in processedData)
            {
                // 使用处理后的数据
            }
        }
    }
}
```

### 与主线程通信
```csharp
public class ThreadCommunication : MonoBehaviour
{
    private readonly Queue<Action> executeOnMainThread = new Queue<Action>();
    private readonly object queueLock = new object();
    
    private void Update()
    {
        // 执行队列中的动作
        lock (queueLock)
        {
            while (executeOnMainThread.Count > 0)
            {
                executeOnMainThread.Dequeue().Invoke();
            }
        }
    }
    
    public void EnqueueAction(Action action)
    {
        lock (queueLock)
        {
            executeOnMainThread.Enqueue(action);
        }
    }
    
    // 在工作线程中使用
    private void WorkerThread()
    {
        // 在后台处理数据
        Vector3 result = ProcessData();
        
        // 将更新UI的操作排队到主线程
        EnqueueAction(() =>
        {
            UpdateUI(result);
        });
    }
}
```

## 线程池

### 使用ThreadPool
```csharp
public class ThreadPoolExample
{
    public void QueueWorkItem()
    {
        // 简单的工作项
        ThreadPool.QueueUserWorkItem(state =>
        {
            Debug.Log("在线程池中执行");
        });
        
        // 带参数的工作项
        ThreadPool.QueueUserWorkItem(state =>
        {
            string data = state as string;
            ProcessData(data);
        }, "sample data");
    }
    
    // 使用线程池等待回调
    public void ProcessDataAsync()
    {
        ManualResetEvent completedEvent = new ManualResetEvent(false);
        
        ThreadPool.QueueUserWorkItem(state =>
        {
            try
            {
                // 执行长时间操作
                Thread.Sleep(1000);
            }
            finally
            {
                completedEvent.Set();
            }
        });
        
        // 等待完成
        completedEvent.WaitOne(2000);
    }
}
```

### 自定义线程池
```csharp
public class CustomThreadPool
{
    private readonly Queue<Action> workItems = new Queue<Action>();
    private readonly List<Thread> workers;
    private readonly object queueLock = new object();
    private bool isShutdown;
    
    public CustomThreadPool(int workerCount)
    {
        workers = new List<Thread>();
        
        for (int i = 0; i < workerCount; i++)
        {
            var worker = new Thread(WorkerThread)
            {
                IsBackground = true
            };
            workers.Add(worker);
            worker.Start();
        }
    }
    
    private void WorkerThread()
    {
        while (!isShutdown)
        {
            Action workItem = null;
            
            lock (queueLock)
            {
                if (workItems.Count > 0)
                {
                    workItem = workItems.Dequeue();
                }
            }
            
            if (workItem != null)
            {
                try
                {
                    workItem();
                }
                catch (Exception ex)
                {
                    Debug.LogError($"工作项执行错误: {ex}");
                }
            }
            else
            {
                Thread.Sleep(100);
            }
        }
    }
    
    public void QueueWorkItem(Action workItem)
    {
        if (isShutdown)
            throw new InvalidOperationException("线程池已关闭");
            
        lock (queueLock)
        {
            workItems.Enqueue(workItem);
        }
    }
    
    public void Shutdown()
    {
        isShutdown = true;
        foreach (var worker in workers)
        {
            worker.Join();
        }
    }
}
```

### 性能优化
```csharp
public class ThreadingPerformance
{
    // 避免过多的线程创建
    private static readonly object syncRoot = new object();
    private static Thread workerThread;
    
    public static Thread GetWorkerThread()
    {
        if (workerThread == null)
        {
            lock (syncRoot)
            {
                if (workerThread == null)
                {
                    workerThread = new Thread(WorkerMethod);
                    workerThread.Start();
                }
            }
        }
        return workerThread;
    }
    
    // 使用线程局部存储
    private static ThreadLocal<Random> random = 
        new ThreadLocal<Random>(() => new Random());
        
    public int GetRandomNumber()
    {
        return random.Value.Next();
    }
}
```
