# Unity游戏运营

## 1. 数据分析

### 1.1 数据收集

```csharp
public class AnalyticsManager : MonoBehaviour
{
    private const string API_ENDPOINT = "http://analytics.game.com/collect";
    
    // 记录玩家行为
    public void TrackPlayerAction(string actionType, Dictionary<string, object> parameters)
    {
        AnalyticsData data = new AnalyticsData
        {
            userId = UserManager.CurrentUserId,
            timestamp = DateTime.UtcNow,
            eventType = actionType,
            parameters = parameters,
            deviceInfo = new DeviceInfo
            {
                platform = Application.platform.ToString(),
                deviceModel = SystemInfo.deviceModel,
                operatingSystem = SystemInfo.operatingSystem
            }
        };
        
        StartCoroutine(SendAnalyticsData(data));
    }
    
    // 记录关卡数据
    public void TrackLevelData(string levelId, bool completed, float timeSpent)
    {
        var parameters = new Dictionary<string, object>
        {
            {"level_id", levelId},
            {"completed", completed},
            {"time_spent", timeSpent},
            {"attempts", PlayerStats.GetLevelAttempts(levelId)}
        };
        
        TrackPlayerAction("level_completed", parameters);
    }
    
    private IEnumerator SendAnalyticsData(AnalyticsData data)
    {
        string json = JsonUtility.ToJson(data);
        byte[] bodyRaw = Encoding.UTF8.GetBytes(json);
        
        using (UnityWebRequest www = new UnityWebRequest(API_ENDPOINT, "POST"))
        {
            www.uploadHandler = new UploadHandlerRaw(bodyRaw);
            www.downloadHandler = new DownloadHandlerBuffer();
            www.SetRequestHeader("Content-Type", "application/json");
            
            yield return www.SendWebRequest();
            
            if (www.result != UnityWebRequest.Result.Success)
            {
                Debug.LogError($"Analytics数据发送失败: {www.error}");
            }
        }
    }
}
```

### 1.2 数据分析工具

```csharp
public class AnalyticsTools
{
    // 用户留存分析
    public class RetentionAnalysis
    {
        public Dictionary<string, float> CalculateRetention(
            List<UserSession> sessions, 
            int[] days)
        {
            var retention = new Dictionary<string, float>();
            
            foreach (int day in days)
            {
                int totalUsers = sessions.Select(s => s.userId).Distinct().Count();
                int retainedUsers = sessions.Where(s => 
                    s.timestamp >= DateTime.Now.AddDays(-day))
                    .Select(s => s.userId)
                    .Distinct()
                    .Count();
                    
                retention[$"Day{day}"] = (float)retainedUsers / totalUsers;
            }
            
            return retention;
        }
    }
    
    // 用户行为分析
    public class BehaviorAnalysis
    {
        public Dictionary<string, int> AnalyzeUserBehavior(
            List<UserAction> actions,
            string actionType)
        {
            return actions
                .Where(a => a.type == actionType)
                .GroupBy(a => a.parameters["item_id"].ToString())
                .ToDictionary(
                    g => g.Key,
                    g => g.Count()
                );
        }
    }
}
```

### 2.1 活动系统

```csharp
public class EventSystem : MonoBehaviour
{
    private List<GameEvent> activeEvents = new List<GameEvent>();
    
    // 活动基类
    public abstract class GameEvent
    {
        public string eventId;
        public string eventName;
        public DateTime startTime;
        public DateTime endTime;
        public Dictionary<string, object> parameters;
        
        public abstract bool IsEventConditionMet(string userId);
        public abstract void GrantEventReward(string userId);
    }
    
    // 签到活动示例
    public class SignInEvent : GameEvent
    {
        private Dictionary<string, List<DateTime>> userSignInRecords = 
            new Dictionary<string, List<DateTime>>();
            
        public override bool IsEventConditionMet(string userId)
        {
            if (!userSignInRecords.ContainsKey(userId))
                return true;
                
            var lastSignIn = userSignInRecords[userId].LastOrDefault();
            return (DateTime.Now - lastSignIn).TotalHours >= 24;
        }
        
        public override void GrantEventReward(string userId)
        {
            if (!userSignInRecords.ContainsKey(userId))
                userSignInRecords[userId] = new List<DateTime>();
                
            userSignInRecords[userId].Add(DateTime.Now);
            
            // 发放奖励
            var rewards = new Dictionary<string, int>
            {
                {"gold", 100},
                {"gem", 10}
            };
            
            RewardManager.GrantRewards(userId, rewards);
        }
    }
    
    // 活动管理
    public void UpdateEvents()
    {
        // 移除过期活动
        activeEvents.RemoveAll(e => e.endTime < DateTime.Now);
        
        // 检查新活动
        var newEvents = EventConfigManager.GetNewEvents();
        activeEvents.AddRange(newEvents);
        
        // 通知客户端
        EventManager.Dispatch("EventsUpdated", activeEvents);
    }
}
```

### 2.2 公告系统

```csharp
public class AnnouncementSystem : MonoBehaviour
{
    private List<Announcement> announcements = new List<Announcement>();
    
    // 公告数据结构
    public class Announcement
    {
        public string id;
        public string title;
        public string content;
        public DateTime startTime;
        public DateTime endTime;
        public AnnouncementType type;
        public Dictionary<string, string> parameters;
    }
    
    // 发布公告
    public void PublishAnnouncement(Announcement announcement)
    {
        announcements.Add(announcement);
        
        // 推送给在线玩家
        var onlinePlayers = NetworkManager.GetOnlinePlayers();
        foreach (var player in onlinePlayers)
        {
            SendAnnouncementToPlayer(player.userId, announcement);
        }
    }
    
    // 获取活动公告
    public List<Announcement> GetActiveAnnouncements()
    {
        return announcements
            .Where(a => a.startTime <= DateTime.Now && a.endTime > DateTime.Now)
            .OrderByDescending(a => a.startTime)
            .ToList();
    }
}
```

## 3. 商城系统

### 3.1 商品管理

```csharp
public class ShopSystem : MonoBehaviour
{
    private List<ShopItem> shopItems = new List<ShopItem>();
    
    // 商品数据结构
    public class ShopItem
    {
        public string id;
        public string name;
        public ItemType type;
        public int price;
        public string currency;
        public Dictionary<string, int> contents;
        public DateTime startTime;
        public DateTime endTime;
        public int limitPerUser;
    }
    
    // 购买商品
    public bool PurchaseItem(string userId, string itemId)
    {
        var item = shopItems.Find(i => i.id == itemId);
        if (item == null)
            return false;
            
        // 检查限购
        if (!CheckPurchaseLimit(userId, item))
            return false;
            
        // 检查货币
        if (!CheckCurrency(userId, item))
            return false;
            
        // 扣除货币
        CurrencyManager.DeductCurrency(userId, item.currency, item.price);
        
        // 发放商品
        InventoryManager.AddItems(userId, item.contents);
        
        // 记录购买
        RecordPurchase(userId, item);
        
        return true;
    }
    
    // 更新商店
    public void UpdateShop()
    {
        // 移除过期商品
        shopItems.RemoveAll(i => i.endTime < DateTime.Now);
        
        // 添加新商品
        var newItems = ShopConfigManager.GetNewItems();
        shopItems.AddRange(newItems);
        
        // 通知客户端
        EventManager.Dispatch("ShopUpdated", shopItems);
    }
}
```

### 3.2 支付系统

```csharp
public class PaymentSystem : MonoBehaviour
{
    // 支付请求
    public async Task<bool> ProcessPayment(
        string userId, 
        string productId, 
        string platform)
    {
        try
        {
            // 创建订单
            var order = await CreateOrder(userId, productId);
            
            // 根据平台调用不同支付接口
            switch (platform.ToLower())
            {
                case "ios":
                    return await ProcessIAPPayment(order);
                case "android":
                    return await ProcessGooglePayment(order);
                case "wechat":
                    return await ProcessWeChatPayment(order);
                default:
                    throw new NotSupportedException($"不支持的支付平台: {platform}");
            }
        }
        catch (Exception e)
        {
            Debug.LogError($"支付处理失败: {e.Message}");
            return false;
        }
    }
    
    // 支付回调
    public void HandlePaymentCallback(PaymentCallback callback)
    {
        if (callback.status == "success")
        {
            // 查询订单
            var order = OrderManager.GetOrder(callback.orderId);
            if (order == null)
                return;
                
            // 发放商品
            var product = ProductManager.GetProduct(order.productId);
            InventoryManager.AddItems(order.userId, product.contents);
            
            // 更新订单状态
            OrderManager.UpdateOrderStatus(order.id, OrderStatus.Completed);
            
            // 通知客户端
            EventManager.Dispatch("PaymentCompleted", order);
        }
    }
}
```

```csharp
public class UserSystem : MonoBehaviour

        profile.lastLoginTime = DateTime.Now;
        await UserDatabase.SaveUser(profile);
        
        // 加载用户数据
        await LoadUserData(userId);
        
        return profile;
    }
```

### 4.2 权限管理

```csharp
public class PermissionSystem : MonoBehaviour
{
    private Dictionary<string, HashSet<string>> userPermissions = 
        new Dictionary<string, HashSet<string>>();
        
    // 检查权限
    public bool CheckPermission(string userId, string permission)
    {
        if (!userPermissions.ContainsKey(userId))
            return false;
            
        return userPermissions[userId].Contains(permission);
    }
    
    // 授予权限
    public void GrantPermission(string userId, string permission)
    {
        if (!userPermissions.ContainsKey(userId))
            userPermissions[userId] = new HashSet<string>();
            
        userPermissions[userId].Add(permission);
    }
    
    // 移除权限
    public void RevokePermission(string userId, string permission)
    {
        if (userPermissions.ContainsKey(userId))
        {
            userPermissions[userId].Remove(permission);
        }
    }
}
```
