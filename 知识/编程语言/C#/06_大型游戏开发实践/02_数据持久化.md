# Unity数据持久化

## 1. 数据持久化概述

### 1.1 什么是数据持久化

数据持久化是将游戏运行时的数据保存到存储设备中，以便在下次运行时可以恢复数据的过程。主要包括：

- 游戏存档数据
- 玩家设置
- 游戏进度
- 成就系统数据
- 排行榜数据

### 1.2 持久化方案选择

根据数据特点选择合适的持久化方案：

- **PlayerPrefs**：适用于小型数据存储
- **JSON/XML**：适用于结构化数据存储
- **二进制序列化**：适用于大型数据和需要加密的数据
- **SQLite**：适用于需要查询和管理的结构化数据
- **云存储**：适用于需要跨平台同步的数据

```csharp
public class PlayerPrefsManager
{
    // 保存基础类型数据
    public static void SaveBasicData()
    {
        PlayerPrefs.SetString("PlayerName", "Hero");
        PlayerPrefs.SetInt("PlayerLevel", 10);
        PlayerPrefs.SetFloat("PlayerHealth", 100.0f);
        PlayerPrefs.Save();
    }
    
    // 加载基础类型数据
    public static void LoadBasicData()
    {
        string playerName = PlayerPrefs.GetString("PlayerName", "Default");
        int playerLevel = PlayerPrefs.GetInt("PlayerLevel", 1);
        float playerHealth = PlayerPrefs.GetFloat("PlayerHealth", 100.0f);
    }
    
    // 保存复杂类型数据
    public static void SaveComplexData(PlayerData data)
    {
        string json = JsonUtility.ToJson(data);
        PlayerPrefs.SetString("PlayerData", json);
        PlayerPrefs.Save();
    }
    
    // 加载复杂类型数据
    public static PlayerData LoadComplexData()
    {
        string json = PlayerPrefs.GetString("PlayerData", "");
        return string.IsNullOrEmpty(json) ? new PlayerData() : JsonUtility.FromJson<PlayerData>(json);
    }
}
```

## 3. JSON数据存储

### 3.1 基于文件的JSON存储

```csharp
public class JsonDataManager
{
    private static readonly string SAVE_PATH = Application.persistentDataPath + "/saves/";
    
    // 保存数据到JSON文件
    public static void SaveToJson<T>(T data, string fileName)
    {
        if (!Directory.Exists(SAVE_PATH))
        {
            Directory.CreateDirectory(SAVE_PATH);
        }
        
        string json = JsonUtility.ToJson(data, true);
        string filePath = SAVE_PATH + fileName + ".json";
        
        try
        {
            File.WriteAllText(filePath, json);
            Debug.Log($"数据已保存到: {filePath}");
        }
        catch (Exception e)
        {
            Debug.LogError($"保存数据失败: {e.Message}");
        }
    }
    
    // 从JSON文件加载数据
    public static T LoadFromJson<T>(string fileName) where T : new()
    {
        string filePath = SAVE_PATH + fileName + ".json";
        
        try
        {
            if (File.Exists(filePath))
            {
                string json = File.ReadAllText(filePath);
                return JsonUtility.FromJson<T>(json);
            }
        }
        catch (Exception e)
        {
            Debug.LogError($"加载数据失败: {e.Message}");
        }
        
        return new T();
    }
}
```

## 4. 二进制序列化

### 4.1 二进制数据管理器

```csharp
public class BinaryDataManager
{
    private static readonly string SAVE_PATH = Application.persistentDataPath + "/binary_saves/";
    
    // 保存数据到二进制文件
    public static void SaveToBinary<T>(T data, string fileName)
    {
        if (!Directory.Exists(SAVE_PATH))
        {
            Directory.CreateDirectory(SAVE_PATH);
        }
        
        string filePath = SAVE_PATH + fileName + ".bin";
        
        try
        {
            using (FileStream fs = new FileStream(filePath, FileMode.Create))
            {
                BinaryFormatter formatter = new BinaryFormatter();
                formatter.Serialize(fs, data);
            }
            Debug.Log($"数据已保存到: {filePath}");
        }
        catch (Exception e)
        {
            Debug.LogError($"保存数据失败: {e.Message}");
        }
    }
    
    // 从二进制文件加载数据
    public static T LoadFromBinary<T>(string fileName) where T : new()
    {
        string filePath = SAVE_PATH + fileName + ".bin";
        
        try
        {
            if (File.Exists(filePath))
            {
                using (FileStream fs = new FileStream(filePath, FileMode.Open))
                {
                    BinaryFormatter formatter = new BinaryFormatter();
                    return (T)formatter.Deserialize(fs);
                }
            }
        }
        catch (Exception e)
        {
            Debug.LogError($"加载数据失败: {e.Message}");
        }
        
        return new T();
    }
}
```

### 4.2 数据加密

```csharp
public class DataEncryption
{
    private static readonly byte[] KEY = new byte[] { 1, 2, 3, 4, 5, 6, 7, 8 };
    private static readonly byte[] IV = new byte[] { 1, 2, 3, 4, 5, 6, 7, 8 };
    
    // 加密数据
    public static byte[] EncryptData(byte[] data)
    {
        using (DESCryptoServiceProvider des = new DESCryptoServiceProvider())
        {
            using (MemoryStream ms = new MemoryStream())
            {
                using (CryptoStream cs = new CryptoStream(ms, des.CreateEncryptor(KEY, IV), CryptoStreamMode.Write))
                {
                    cs.Write(data, 0, data.Length);
                    cs.FlushFinalBlock();
                    return ms.ToArray();
                }
            }
        }
    }
    
    // 解密数据
    public static byte[] DecryptData(byte[] data)
    {
        using (DESCryptoServiceProvider des = new DESCryptoServiceProvider())
        {
            using (MemoryStream ms = new MemoryStream())
            {
                using (CryptoStream cs = new CryptoStream(ms, des.CreateDecryptor(KEY, IV), CryptoStreamMode.Write))
                {
                    cs.Write(data, 0, data.Length);
                    cs.FlushFinalBlock();
                    return ms.ToArray();
                }
            }
        }
    }
}
```

## 5. SQLite数据库

### 5.1 数据库管理器

```csharp
public class SQLiteManager
{
    private static SQLiteConnection connection;
    private static readonly string DB_PATH = Application.persistentDataPath + "/game.db";
    
    // 初始化数据库
    public static void InitDatabase()
    {
        connection = new SQLiteConnection($"Data Source={DB_PATH}");
        connection.Open();
        
        // 创建表
        using (var command = connection.CreateCommand())
        {
            command.CommandText = @"
                CREATE TABLE IF NOT EXISTS Players (
                    Id INTEGER PRIMARY KEY,
                    Name TEXT NOT NULL,
                    Level INTEGER,
                    Experience INTEGER
                );
                
                CREATE TABLE IF NOT EXISTS Inventory (
                    Id INTEGER PRIMARY KEY,
                    PlayerId INTEGER,
                    ItemId INTEGER,
                    Count INTEGER,
                    FOREIGN KEY(PlayerId) REFERENCES Players(Id)
                );";
            command.ExecuteNonQuery();
        }
    }
    
    // 保存玩家数据
    public static void SavePlayer(PlayerData player)
    {
        using (var command = connection.CreateCommand())
        {
            command.CommandText = @"
                INSERT OR REPLACE INTO Players (Id, Name, Level, Experience)
                VALUES (@Id, @Name, @Level, @Experience);";
                
            command.Parameters.AddWithValue("@Id", player.id);
            command.Parameters.AddWithValue("@Name", player.playerName);
            command.Parameters.AddWithValue("@Level", player.level);
            command.Parameters.AddWithValue("@Experience", player.experience);
            
            command.ExecuteNonQuery();
        }
    }
    
    // 加载玩家数据
    public static PlayerData LoadPlayer(int playerId)
    {
        using (var command = connection.CreateCommand())
        {
            command.CommandText = "SELECT * FROM Players WHERE Id = @Id;";
            command.Parameters.AddWithValue("@Id", playerId);
            
            using (var reader = command.ExecuteReader())
            {
                if (reader.Read())
                {
                    return new PlayerData
                    {
                        id = reader.GetInt32(0),
                        playerName = reader.GetString(1),
                        level = reader.GetInt32(2),
                        experience = reader.GetInt32(3)
                    };
                }
            }
        }
        return null;
    }
}
```

## 6. 云存储

### 6.1 云存储管理器

```csharp
public class CloudSaveManager
{
    private const string API_URL = "https://api.example.com/save";
    
    // 上传存档
    public static async Task<bool> UploadSaveData(GameSaveData saveData)
    {
        try
        {
            string json = JsonUtility.ToJson(saveData);
            using (UnityWebRequest request = UnityWebRequest.Post(API_URL, json))
            {
                request.SetRequestHeader("Content-Type", "application/json");
                request.SetRequestHeader("Authorization", "Bearer " + PlayerToken);
                
                await request.SendWebRequest();
                
                if (request.result == UnityWebRequest.Result.Success)
                {
                    Debug.Log("存档上传成功");
                    return true;
                }
            }
        }
        catch (Exception e)
        {
            Debug.LogError($"存档上传失败: {e.Message}");
        }
        return false;
    }
    
    // 下载存档
    public static async Task<GameSaveData> DownloadSaveData()
    {
        try
        {
            using (UnityWebRequest request = UnityWebRequest.Get(API_URL))
            {
                request.SetRequestHeader("Authorization", "Bearer " + PlayerToken);
                
                await request.SendWebRequest();
                
                if (request.result == UnityWebRequest.Result.Success)
                {
                    string json = request.downloadHandler.text;
                    return JsonUtility.FromJson<GameSaveData>(json);
                }
            }
        }
        catch (Exception e)
        {
            Debug.LogError($"存档下载失败: {e.Message}");
        }
        return null;
    }
}
```

### 6.2 存档同步

```csharp
public class SaveSynchronizer
{
    // 检查存档版本
    public static async Task<bool> CheckSaveVersion()
    {
        GameSaveData localSave = JsonDataManager.LoadFromJson<GameSaveData>("save");
        GameSaveData cloudSave = await CloudSaveManager.DownloadSaveData();
        
        if (cloudSave != null && cloudSave.saveTime > localSave.saveTime)
        {
            return true; // 云存档更新
        }
        return false;
    }
    
    // 合并存档
    public static async Task MergeSaveData()
    {
        GameSaveData localSave = JsonDataManager.LoadFromJson<GameSaveData>("save");
        GameSaveData cloudSave = await CloudSaveManager.DownloadSaveData();
        
        if (cloudSave != null)
        {
            // 合并逻辑
            GameSaveData mergedSave = new GameSaveData();
            mergedSave.playerData = cloudSave.saveTime > localSave.saveTime ? 
                cloudSave.playerData : localSave.playerData;
            
            // 保存合并后的存档
            JsonDataManager.SaveToJson(mergedSave, "save");
            await CloudSaveManager.UploadSaveData(mergedSave);
        }
    }
}
```
