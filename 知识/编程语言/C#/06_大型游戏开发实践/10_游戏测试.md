# Unity游戏测试

## 1. 单元测试

### 1.1 测试框架设置

```csharp
using NUnit.Framework;
using UnityEngine;
using UnityEngine.TestTools;

public class UnitTestExample
{
    [Test]
    public void SimpleTest()
    {
        // 准备测试数据
        int result = Calculator.Add(2, 3);
        
        // 验证结果
        Assert.AreEqual(5, result);
    }
    
    [UnityTest]
    public IEnumerator AsyncTest()
    {
        // 创建游戏对象
        GameObject gameObject = new GameObject();
        var component = gameObject.AddComponent<TestComponent>();
        
        // 等待一帧
        yield return null;
        
        // 验证组件状态
        Assert.IsTrue(component.IsInitialized);
    }
}
```

## 2. 集成测试

### 2.1 场景测试

```csharp
public class SceneTest : MonoBehaviour
{
    [Test]
    public void TestSceneLoad()
    {
        // 加载测试场景
        SceneManager.LoadScene("TestScene");
        
        // 等待场景加载完成
        yield return new WaitForSeconds(1);
        
        // 验证场景对象
        var player = GameObject.FindWithTag("Player");
        Assert.IsNotNull(player);
        
        var enemies = GameObject.FindGameObjectsWithTag("Enemy");
        Assert.Greater(enemies.Length, 0);
    }
    
    [Test]
    public void TestGameFlow()
    {
        // 初始化游戏状态
        GameManager.Instance.StartGame();
        
        // 模拟游戏流程
        yield return new WaitForSeconds(5);
        
        // 验证游戏状态
        Assert.AreEqual(GameState.Playing, GameManager.Instance.CurrentState);
        Assert.Greater(ScoreManager.Instance.CurrentScore, 0);
    }
}
```

## 3. 性能测试

### 3.1 性能分析器

```csharp
public class PerformanceTest : MonoBehaviour
{
    private Profiler profiler;
    
    [Test]
    public void TestResourceLoading()
    {
        Profiler.BeginSample("Resource Loading Test");
        
        // 加载测试资源
        for (int i = 0; i < 100; i++)
        {
            var prefab = Resources.Load<GameObject>("TestPrefab");
            Instantiate(prefab);
        }
        
        Profiler.EndSample();
        
        // 验证性能指标
        Assert.Less(Time.deltaTime, 0.033f); // 保持30FPS
    }
    
    [Test]
    public void TestMemoryUsage()
    {
        long beforeMemory = Profiler.GetTotalAllocatedMemoryLong();
        
        // 执行内存密集操作
        List<GameObject> objects = new List<GameObject>();
        for (int i = 0; i < 1000; i++)
        {
            objects.Add(new GameObject());
        }
        
        long afterMemory = Profiler.GetTotalAllocatedMemoryLong();
        
        // 清理
        foreach (var obj in objects)
        {
            Destroy(obj);
        }
        
        // 验证内存使用
        Assert.Less(afterMemory - beforeMemory, 10 * 1024 * 1024); // 小于10MB
    }
}
```

### 3.2 压力测试

```csharp
public class StressTest : MonoBehaviour
{
    [Test]
    public void TestMassiveObjectCreation()
    {
        int objectCount = 10000;
        float startTime = Time.realtimeSinceStartup;
        
        // 创建大量对象
        for (int i = 0; i < objectCount; i++)
        {
            GameObject obj = new GameObject($"Test_{i}");
            obj.AddComponent<Rigidbody>();
            obj.AddComponent<BoxCollider>();
        }
        
        float endTime = Time.realtimeSinceStartup;
        float duration = endTime - startTime;
        
        // 验证创建时间
        Assert.Less(duration, 5.0f); // 5秒内完成
        
        // 验证帧率
        Assert.Greater(1.0f / Time.deltaTime, 30.0f); // 保持30FPS以上
    }
    
    [Test]
    public void TestNetworkStress()
    {
        NetworkStressConfig config = new NetworkStressConfig
        {
            clientCount = 100,
            messagePerSecond = 10,
            testDuration = 60
        };
        
        StartCoroutine(RunNetworkStressTest(config));
        
        yield return new WaitForSeconds(config.testDuration);
        
        // 验证网络性能
        Assert.Less(NetworkStats.PacketLoss, 0.01f); // 丢包率小于1%
        Assert.Less(NetworkStats.AverageLatency, 100); // 延迟小于100ms
    }
}
```

## 4. 自动化测试

### 4.1 测试自动化框架

```csharp
public class AutomationFramework
{
    private TestRunner testRunner;
    private TestReporter reporter;
    
    public async Task RunAutomatedTests()
    {
        // 初始化测试环境
        testRunner = new TestRunner();
        reporter = new TestReporter();
        
        // 收集测试用例
        var testCases = CollectTestCases();
        
        // 执行测试
        foreach (var testCase in testCases)
        {
            var result = await testRunner.RunTest(testCase);
            reporter.AddResult(result);
        }
        
        // 生成报告
        reporter.GenerateReport();
    }
    
    private List<TestCase> CollectTestCases()
    {
        return Assembly.GetExecutingAssembly()
            .GetTypes()
            .Where(t => t.GetCustomAttribute<TestClassAttribute>() != null)
            .SelectMany(t => t.GetMethods())
            .Where(m => m.GetCustomAttribute<TestMethodAttribute>() != null)
            .Select(m => new TestCase(m))
            .ToList();
    }
}
```
