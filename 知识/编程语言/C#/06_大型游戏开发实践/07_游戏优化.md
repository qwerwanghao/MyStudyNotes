## 2. CPU性能优化

### 2.1 代码优化

```csharp
public class CodeOptimization : MonoBehaviour
{
    // 缓存组件引用
    private Transform cachedTransform;
    private Rigidbody cachedRigidbody;
    
    void Awake()
    {
        // 在Awake中缓存组件引用
        cachedTransform = transform;
        cachedRigidbody = GetComponent<Rigidbody>();
    }
    
    // 使用对象池
    public class ObjectPool<T> where T : Component
    {
        private Queue<T> pool = new Queue<T>();
        private T prefab;
        
        public T Get()
        {
            if (pool.Count > 0)
            {
                return pool.Dequeue();
            }
            return Instantiate(prefab);
        }
        
        public void Return(T obj)
        {
            obj.gameObject.SetActive(false);
            pool.Enqueue(obj);
        }
    }
    
    // 优化Update方法
    private float updateInterval = 0.1f;
    private float nextUpdateTime = 0;
    
    void Update()
    {
        if (Time.time >= nextUpdateTime)
        {
            UpdateLogic();
            nextUpdateTime = Time.time + updateInterval;
        }
    }
}
```

```csharp
public class PhysicsOptimization : MonoBehaviour
{

    {
        // 设置物理更新频率
        Time.fixedDeltaTime = 0.02f;
        
        // 设置最大允许的物理模拟步数

    }

    {
        // 使用层级过滤碰撞

        {
            HandleEnemyCollision(collision);
        }
    }
    
    // 使用触发器代替连续碰撞检测
    void OnTriggerStay(Collider other)
    {
        // 在触发器内进行简单检测
        if (Vector3.Distance(transform.position, other.transform.position) < 1.0f)
        {
            HandleInteraction(other);
        }
    }
}
```

```csharp
public class RenderingOptimization : MonoBehaviour
{
    void Start()
    {
        // 设置LOD
        LODGroup lodGroup = GetComponent<LODGroup>();
        LOD[] lods = new LOD[3];
        
        Renderer[] renderers = GetComponents<Renderer>();
        lods[0] = new LOD(0.6f, renderers);
        lods[1] = new LOD(0.3f, renderers);
        lods[2] = new LOD(0.1f, renderers);
        
        lodGroup.SetLODs(lods);
        
        // 静态批处理
        StaticBatchingUtility.Combine(gameObject);
    }
    
    // 优化材质
    void OptimizeMaterials()
    {
        Renderer renderer = GetComponent<Renderer>();
        Material material = renderer.material;
        
        // 减少材质属性
        material.EnableKeyword("_SPECULARHIGHLIGHTS_OFF");
        material.EnableKeyword("_GLOSSYREFLECTIONS_OFF");
    }
}
```

### 3.2 着色器优化

```csharp
// 优化的简单着色器示例
Shader "Custom/OptimizedShader"
{
    Properties
    {
        _MainTex ("Texture", 2D) = "white" {}
    }
    SubShader
    {
        Tags { "RenderType"="Opaque" }
        LOD 100
        
        Pass
        {
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #pragma multi_compile_fog
            
            #include "UnityCG.cginc"
            
            struct appdata
            {
                float4 vertex : POSITION;
                float2 uv : TEXCOORD0;
            };
            
            struct v2f
            {
                float2 uv : TEXCOORD0;
                UNITY_FOG_COORDS(1)
                float4 vertex : SV_POSITION;
            };
            
            sampler2D _MainTex;
            float4 _MainTex_ST;
            
            v2f vert (appdata v)
            {
                v2f o;
                o.vertex = UnityObjectToClipPos(v.vertex);
                o.uv = TRANSFORM_TEX(v.uv, _MainTex);
                UNITY_TRANSFER_FOG(o,o.vertex);
                return o;
            }
            
            fixed4 frag (v2f i) : SV_Target
            {
                fixed4 col = tex2D(_MainTex, i.uv);
                UNITY_APPLY_FOG(i.fogCoord, col);
                return col;
            }
            ENDCG
        }
    }
}
```

## 4. 内存优化

### 4.1 资源管理

```csharp
public class ResourceManagement : MonoBehaviour
{
    // 资源引用计数
    private Dictionary<string, ResourceRef> resourceRefs = new Dictionary<string, ResourceRef>();
    
    private class ResourceRef
    {
        public UnityEngine.Object resource;
        public int refCount;
    }
    
    public T LoadResource<T>(string path) where T : UnityEngine.Object
    {
        if (resourceRefs.TryGetValue(path, out ResourceRef resourceRef))
        {
            resourceRef.refCount++;
            return resourceRef.resource as T;
        }
        
        T resource = Resources.Load<T>(path);
        resourceRefs[path] = new ResourceRef { resource = resource, refCount = 1 };
        return resource;
    }
    
    public void UnloadResource(string path)
    {
        if (resourceRefs.TryGetValue(path, out ResourceRef resourceRef))
        {
            resourceRef.refCount--;
            if (resourceRef.refCount <= 0)
            {
                Resources.UnloadAsset(resourceRef.resource);
                resourceRefs.Remove(path);
            }
        }
    }
}
```

### 4.2 内存泄漏检测

```csharp
public class MemoryLeakDetector : MonoBehaviour
{
    private Dictionary<string, WeakReference> objectTracker = new Dictionary<string, WeakReference>();
    
    public void TrackObject(UnityEngine.Object obj, string identifier)
    {
        objectTracker[identifier] = new WeakReference(obj);
    }
    
    public void CheckLeaks()
    {
        foreach (var kvp in objectTracker)
        {
            if (kvp.Value.IsAlive)
            {
                Debug.LogWarning($"可能的内存泄漏: {kvp.Key}");
            }
        }
    }
    
    // 使用Profiler API检测内存
    public void AnalyzeMemory()
    {
        UnityEngine.Profiling.Profiler.BeginSample("Memory Analysis");
        
        // 进行内存分析
        System.GC.Collect();
        System.GC.WaitForPendingFinalizers();
        
        UnityEngine.Profiling.Profiler.EndSample();
    }
}
```

## 5. 加载性能优化

### 5.1 异步加载

```csharp
public class AsyncLoadManager : MonoBehaviour
{
    private Queue<AsyncOperation> loadQueue = new Queue<AsyncOperation>();
    
    public async Task LoadSceneAsync(string sceneName)
    {
        AsyncOperation operation = SceneManager.LoadSceneAsync(sceneName);
        operation.allowSceneActivation = false;
        loadQueue.Enqueue(operation);
        
        while (operation.progress < 0.9f)
        {
            await Task.Yield();
            UpdateLoadingProgress(operation.progress);
        }
        
        // 完成加载
        operation.allowSceneActivation = true;
    }
    
    public async Task<T> LoadAssetAsync<T>(string path) where T : UnityEngine.Object
    {
        ResourceRequest request = Resources.LoadAsync<T>(path);
        while (!request.isDone)
        {
            await Task.Yield();
        }
        return request.asset as T;
    }
    
    private void UpdateLoadingProgress(float progress)
    {
        // 更新加载进度UI
        EventManager.Dispatch("LoadingProgress", progress);
    }
}
```

### 5.2 资源预加载

```csharp
public class PreloadManager : MonoBehaviour
{
    private Dictionary<string, UnityEngine.Object> preloadedAssets = 
        new Dictionary<string, UnityEngine.Object>();
    
    public async Task PreloadAssets(string[] assetPaths)
    {
        foreach (string path in assetPaths)
        {
            if (!preloadedAssets.ContainsKey(path))
            {
                var asset = await LoadAssetAsync<UnityEngine.Object>(path);
                preloadedAssets[path] = asset;
            }
        }
    }
    
    public UnityEngine.Object GetPreloadedAsset(string path)
    {
        if (preloadedAssets.TryGetValue(path, out UnityEngine.Object asset))
        {
            return asset;
        }
        return null;
    }
    
    public void UnloadUnusedAssets()
    {
        var unusedAssets = preloadedAssets.Keys
            .Where(k => !IsAssetInUse(k))
            .ToList();
            
        foreach (var key in unusedAssets)
        {
            Resources.UnloadAsset(preloadedAssets[key]);
            preloadedAssets.Remove(key);
        }
    }
}
```

## 6. 网络性能优化

### 6.1 网络请求优化

```csharp
public class NetworkOptimization : MonoBehaviour
{
    private Queue<NetworkRequest> requestQueue = new Queue<NetworkRequest>();
    private const int MaxConcurrentRequests = 3;
    private int currentRequests = 0;
    
    private class NetworkRequest
    {
        public string url;
        public TaskCompletionSource<string> tcs;
    }
    
    public async Task<string> SendRequest(string url)
    {
        var request = new NetworkRequest
        {
            url = url,
            tcs = new TaskCompletionSource<string>()
        };
        
        requestQueue.Enqueue(request);
        ProcessQueue();
        
        return await request.tcs.Task;
    }
    
    private async void ProcessQueue()
    {
        while (requestQueue.Count > 0 && currentRequests < MaxConcurrentRequests)
        {
            var request = requestQueue.Dequeue();
            currentRequests++;
            
            try
            {
                using (UnityWebRequest www = UnityWebRequest.Get(request.url))
                {
                    await www.SendWebRequest();
                    
                    if (www.result == UnityWebRequest.Result.Success)
                    {
                        request.tcs.SetResult(www.downloadHandler.text);
                    }
                    else
                    {
                        request.tcs.SetException(new Exception(www.error));
                    }
                }
            }
            finally
            {
                currentRequests--;
                ProcessQueue();
            }
        }
    }
}
```

### 6.2 数据压缩

```csharp
public class DataCompression
{
    public byte[] Compress(byte[] data)
    {
        using (MemoryStream output = new MemoryStream())
        {
            using (GZipStream gzip = new GZipStream(output, CompressionLevel.Optimal))
            {
                gzip.Write(data, 0, data.length);
            }
            return output.ToArray();
        }
    }
    
    public byte[] Decompress(byte[] compressedData)
    {
        using (MemoryStream input = new MemoryStream(compressedData))
        using (GZipStream gzip = new GZipStream(input, CompressionMode.Decompress))
        using (MemoryStream output = new MemoryStream())
        {
            gzip.CopyTo(output);
            return output.ToArray();
        }
    }
    
    // Protocol Buffers序列化
    public byte[] SerializeData<T>(T data) where T : IMessage
    {
        using (MemoryStream stream = new MemoryStream())
        {
            data.WriteTo(stream);
            return stream.ToArray();
        }
    }
}
```

## 7. 性能监控

### 7.1 性能分析器

```csharp
public class PerformanceAnalyzer : MonoBehaviour
{
    private float[] fpsBuffer = new float[60];
    private int fpsBufferIndex = 0;
    
    void Update()
    {
        // 记录FPS
        fpsBuffer[fpsBufferIndex] = 1.0f / Time.deltaTime;
        fpsBufferIndex = (fpsBufferIndex + 1) % fpsBuffer.Length;
        
        // 分析性能
        AnalyzePerformance();
    }
    
    private void AnalyzePerformance()
    {
        float averageFps = fpsBuffer.Average();
        float minFps = fpsBuffer.Min();
        
        if (minFps < 30)
        {
            Debug.LogWarning($"性能警告: FPS低于30 (平均: {averageFps:F1}, 最低: {minFps:F1})");
        }
        
        // 监控内存使用
        long totalMemory = UnityEngine.Profiling.Profiler.GetTotalAllocatedMemoryLong();
        if (totalMemory > 1024 * 1024 * 1024) // 1GB
        {
            Debug.LogWarning($"内存使用警告: {totalMemory / (1024 * 1024)}MB");
        }
    }
}
```

### 7.2 性能报告

```csharp
public class PerformanceReporter : MonoBehaviour
{
    private struct PerformanceMetrics
    {
        public float fps;
        public float memoryUsage;
        public float drawCalls;
        public float triangles;
    }
    
    private List<PerformanceMetrics> metricsHistory = new List<PerformanceMetrics>();
    
    public void RecordMetrics()
    {
        PerformanceMetrics metrics = new PerformanceMetrics
        {
            fps = 1.0f / Time.deltaTime,
            memoryUsage = UnityEngine.Profiling.Profiler.GetTotalAllocatedMemoryLong() / (1024f * 1024f),
            drawCalls = UnityEngine.Profiling.Profiler.GetRenderScope().drawCalls,
            triangles = UnityEngine.Profiling.Profiler.GetRenderScope().triangles
        };
        
        metricsHistory.Add(metrics);
        
        if (metricsHistory.Count > 1000)
        {
            GenerateReport();
        }
    }
    
    private void GenerateReport()
    {
        var report = new StringBuilder();
        report.AppendLine("性能报告");
        report.AppendLine($"平均FPS: {metricsHistory.Average(m => m.fps):F1}");
        report.AppendLine($"最低FPS: {metricsHistory.Min(m => m.fps):F1}");
        report.AppendLine($"平均内存使用: {metricsHistory.Average(m => m.memoryUsage):F1}MB");
        report.AppendLine($"平均DrawCalls: {metricsHistory.Average(m => m.drawCalls):F0}");
        
        Debug.Log(report.ToString());
        metricsHistory.Clear();
    }
}
```
