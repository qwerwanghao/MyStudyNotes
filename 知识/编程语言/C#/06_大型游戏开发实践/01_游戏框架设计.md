# Unity游戏框架设计

## 1. 框架设计原则

### 1.1 基本原则

- **单一职责原则（SRP）**：每个类只负责一个功能
- **开闭原则（OCP）**：对扩展开放，对修改关闭
- **里氏替换原则（LSP）**：子类必须能够替换其父类
- **接口隔离原则（ISP）**：使用多个专门的接口比使用单个总接口要好
- **依赖倒置原则（DIP）**：高层模块不应该依赖低层模块，都应该依赖抽象

### 1.2 架构模式

- **MVC模式**：Model-View-Controller
- **MVVM模式**：Model-View-ViewModel
- **ECS架构**：Entity-Component-System

### 2.2 资源管理器

```csharp
public class ResourceManager : MonoSingleton<ResourceManager>
{
    private Dictionary<string, UnityEngine.Object> resourceCache = new Dictionary<string, UnityEngine.Object>();
    
    // 异步加载资源
    public async Task<T> LoadAssetAsync<T>(string path) where T : UnityEngine.Object
    {
        if (resourceCache.TryGetValue(path, out UnityEngine.Object cachedResource))
        {
            return cachedResource as T;
        }
        
        var request = Resources.LoadAsync<T>(path);
        while (!request.isDone)
        {
            await Task.Yield();
        }
        
        var asset = request.asset as T;
        resourceCache[path] = asset;
        return asset;
    }
    
    // 卸载资源
    public void UnloadAsset(string path)
    {
        if (resourceCache.TryGetValue(path, out UnityEngine.Object resource))
        {
            Resources.UnloadAsset(resource);
            resourceCache.Remove(path);
        }
    }
    
    // 清理所有资源
    public void ClearAllResources()
    {
        foreach (var resource in resourceCache.Values)
        {
            Resources.UnloadAsset(resource);
        }
        resourceCache.Clear();
        Resources.UnloadUnusedAssets();
        System.GC.Collect();
    }
}
```

### 2.3 音频管理器

```csharp
public class AudioManager : MonoSingleton<AudioManager>
{
    private AudioSource musicSource;
    private Dictionary<string, AudioSource> soundEffects = new Dictionary<string, AudioSource>();
    
    private void Start()
    {
        musicSource = gameObject.AddComponent<AudioSource>();
        musicSource.loop = true;
    }
    
    // 播放背景音乐
    public async void PlayMusic(string musicPath, float volume = 1f)
    {
        var clip = await ResourceManager.Instance.LoadAssetAsync<AudioClip>(musicPath);
        musicSource.clip = clip;
        musicSource.volume = volume;
        musicSource.Play();
    }
    
    // 播放音效
    public async void PlaySound(string soundPath, float volume = 1f)
    {
        if (!soundEffects.TryGetValue(soundPath, out AudioSource source))
        {
            source = gameObject.AddComponent<AudioSource>();
            soundEffects[soundPath] = source;
        }
        
        var clip = await ResourceManager.Instance.LoadAssetAsync<AudioClip>(soundPath);
        source.clip = clip;
        source.volume = volume;
        source.Play();
    }
}
```

## 3. UI框架

### 3.1 UI管理器

```csharp
public class UIManager : MonoSingleton<UIManager>
{
    private Dictionary<string, UIPanel> panels = new Dictionary<string, UIPanel>();
    private Stack<UIPanel> panelStack = new Stack<UIPanel>();
    
    // 打开UI面板
    public async Task<UIPanel> OpenPanel(string panelPath)
    {
        if (panels.TryGetValue(panelPath, out UIPanel panel))
        {
            panel.gameObject.SetActive(true);
        }
        else
        {
            var prefab = await ResourceManager.Instance.LoadAssetAsync<GameObject>(panelPath);
            var instance = Instantiate(prefab, transform);
            panel = instance.GetComponent<UIPanel>();
            panels[panelPath] = panel;
        }
        
        panelStack.Push(panel);
        panel.OnOpen();
        return panel;
    }
    
    // 关闭当前面板
    public void CloseTopPanel()
    {
        if (panelStack.Count > 0)
        {
            var panel = panelStack.Pop();
            panel.OnClose();
            panel.gameObject.SetActive(false);
        }
    }
}

// UI面板基类
public abstract class UIPanel : MonoBehaviour
{
    public virtual void OnOpen() { }
    public virtual void OnClose() { }
}
```

### 3.2 UI事件系统

```csharp
public class UIEventSystem
{
    private static Dictionary<string, System.Action<object>> eventHandlers = 
        new Dictionary<string, System.Action<object>>();
    
    // 注册事件
    public static void Register(string eventName, System.Action<object> handler)
    {
        if (!eventHandlers.ContainsKey(eventName))
        {
            eventHandlers[eventName] = handler;
        }
        else
        {
            eventHandlers[eventName] += handler;
        }
    }
    
    // 取消注册
    public static void Unregister(string eventName, System.Action<object> handler)
    {
        if (eventHandlers.ContainsKey(eventName))
        {
            eventHandlers[eventName] -= handler;
        }
    }
    
    // 触发事件
    public static void Trigger(string eventName, object data = null)
    {
        if (eventHandlers.TryGetValue(eventName, out var handler))
        {
            handler?.Invoke(data);
        }
    }
}
```

## 4. 数据管理

### 4.1 数据存储

```csharp
public class DataManager : MonoSingleton<DataManager>
{
    private Dictionary<string, object> gameData = new Dictionary<string, object>();
    
    // 保存数据
    public void SaveData<T>(string key, T data)
    {
        string json = JsonUtility.ToJson(data);
        PlayerPrefs.SetString(key, json);
        gameData[key] = data;
    }
    
    // 加载数据
    public T LoadData<T>(string key) where T : new()
    {
        if (gameData.TryGetValue(key, out object cachedData))
        {
            return (T)cachedData;
        }
        
        if (PlayerPrefs.HasKey(key))
        {
            string json = PlayerPrefs.GetString(key);
            T data = JsonUtility.FromJson<T>(json);
            gameData[key] = data;
            return data;
        }
        
        return new T();
    }
    
    // 清除数据
    public void ClearData(string key)
    {
        PlayerPrefs.DeleteKey(key);
        gameData.Remove(key);
    }
}
```

### 4.2 配置管理

```csharp
public class ConfigManager : MonoSingleton<ConfigManager>
{
    private Dictionary<string, ScriptableObject> configs = 
        new Dictionary<string, ScriptableObject>();
    
    // 加载配置
    public async Task LoadConfigs()
    {
        var configFiles = await ResourceManager.Instance.LoadAssetAsync<TextAsset>("Configs/ConfigList");
        var configList = JsonUtility.FromJson<ConfigList>(configFiles.text);
        
        foreach (var configPath in configList.configs)
        {
            var config = await ResourceManager.Instance.LoadAssetAsync<ScriptableObject>(configPath);
            configs[configPath] = config;
        }
    }
    
    // 获取配置
    public T GetConfig<T>(string configPath) where T : ScriptableObject
    {
        if (configs.TryGetValue(configPath, out ScriptableObject config))
        {
            return config as T;
        }
        return null;
    }
}
```

## 5. 对象池系统

```csharp
public class ObjectPool : MonoSingleton<ObjectPool>
{
    private Dictionary<string, Queue<GameObject>> pools = 
        new Dictionary<string, Queue<GameObject>>();
    
    // 获取对象
    public async Task<GameObject> GetObject(string prefabPath)
    {
        if (!pools.TryGetValue(prefabPath, out Queue<GameObject> pool))
        {
            pool = new Queue<GameObject>();
            pools[prefabPath] = pool;
        }
        
        GameObject obj;
        if (pool.Count > 0)
        {
            obj = pool.Dequeue();
            obj.SetActive(true);
        }
        else
        {
            var prefab = await ResourceManager.Instance.LoadAssetAsync<GameObject>(prefabPath);
            obj = Instantiate(prefab);
        }
        
        return obj;
    }
    
    // 回收对象
    public void ReturnObject(string prefabPath, GameObject obj)
    {
        if (!pools.TryGetValue(prefabPath, out Queue<GameObject> pool))
        {
            pool = new Queue<GameObject>();
            pools[prefabPath] = pool;
        }
        
        obj.SetActive(false);
        pool.Enqueue(obj);
    }
}
```

## 7. 工具类

### 7.1 单例基类

```csharp
public abstract class MonoSingleton<T> : MonoBehaviour where T : MonoSingleton<T>
{
    private static T instance;
    public static T Instance
    {
        get
        {
            if (instance == null)
            {
                instance = FindObjectOfType<T>();
                if (instance == null)
                {
                    GameObject obj = new GameObject(typeof(T).Name);
                    instance = obj.AddComponent<T>();
                }
            }
            return instance;
        }
    }
    
    protected virtual void Awake()
    {
        if (instance == null)
        {
            instance = this as T;
            DontDestroyOnLoad(gameObject);
        }
        else
        {
            Destroy(gameObject);
        }
    }
}
```

### 7.2 扩展方法

```csharp
public static class Extensions
{
    // Transform扩展
    public static void Reset(this Transform transform)
    {
        transform.localPosition = Vector3.zero;
        transform.localRotation = Quaternion.identity;
        transform.localScale = Vector3.one;
    }
    
    // GameObject扩展
    public static T GetOrAddComponent<T>(this GameObject obj) where T : Component
    {
        T component = obj.GetComponent<T>();
        if (component == null)
        {
            component = obj.AddComponent<T>();
        }
        return component;
    }
    
    // 集合扩展
    public static T RandomElement<T>(this IList<T> list)
    {
        if (list.Count == 0) return default(T);
        return list[Random.Range(0, list.Count)];
    }
}
```

## 8. 最佳实践

### 8.1 框架使用规范

1. 遵循命名规范
   - 类名使用PascalCase
   - 方法名使用PascalCase
   - 私有变量使用camelCase
   - 常量使用UPPER_CASE

2. 代码组织
   - 按功能模块分文件夹
   - 相关的类放在同一命名空间
   - 使用Region组织代码块

3. 性能优化
   - 合理使用对象池
   - 避免频繁的GC
   - 缓存组件引用

### 8.2 调试与测试

1. 日志系统
   - 不同级别的日志
   - 条件编译控制日志输出
