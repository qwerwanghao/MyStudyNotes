# Unity游戏安全

## 1. 数据安全

### 1.1 数据加密

```csharp
public class SecurityUtils
{
    // AES加密
    public static string AESEncrypt(string content, string key)
    {
        try
        {
            byte[] keyBytes = Encoding.UTF8.GetBytes(key);
            byte[] contentBytes = Encoding.UTF8.GetBytes(content);
            
            using (Aes aes = Aes.Create())
            {
                aes.Key = keyBytes;
                aes.GenerateIV();
                
                using (MemoryStream ms = new MemoryStream())
                {
                    ms.Write(aes.IV, 0, aes.IV.Length);
                    
                    using (CryptoStream cs = new CryptoStream(
                        ms, aes.CreateEncryptor(), CryptoStreamMode.Write))
                    {
                        cs.Write(contentBytes, 0, contentBytes.Length);
                        cs.FlushFinalBlock();
                    }
                    
                    return Convert.ToBase64String(ms.ToArray());
                }
            }
        }
        catch (Exception e)
        {
            Debug.LogError($"加密失败: {e.Message}");
            return null;
        }
    }
    
    // AES解密
    public static string AESDecrypt(string encryptedContent, string key)
    {
        try
        {
            byte[] keyBytes = Encoding.UTF8.GetBytes(key);
            byte[] encryptedBytes = Convert.FromBase64String(encryptedContent);
            
            using (Aes aes = Aes.Create())
            {
                byte[] iv = new byte[aes.IV.Length];
                Array.Copy(encryptedBytes, 0, iv, 0, iv.Length);
                
                using (MemoryStream ms = new MemoryStream())
                {
                    using (CryptoStream cs = new CryptoStream(
                        ms, aes.CreateDecryptor(keyBytes, iv), CryptoStreamMode.Write))
                    {
                        cs.Write(encryptedBytes, iv.Length, 
                            encryptedBytes.Length - iv.Length);
                        cs.FlushFinalBlock();
                    }
                    
                    return Encoding.UTF8.GetString(ms.ToArray());
                }
            }
        }
        catch (Exception e)
        {
            Debug.LogError($"解密失败: {e.Message}");
            return null;
        }
    }
}
```

### 1.2 存储安全

```csharp
public class SecureStorage
{
    private static readonly string KEY = "your_secure_key";
    
    public static void SaveSecureData<T>(string key, T data)
    {
        try
        {
            string json = JsonUtility.ToJson(data);
            string encrypted = SecurityUtils.AESEncrypt(json, KEY);
            PlayerPrefs.SetString(key, encrypted);
            PlayerPrefs.Save();
        }
        catch (Exception e)
        {
            Debug.LogError($"安全存储失败: {e.Message}");
        }
    }
    
    public static T LoadSecureData<T>(string key)
    {
        try
        {
            string encrypted = PlayerPrefs.GetString(key);
            if (string.IsNullOrEmpty(encrypted))
                return default(T);
                
            string json = SecurityUtils.AESDecrypt(encrypted, KEY);
            return JsonUtility.FromJson<T>(json);
        }
        catch (Exception e)
        {
            Debug.LogError($"安全读取失败: {e.Message}");
            return default(T);
        }
    }
}
```

## 2. 网络安全

### 2.1 通信加密

```csharp
public class NetworkSecurity
{
    private static RSACryptoServiceProvider rsa;
    
    // 初始化RSA
    public static void InitializeRSA()
    {
        rsa = new RSACryptoServiceProvider(2048);
    }
    
    // 获取公钥
    public static string GetPublicKey()
    {
        return Convert.ToBase64String(rsa.ExportRSAPublicKey());
    }
    
    // RSA加密
    public static byte[] RSAEncrypt(byte[] data)
    {
        try
        {
            return rsa.Encrypt(data, RSAEncryptionPadding.OaepSHA256);
        }
        catch (Exception e)
        {
            Debug.LogError($"RSA加密失败: {e.Message}");
```

### 2.2 防护措施

```csharp
public class SecurityProtection : MonoBehaviour
{
    // 防止重放攻击
    private class RequestInfo
    {
        public string timestamp;
        public string nonce;
        public string signature;
    }
    
    private HashSet<string> usedNonces = new HashSet<string>();
    
    public bool ValidateRequest(RequestInfo request)
    {
        // 检查时间戳
        if (!IsTimestampValid(request.timestamp))
            return false;
            
        // 检查nonce是否重复
        if (usedNonces.Contains(request.nonce))
            return false;
            
        // 验证签名
        if (!VerifySignature(request))
            return false;
            
        usedNonces.Add(request.nonce);
        return true;
    }
    
    private bool IsTimestampValid(string timestamp)
    {
        long ts = long.Parse(timestamp);
        long now = DateTimeOffset.UtcNow.ToUnixTimeSeconds();
        return Math.Abs(now - ts) <= 300; // 5分钟内有效
    }
    
    private bool VerifySignature(RequestInfo request)
    {
        string rawData = $"{request.timestamp}{request.nonce}";
        string expectedSignature = ComputeHMAC(rawData, "your_secret_key");
        return request.signature == expectedSignature;
    }
    
    private string ComputeHMAC(string message, string secret)
    {
        byte[] keyBytes = Encoding.UTF8.GetBytes(secret);
        byte[] messageBytes = Encoding.UTF8.GetBytes(message);
        
        using (var hmac = new HMACSHA256(keyBytes))
        {
            byte[] hashBytes = hmac.ComputeHash(messageBytes);
            return Convert.ToBase64String(hashBytes);
        }
    }
}
```

## 3. 反作弊系统

### 3.1 内存检测

```csharp
public class MemoryCheatDetection : MonoBehaviour
{
    private Dictionary<string, byte[]> memorySnapshots = 
        new Dictionary<string, byte[]>();
    
    // 记录关键数据的内存快照
    public void TakeMemorySnapshot(string key, object data)
    {
        byte[] snapshot = SerializeObject(data);
        memorySnapshots[key] = snapshot;
    }
    
    // 验证内存数据是否被修改
    public bool ValidateMemorySnapshot(string key, object currentData)
    {
        if (!memorySnapshots.ContainsKey(key))
            return false;
            
        byte[] currentSnapshot = SerializeObject(currentData);
        byte[] originalSnapshot = memorySnapshots[key];
        
        return CompareByteArrays(currentSnapshot, originalSnapshot);
    }
    
    private byte[] SerializeObject(object obj)
    {
        BinaryFormatter formatter = new BinaryFormatter();
        using (MemoryStream ms = new MemoryStream())
        {
            formatter.Serialize(ms, obj);
            return ms.ToArray();
        }
    }
    
    private bool CompareByteArrays(byte[] array1, byte[] array2)
    {
        if (array1.Length != array2.Length)
            return false;
            
        for (int i = 0; i < array1.Length; i++)
        {
            if (array1[i] != array2[i])
                return false;
        }
        
        return true;
    }
}
```

### 3.2 行为检测

```csharp
public class BehaviorDetection : MonoBehaviour
{
    private struct PlayerAction
    {
        public Vector3 position;
        public float timestamp;
        public string actionType;
    }
    
    private Queue<PlayerAction> actionHistory = new Queue<PlayerAction>();
    private const int MAX_HISTORY_SIZE = 100;
    
    // 记录玩家行为
    public void RecordAction(Vector3 position, string actionType)
    {
        PlayerAction action = new PlayerAction
        {
            position = position,
            timestamp = Time.time,
            actionType = actionType
        };
        
        actionHistory.Enqueue(action);
        
        if (actionHistory.Count > MAX_HISTORY_SIZE)
            actionHistory.Dequeue();
            
        // 检测异常行为
        DetectAbnormalBehavior();
    }
    
    private void DetectAbnormalBehavior()
    {
        if (actionHistory.Count < 2)
            return;
            
        PlayerAction[] actions = actionHistory.ToArray();
        
        // 检测速度异常
        for (int i = 1; i < actions.Length; i++)
        {
            float distance = Vector3.Distance(
                actions[i].position, 
                actions[i-1].position);
            float timeSpan = actions[i].timestamp - actions[i-1].timestamp;
            float speed = distance / timeSpan;
            
            if (speed > 100) // 速度阈值
            {
                ReportCheating("Speed hack detected");
            }
        }
        
        // 检测操作频率异常
        int actionCount = actions.Count(a => 
            a.timestamp > Time.time - 1.0f);
        if (actionCount > 50) // 频率阈值
        {
            ReportCheating("Action frequency abnormal");
        }
    }
    
    private void ReportCheating(string reason)
    {
        Debug.LogWarning($"检测到作弊行为: {reason}");
```

```csharp
public class DebugProtection : MonoBehaviour
{
    void Start()
    {
        StartCoroutine(CheckDebugger());
    }
    
    IEnumerator CheckDebugger()
    {
        while (true)
        {
            if (Debug.isDebugBuild)
            {
                HandleDebuggerDetected();
            }
            
            if (IsBeingDebugged())
            {
                HandleDebuggerDetected();
            }
            
            yield return new WaitForSeconds(1.0f);
        }
    }
    
    private bool IsBeingDebugged()
    {
        #if !UNITY_EDITOR

            if (System.Diagnostics.Debugger.IsAttached)
            {
                return true;
            }
        }
        catch
        {
            // 忽略异常
        }
        #endif
        
        return false;
    }
```
