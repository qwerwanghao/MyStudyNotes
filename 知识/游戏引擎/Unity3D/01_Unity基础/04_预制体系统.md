# 预制体系统

<!-- DEEP-TOPIC-GUIDE-START -->
> 预制体系统解决的是“如何把对象做成可复用的模板”。它是 Unity 从随手搭场景走向可维护项目结构的重要一步。

## 先读这篇时，建议带着这些问题

- 为什么预制体不是简单复制对象，而是结构复用机制
- 预制体、实例、变体之间的关系是什么
- 什么时候适合直接 Instantiate，什么时候要引入对象池
- 预制体为什么会影响场景组织、资源管理和性能优化

## 推荐阅读顺序

1. 预制体基础
2. 创建与管理
3. 预制体变体

5. 对象池

## 这一页最值得重点抓住的概念

- **预制体本质上是可复用结构，而不只是复制方便**
- **变体可以降低重复维护成本**
- **对象池是高频实例化场景里的关键优化手段**
- **预制体设计混乱，后面资源和性能都会被拖累**

public class PrefabManager : MonoBehaviour
{
    public GameObject prefab;

    {
        // 实例化预制体
        GameObject instance = Instantiate(prefab);

        GameObject instance2 = Instantiate(

## 预制体加载

### 资源加载
```csharp
public class PrefabLoader : MonoBehaviour
{
    // 从Resources加载
    void LoadFromResources()
    {
        GameObject prefab = Resources.Load<GameObject>("Prefabs/Enemy");
        if (prefab != null)
        {
            Instantiate(prefab);
        }
    }
    
    // 异步加载
    async Task<GameObject> LoadPrefabAsync(string path)
    {
        var request = Resources.LoadAsync<GameObject>(path);
        while (!request.isDone)
        {
            await Task.Yield();
        }
        return request.asset as GameObject;
    }
    
    // 批量加载
    IEnumerator LoadMultiplePrefabs(string[] paths)
    {
        foreach (string path in paths)
        {
            var request = Resources.LoadAsync<GameObject>(path);
            yield return request;
            
            if (request.asset != null)
            {
                Instantiate(request.asset as GameObject);
            }
        }
    }
}
```

### 对象池
```csharp
public class PrefabPool : MonoBehaviour
{
    [System.Serializable]
    public class Pool
    {
        public string tag;
        public GameObject prefab;
        public int size;
    }
    
    public List<Pool> pools;
    private Dictionary<string, Queue<GameObject>> poolDictionary;
    
    void Start()
    {
        poolDictionary = new Dictionary<string, Queue<GameObject>>();
        
        // 初始化对象池
        foreach (Pool pool in pools)
        {
            Queue<GameObject> objectPool = new Queue<GameObject>();
            
            for (int i = 0; i < pool.size; i++)
            {
                GameObject obj = Instantiate(pool.prefab);
                obj.SetActive(false);
                objectPool.Enqueue(obj);
            }
            
            poolDictionary.Add(pool.tag, objectPool);
        }
    }
    
    public GameObject SpawnFromPool(string tag, Vector3 position, 
        Quaternion rotation)
    {
        if (!poolDictionary.ContainsKey(tag))
        {
            Debug.LogWarning($"Pool with tag {tag} doesn't exist.");
            return null;
        }
        
        Queue<GameObject> pool = poolDictionary[tag];
        
        // 如果池为空，创建新对象
        if (pool.Count == 0)
        {
            Pool poolInfo = pools.Find(x => x.tag == tag);
            GameObject obj = Instantiate(poolInfo.prefab);
            pool.Enqueue(obj);
        }
        
        GameObject objectToSpawn = pool.Dequeue();
        objectToSpawn.SetActive(true);
        objectToSpawn.transform.position = position;
        objectToSpawn.transform.rotation = rotation;
        
        // 返回对象到池
        IPooledObject pooledObj = objectToSpawn.GetComponent<IPooledObject>();
        if (pooledObj != null)
        {
            pooledObj.OnObjectSpawn();
        }
        
        return objectToSpawn;
    }
    
    public void ReturnToPool(string tag, GameObject objectToReturn)
    {
        if (!poolDictionary.ContainsKey(tag))
        {
            Debug.LogWarning($"Pool with tag {tag} doesn't exist.");
            return;
        }
        
        objectToReturn.SetActive(false);
        poolDictionary[tag].Enqueue(objectToReturn);
    }
}

// 池化对象接口
public interface IPooledObject
{
    void OnObjectSpawn();
}
```

### 预制体链接
```csharp
public class PrefabLink : MonoBehaviour
{
    // 保持与预制体的链接
    public void UpdatePrefabLink(GameObject instance)
    {
        #if UNITY_EDITOR
        UnityEditor.PrefabUtility.RecordPrefabInstancePropertyModifications(
            instance);
        #endif
    }
    
    // 断开预制体链接
    public void BreakPrefabLink(GameObject instance)
    {
        #if UNITY_EDITOR
        UnityEditor.PrefabUtility.UnpackPrefabInstance(
            instance,
            UnityEditor.PrefabUnpackMode.Completely,
            UnityEditor.InteractionMode.AutomatedAction);
        #endif
    }
}

## 预制体优化

### 性能优化
```csharp
public class PrefabOptimization : MonoBehaviour
{
    // 预加载预制体
    void PreloadPrefabs(GameObject[] prefabs)
    {
        foreach (var prefab in prefabs)
        {
            // 预加载到内存
            var instance = Instantiate(prefab);
            instance.SetActive(false);
        }
    }
    
    // 延迟加载
    IEnumerator LoadPrefabsProgressively(GameObject[] prefabs)
    {
        foreach (var prefab in prefabs)
        {
            Instantiate(prefab);
            yield return new WaitForSeconds(0.1f);
        }
    }
    
    // 资源释放
    void CleanupPrefabs()
    {
        // 卸载未使用的资源
        Resources.UnloadUnusedAssets();
    }
}
```

### 预制体管理器
```csharp
public class PrefabRegistry : MonoBehaviour
{
    private static PrefabRegistry instance;
    public static PrefabRegistry Instance => instance;
    
    private Dictionary<string, GameObject> prefabCache = 
        new Dictionary<string, GameObject>();
    
    void Awake()
    {
        if (instance == null)
        {
            instance = this;
            DontDestroyOnLoad(gameObject);
        }
        else
        {
            Destroy(gameObject);
        }
    }
    
    // 注册预制体
    public void RegisterPrefab(string key, GameObject prefab)
    {
        if (!prefabCache.ContainsKey(key))
        {
            prefabCache.Add(key, prefab);
        }
    }
    
    // 获取预制体
    public GameObject GetPrefab(string key)
    {
        if (prefabCache.TryGetValue(key, out GameObject prefab))
        {
            return prefab;
        }
        return null;
    }
    
    // 清理缓存
    public void ClearCache()
    {
        prefabCache.Clear();
        Resources.UnloadUnusedAssets();
    }
}
