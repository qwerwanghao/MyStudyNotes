# 摄像机系统

<!-- DEEP-TOPIC-GUIDE-START -->
> 摄像机系统直接决定玩家“看见什么、怎么感知世界、手感是否舒服”。很多体验问题表面上像控制问题，实际根源常常在摄像机。

## 先读这篇时，建议带着这些问题

- 摄像机的视场角、裁剪面、投影方式会怎样影响体验
- 跟随、观察、震动这些常见能力应该怎么组织
- 摄像机为什么会和输入系统、角色移动、场景组织强耦合
- 为什么摄像机系统往往是“玩法感觉”的放大器

## 推荐阅读顺序

1. 基础摄像机设置
2. 摄像机属性
3. 摄像机控制
4. 摄像机跟随
5. 相机震动
6. 多摄像机与特殊效果

## 这一页最值得重点抓住的概念

- **摄像机是玩家观察世界的接口，不只是一个组件**
- **不同投影和参数设置会直接改变空间感受**
- **跟随逻辑是否稳定，会决定操作手感**
- **摄像机通常要和输入、目标对象、场景结构一起设计**

## 基础摄像机设置

### 摄像机属性
```csharp
public class CameraSetup : MonoBehaviour
{
    private Camera mainCamera;
    
    void Start()
    {
        mainCamera = GetComponent<Camera>();
        
        // 基本属性设置
        mainCamera.fieldOfView = 60f;        // 视场角
        mainCamera.nearClipPlane = 0.3f;     // 近裁剪面
        mainCamera.farClipPlane = 1000f;     // 远裁剪面
        mainCamera.depth = 0;                 // 渲染顺序
        mainCamera.clearFlags = CameraClearFlags.Skybox; // 清除标记
        
        // 投影方式
        mainCamera.orthographic = false;      // 透视投影
        mainCamera.orthographicSize = 5f;     // 正交大小
        
        // 渲染设置
        mainCamera.backgroundColor = Color.black;
        mainCamera.cullingMask = LayerMask.GetMask("Default");
    }
    
    // 视口转换
    void ViewportHandling()
    {
        // 屏幕坐标转世界坐标
        Vector3 screenPoint = Input.mousePosition;
        Ray ray = mainCamera.ScreenPointToRay(screenPoint);
        
        // 视口坐标转世界坐标
        Vector3 viewportPoint = new Vector3(0.5f, 0.5f, 0f);
        Vector3 worldPoint = mainCamera.ViewportToWorldPoint(viewportPoint);
    }
}
```

### 摄像机控制
```csharp
public class CameraController : MonoBehaviour
{
    public float moveSpeed = 5f;
    public float rotationSpeed = 2f;
    public float zoomSpeed = 5f;
    
    void Update()
    {
        // 平移控制
        float horizontal = Input.GetAxis("Horizontal");
        float vertical = Input.GetAxis("Vertical");
        Vector3 moveDirection = new Vector3(horizontal, 0, vertical);
        transform.Translate(moveDirection * moveSpeed * Time.deltaTime);
        
        // 旋转控制
        if (Input.GetMouseButton(1))
        {
            float mouseX = Input.GetAxis("Mouse X");
            float mouseY = Input.GetAxis("Mouse Y");
            
            transform.RotateAround(transform.position, Vector3.up, 
                mouseX * rotationSpeed);
            transform.RotateAround(transform.position, transform.right, 
                -mouseY * rotationSpeed);
        }
        
        // 缩放控制
        float scroll = Input.GetAxis("Mouse ScrollWheel");
        Camera.main.fieldOfView = Mathf.Clamp(
            Camera.main.fieldOfView - scroll * zoomSpeed,
            20f,
            90f
        );
    }
}
```

## 高级摄像机功能

### 摄像机跟随
```csharp
public class CameraFollow : MonoBehaviour
{
    public Transform target;
    public float smoothSpeed = 0.125f;
    public Vector3 offset;
    
    // 平滑跟随
    void LateUpdate()
    {
        if (target == null) return;
        
        Vector3 desiredPosition = target.position + offset;
        Vector3 smoothedPosition = Vector3.Lerp(
            transform.position,
            desiredPosition,
            smoothSpeed
        );
        transform.position = smoothedPosition;
        
        transform.LookAt(target);
    }
    
    // 带预测的跟随
    public void FollowWithPrediction(Rigidbody targetRb)
    {
        if (targetRb == null) return;
        
        Vector3 predictedPosition = targetRb.position + 
            targetRb.velocity * Time.deltaTime;
        Vector3 desiredPosition = predictedPosition + offset;
        
        transform.position = Vector3.Lerp(
            transform.position,
            desiredPosition,
            smoothSpeed
        );
    }
}
```

### 相机震动
```csharp
public class CameraShake : MonoBehaviour
{
    private Vector3 originalPosition;
    
    // 普通震动
    public IEnumerator Shake(float duration, float magnitude)
    {
        originalPosition = transform.localPosition;
        float elapsed = 0f;
        
        while (elapsed < duration)
        {
            float x = Random.Range(-1f, 1f) * magnitude;
            float y = Random.Range(-1f, 1f) * magnitude;
            
            transform.localPosition = originalPosition + new Vector3(x, y, 0);
            elapsed += Time.deltaTime;
            yield return null;
        }
        
        transform.localPosition = originalPosition;
    }
    
    // 衰减震动
    public IEnumerator ShakeWithDamping(float duration, float magnitude)
    {
        originalPosition = transform.localPosition;
        float elapsed = 0f;
        
        while (elapsed < duration)
        {
            float damping = 1f - (elapsed / duration);
            float x = Random.Range(-1f, 1f) * magnitude * damping;
            float y = Random.Range(-1f, 1f) * magnitude * damping;
            
            transform.localPosition = originalPosition + new Vector3(x, y, 0);
            elapsed += Time.deltaTime;
            yield return null;
        }
        
        transform.localPosition = originalPosition;
    }
}
```

public class CinemachineManager : MonoBehaviour
{
    public CinemachineVirtualCamera virtualCamera;
    public float defaultFOV = 60f;
    
    void SetupVirtualCamera()
    {
        // 基本设置
        virtualCamera.m_Lens.FieldOfView = defaultFOV;
        
        // 跟随设置
        virtualCamera.Follow = target;
        virtualCamera.LookAt = target;
        
        // 设置相机阻尼
        var transposer = virtualCamera.GetCinemachineComponent<
            CinemachineTransposer>();
        if (transposer != null)
        {
            transposer.m_XDamping = 1f;
            transposer.m_YDamping = 1f;
            transposer.m_ZDamping = 1f;
        }
    }
    
    // 相机切换
    public void SwitchCamera(CinemachineVirtualCamera newCamera)
    {
        virtualCamera.Priority = 0;
        newCamera.Priority = 10;
    }
}

## 性能优化

### 视锥体剔除
```csharp
public class CameraOptimization : MonoBehaviour
{
    private Camera cam;
    private Plane[] planes;
    
    void Update()
    {
        // 获取视锥体平面
        planes = GeometryUtility.CalculateFrustumPlanes(cam);
        
        // 检查物体是否在视野内
        Renderer[] renderers = FindObjectsOfType<Renderer>();
        foreach (Renderer renderer in renderers)
        {
            if (GeometryUtility.TestPlanesAABB(planes, renderer.bounds))
            {
                // 物体在视野内
                renderer.enabled = true;
            }
            else
            {
                // 物体在视野外
                renderer.enabled = false;
            }
        }
    }
}
```

### 摄像机分层渲染
```csharp
public class CameraLayerManager : MonoBehaviour
{
    public Camera mainCamera;
    public Camera uiCamera;
    
    void SetupCameras()
    {
        // 主相机只渲染游戏世界
        mainCamera.cullingMask = ~(1 << LayerMask.NameToLayer("UI"));
        
        // UI相机只渲染UI层
        uiCamera.cullingMask = 1 << LayerMask.NameToLayer("UI");
        uiCamera.clearFlags = CameraClearFlags.Depth;
        uiCamera.depth = 1;
    }
    
    // 动态调整渲染层
    public void UpdateRenderingLayers(string[] layerNames)
    {
        int mask = 0;
        foreach (string layerName in layerNames)
        {
            mask |= 1 << LayerMask.NameToLayer(layerName);
        }
        mainCamera.cullingMask = mask;
    }
}
