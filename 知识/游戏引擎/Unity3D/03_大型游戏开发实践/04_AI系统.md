## 2. 决策系统

### 2.1 基础决策系统

```csharp
public class AIDecisionMaker : MonoBehaviour
{
    private List<AIAction> availableActions;
    private float decisionInterval = 0.5f;
    
    void Start()
    {
        availableActions = new List<AIAction>();
        InitializeActions();
        StartCoroutine(DecisionRoutine());
    }
    
    private void InitializeActions()
    {
        availableActions.Add(new AttackAction());
        availableActions.Add(new FleeAction());
        availableActions.Add(new PatrolAction());
    }
    
    private IEnumerator DecisionRoutine()
    {
        while (true)
        {
            AIAction bestAction = EvaluateActions();
            if (bestAction != null)
            {
                bestAction.Execute();
            }
            yield return new WaitForSeconds(decisionInterval);
        }
    }
    
    private AIAction EvaluateActions()
    {
        float highestScore = float.MinValue;
        AIAction bestAction = null;
        
        foreach (var action in availableActions)
        {
            float score = action.EvaluateScore();
            if (score > highestScore)
            {
                highestScore = score;
                bestAction = action;
            }
        }
        
        return bestAction;
    }
}

public abstract class AIAction
{
    public abstract float EvaluateScore();
    public abstract void Execute();
}
```

```csharp
public class UtilityAI : MonoBehaviour
{
    [System.Serializable]
    public class UtilityAction
    {
        public string name;
        public List<UtilityConsideration> considerations;
        public UnityEvent onExecute;
        
        public float EvaluateUtility()
        {
            float utility = 1f;
            foreach (var consideration in considerations)
            {
                utility *= consideration.EvaluateUtility();
            }
            return utility;
        }
    }
    
    [System.Serializable]
    public class UtilityConsideration
    {
        public string name;
        public AnimationCurve responseCurve;
        public float minValue;
        public float maxValue;
        
        public float EvaluateUtility()
        {
            float normalizedValue = GetNormalizedValue();
            return responseCurve.Evaluate(normalizedValue);
        }
        
        private float GetNormalizedValue()
        {
            // 获取当前值并归一化
            float currentValue = GetCurrentValue();
            return Mathf.InverseLerp(minValue, maxValue, currentValue);
        }
        
        private float GetCurrentValue()
        {
            // 实现具体的值获取逻辑
            return 0f;
        }
    }
}
```

### 3.2 动态避障

```csharp
public class DynamicAvoidance : MonoBehaviour
{
    private NavMeshAgent agent;
    private float avoidanceRadius = 5f;
    private LayerMask obstacleLayer;
    
    void Start()
    {
        agent = GetComponent<NavMeshAgent>();
        obstacleLayer = LayerMask.GetMask("Obstacle");
    }
    
    void Update()
    {
        AvoidObstacles();
    }
    
    private void AvoidObstacles()
    {
        Collider[] obstacles = Physics.OverlapSphere(transform.position, avoidanceRadius, obstacleLayer);
        
        Vector3 avoidanceDirection = Vector3.zero;
        foreach (var obstacle in obstacles)
        {
            Vector3 directionToObstacle = obstacle.transform.position - transform.position;
            float distance = directionToObstacle.magnitude;
            
            avoidanceDirection += -directionToObstacle.normalized * (avoidanceRadius - distance);
        }
        
        if (avoidanceDirection != Vector3.zero)
        {
            Vector3 newDirection = agent.velocity.normalized + avoidanceDirection;
            agent.velocity = newDirection.normalized * agent.speed;
        }
    }
}
```

## 4. 行为树

### 4.1 行为树节点

```csharp
public abstract class BehaviorNode
{
    public enum Status
    {
        Success,
        Failure,
        Running
    }
    
    public abstract Status Execute();
}

public class Sequence : BehaviorNode
{
    private List<BehaviorNode> children = new List<BehaviorNode>();
    private int currentChild = 0;
    
    public override Status Execute()
    {
        if (currentChild >= children.Count)
        {
            currentChild = 0;
            return Status.Success;
        }
        
        Status childStatus = children[currentChild].Execute();
        
        switch (childStatus)
        {
            case Status.Success:
                currentChild++;
                return Status.Running;
            case Status.Failure:
                currentChild = 0;
                return Status.Failure;
            case Status.Running:
                return Status.Running;
            default:
                return Status.Success;
        }
    }
}

public class Selector : BehaviorNode
{
    private List<BehaviorNode> children = new List<BehaviorNode>();
    private int currentChild = 0;
    
    public override Status Execute()
    {
        if (currentChild >= children.Count)
        {
            currentChild = 0;
            return Status.Failure;
        }
        
        Status childStatus = children[currentChild].Execute();
        
        switch (childStatus)
        {
            case Status.Success:
                currentChild = 0;
                return Status.Success;
            case Status.Failure:
                currentChild++;
                return Status.Running;
            case Status.Running:
                return Status.Running;
            default:
                return Status.Failure;
        }
    }
}
```

```csharp
public class BehaviorTree : MonoBehaviour
{
    private BehaviorNode rootNode;
    
    void Start()
    {
        BuildTree();
        StartCoroutine(ExecuteTree());
    }
    
    private void BuildTree()
    {
        // 构建行为树
        Sequence mainSequence = new Sequence();
        
        // 添加检查节点
        mainSequence.AddChild(new CheckHealthNode());
        
        // 添加选择节点
        Selector combatSelector = new Selector();
        combatSelector.AddChild(new FleeNode());
        combatSelector.AddChild(new AttackNode());
        
        mainSequence.AddChild(combatSelector);
        
        rootNode = mainSequence;
    }
    
    private IEnumerator ExecuteTree()
    {
        while (true)
        {
            rootNode.Execute();
            yield return new WaitForSeconds(0.1f);
        }
    }
}
```

## 5. 感知系统

### 5.1 视觉感知

```csharp
public class VisionSensor : MonoBehaviour
{
    public float viewRadius = 10f;
    public float viewAngle = 90f;
    public LayerMask targetMask;
    public LayerMask obstacleMask;
    
    private List<Transform> visibleTargets = new List<Transform>();
    
    void Update()
    {
        FindVisibleTargets();
    }
    
    void FindVisibleTargets()
    {
        visibleTargets.Clear();
        
        Collider[] targetsInViewRadius = Physics.OverlapSphere(transform.position, viewRadius, targetMask);
        
        foreach (Collider target in targetsInViewRadius)
        {
            Transform targetTransform = target.transform;
            Vector3 directionToTarget = (targetTransform.position - transform.position).normalized;
            
            if (Vector3.Angle(transform.forward, directionToTarget) < viewAngle / 2)
            {
                float distanceToTarget = Vector3.Distance(transform.position, targetTransform.position);
                
                if (!Physics.Raycast(transform.position, directionToTarget, distanceToTarget, obstacleMask))
                {
                    visibleTargets.Add(targetTransform);
                }
            }
        }
    }
}
```

## 6. 群体行为

### 6.1 群组移动

```csharp
public class FlockingBehavior : MonoBehaviour
{
    public float cohesionWeight = 1f;
    public float separationWeight = 1f;
    public float alignmentWeight = 1f;
    public float neighborRadius = 5f;
    
    private List<FlockingBehavior> neighbors = new List<FlockingBehavior>();
    private Rigidbody rb;
    
    void Start()
    {
        rb = GetComponent<Rigidbody>();
    }
    
    void Update()
    {
        FindNeighbors();
        Vector3 movement = CalculateMovement();
        ApplyMovement(movement);
    }
    
    private Vector3 CalculateMovement()
    {
        Vector3 cohesion = CalculateCohesion() * cohesionWeight;
        Vector3 separation = CalculateSeparation() * separationWeight;
        Vector3 alignment = CalculateAlignment() * alignmentWeight;
        
        return cohesion + separation + alignment;
    }
    
    private Vector3 CalculateCohesion()
    {
        if (neighbors.Count == 0) return Vector3.zero;
        
        Vector3 centerOfMass = Vector3.zero;
        foreach (var neighbor in neighbors)
        {
            centerOfMass += neighbor.transform.position;
        }
        centerOfMass /= neighbors.Count;
        
        return (centerOfMass - transform.position).normalized;
    }
    
    private Vector3 CalculateSeparation()
    {
        if (neighbors.Count == 0) return Vector3.zero;
        
        Vector3 separationForce = Vector3.zero;
        foreach (var neighbor in neighbors)
        {
            Vector3 direction = transform.position - neighbor.transform.position;
            float distance = direction.magnitude;
            separationForce += direction.normalized / distance;
        }
        
        return separationForce.normalized;
    }
    
    private Vector3 CalculateAlignment()
    {
        if (neighbors.Count == 0) return transform.forward;
        
        Vector3 averageDirection = Vector3.zero;
        foreach (var neighbor in neighbors)
        {
            averageDirection += neighbor.transform.forward;
        }
        
        return averageDirection.normalized;
    }
}
```

### 6.2 群组协调

```csharp
public class GroupCoordinator : MonoBehaviour
{
    private List<AIAgent> groupMembers = new List<AIAgent>();
    private Dictionary<AIAgent, Vector3> formationPositions = new Dictionary<AIAgent, Vector3>();
    
    public void AssignFormationPositions(Vector3 centerPoint, float radius)
    {
        int memberCount = groupMembers.Count;
        float angleStep = 360f / memberCount;
        
        for (int i = 0; i < memberCount; i++)
        {
            float angle = angleStep * i;
            Vector3 position = centerPoint + Quaternion.Euler(0, angle, 0) * Vector3.forward * radius;
            formationPositions[groupMembers[i]] = position;
        }
    }
    
    public void UpdateGroupBehavior(AICommand command)
    {
        switch (command)
        {
            case AICommand.Attack:
                CoordinateAttack();
                break;
            case AICommand.Defend:
                CoordinateDefense();
                break;
            case AICommand.Move:
                CoordinateMovement();
                break;
        }
    }
    
    private void CoordinateAttack()
    {
        foreach (var member in groupMembers)
        {
            member.SetBehavior(AIBehavior.Aggressive);
        }
    }
}
```

## 7. AI优化

### 7.1 性能优化

```csharp
public class AIOptimizer : MonoBehaviour
{
    private List<AIAgent> activeAgents = new List<AIAgent>();
    private float updateInterval = 0.1f;
    private int maxUpdatesPerFrame = 5;
    
    void Update()
    {
        int updateCount = 0;
        float currentTime = Time.time;
        
        foreach (var agent in activeAgents)
        {
            if (currentTime >= agent.NextUpdateTime)
            {
                agent.UpdateAI();
                agent.NextUpdateTime = currentTime + updateInterval;
                
                updateCount++;
                if (updateCount >= maxUpdatesPerFrame)
                    break;
            }
        }
    }
    
    public void OptimizePath(List<Vector3> path, float simplifyTolerance)
    {
        for (int i = path.Count - 1; i >= 2; i--)
        {
            Vector3 start = path[i - 2];
            Vector3 end = path[i];
            Vector3 middle = path[i - 1];
            
            Vector3 direction = (end - start).normalized;
            float distance = Vector3.Distance(middle, start + direction * Vector3.Distance(start, end));
            
            if (distance < simplifyTolerance)
            {
                path.RemoveAt(i - 1);
            }
        }
    }
}
```

### 7.2 LOD系统

```csharp
public class AILOD : MonoBehaviour
{
    private float[] lodDistances = { 10f, 20f, 30f };
    private float[] updateIntervals = { 0.1f, 0.2f, 0.5f };
    
    private Transform player;
    private AIBehavior behavior;
    private int currentLOD = 0;
    
    void Update()
    {
        float distanceToPlayer = Vector3.Distance(transform.position, player.position);
        UpdateLOD(distanceToPlayer);
    }
    
    private void UpdateLOD(float distance)
    {
        for (int i = 0; i < lodDistances.Length; i++)
        {
            if (distance <= lodDistances[i])
            {
                if (currentLOD != i)
                {
                    currentLOD = i;
                    behavior.UpdateInterval = updateIntervals[i];
                    AdjustBehaviorComplexity(i);
                }
                return;
            }
        }
    }
    
    private void AdjustBehaviorComplexity(int lodLevel)
    {
        switch (lodLevel)
        {
            case 0: // 最高质量AI
                behavior.EnableAllFeatures();
                break;
            case 1: // 中等质量AI
                behavior.DisableSecondaryFeatures();
                break;
            case 2: // 最低质量AI
                behavior.EnableBasicFeaturesOnly();
                break;
        }
    }
}
```

```csharp
public class AIDebugVisualizer : MonoBehaviour
{
    public bool showVision = true;
    public bool showPath = true;
    public bool showState = true;
    
    void OnDrawGizmos()
    {
        if (showVision)
        {
            DrawVisionCone();
        }
        
        if (showPath)
        {
            DrawPath();
        }
        
        if (showState)
        {
            DrawState();
        }
    }
    
    private void DrawVisionCone()
    {
        VisionSensor vision = GetComponent<VisionSensor>();
        if (vision != null)
        {
            Gizmos.color = Color.yellow;
            Gizmos.DrawWireSphere(transform.position, vision.viewRadius);
            
            Vector3 viewAngleA = DirFromAngle(-vision.viewAngle / 2);
            Vector3 viewAngleB = DirFromAngle(vision.viewAngle / 2);
            
            Gizmos.DrawLine(transform.position, transform.position + viewAngleA * vision.viewRadius);
            Gizmos.DrawLine(transform.position, transform.position + viewAngleB * vision.viewRadius);
        }
    }
    
    private Vector3 DirFromAngle(float angleInDegrees)
    {
        angleInDegrees += transform.eulerAngles.y;
        return new Vector3(Mathf.Sin(angleInDegrees * Mathf.Deg2Rad), 0, Mathf.Cos(angleInDegrees * Mathf.Deg2Rad));
    }
}
```

### 8.2 性能监控

```csharp
public class AIPerformanceMonitor : MonoBehaviour
{
    private Dictionary<string, float> aiTimings = new Dictionary<string, float>();
    private int frameCount = 0;
    private float totalTime = 0f;
    
    public void BeginSample(string sampleName)
    {
        if (!aiTimings.ContainsKey(sampleName))
        {
            aiTimings[sampleName] = 0f;
        }
        aiTimings[sampleName] -= Time.realtimeSinceStartup;
    }
    
    public void EndSample(string sampleName)
    {
        aiTimings[sampleName] += Time.realtimeSinceStartup;
    }
    
    void Update()
    {
        frameCount++;
        totalTime += Time.deltaTime;
        
        if (totalTime >= 1f)
        {
            LogPerformance();
            ResetStats();
        }
    }
    
    private void LogPerformance()
    {
        foreach (var timing in aiTimings)
        {
            float averageTime = timing.Value / frameCount * 1000f;
            Debug.Log($"{timing.Key}: {averageTime:F2}ms");
        }
    }
}
```
