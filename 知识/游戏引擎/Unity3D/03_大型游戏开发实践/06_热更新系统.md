# Unity热更新系统

<!-- DEEP-TOPIC-GUIDE-START -->
> 热更新是项目上线后最容易“牵一发而动全身”的能力之一。它不只是更新文件，而是版本、资源、代码、配置、灰度和回滚策略的组合问题。

## 先读这篇时，建议带着这些问题

- 为什么热更新不是简单“下载新文件”就完了
- 资源热更新、代码热更新、配置热更新分别解决什么问题
- 版本、依赖、校验、回滚怎么设计才稳
- 热更新为什么会和资源管理、框架设计、发布部署紧密耦合

## 1. 热更新概述

### 1.1 什么是热更新

- 在不重启游戏的情况下更新游戏内容
- 减少用户等待时间
- 节省流量和存储空间
- 快速修复Bug和内容更新

### 1.2 热更新的类型

- 资源热更新
- 代码热更新
- 配置热更新
- 场景热更新

## 2. 资源热更新

### 2.1 资源管理系统

```csharp
public class ResourceUpdateManager : MonoBehaviour
{
    private string serverURL = "http://game-server.com/resources/";
    private string versionFile = "version.json";
    
    [System.Serializable]
    public class VersionInfo
    {
        public string version;
        public Dictionary<string, string> files; // 文件名和MD5
    }
    
    public async Task CheckForUpdates()
    {
        try
        {
            // 获取服务器版本信息
            string serverVersion = await GetServerVersion();
            string localVersion = PlayerPrefs.GetString("ResourceVersion", "1.0.0");
            
            if (serverVersion != localVersion)
            {
                // 需要更新
                await UpdateResources();
            }
        }
        catch (Exception e)
        {
            Debug.LogError($"检查更新失败: {e.Message}");
        }
    }
    
    private async Task UpdateResources()
    {
        // 获取需要更新的文件列表
        List<string> updateFiles = await GetUpdateFileList();
        
        foreach (string file in updateFiles)
        {
            await DownloadFile(file);
        }
        
        // 更新本地版本号
        PlayerPrefs.SetString("ResourceVersion", await GetServerVersion());
        PlayerPrefs.Save();
    }
    
    private async Task DownloadFile(string fileName)
    {
        using (UnityWebRequest www = UnityWebRequest.Get($"{serverURL}{fileName}"))
        {
            await www.SendWebRequest();
            
            if (www.result == UnityWebRequest.Result.Success)
            {
                // 保存文件
                string path = Path.Combine(Application.persistentDataPath, fileName);
                File.WriteAllBytes(path, www.downloadHandler.data);
            }
            else
            {
                throw new Exception($"下载失败: {www.error}");
            }
        }
    }
}
```

### 2.2 资源版本控制

```csharp
public class ResourceVersionControl
{
    private Dictionary<string, AssetBundleInfo> bundleInfos = new Dictionary<string, AssetBundleInfo>();
    
    [System.Serializable]
    public class AssetBundleInfo
    {
        public string version;
        public string md5;
        public long size;
        public string[] dependencies;
    }
    
    public void GenerateVersionFile()
    {
        string[] bundles = Directory.GetFiles(Application.streamingAssetsPath, "*.bundle");
        
        foreach (string bundle in bundles)
        {
            string fileName = Path.GetFileName(bundle);
            string md5 = CalculateMD5(bundle);
            long size = new FileInfo(bundle).Length;
            
            bundleInfos[fileName] = new AssetBundleInfo
            {
                version = "1.0.0",
                md5 = md5,
                size = size,
                dependencies = AssetDatabase.GetAssetBundleDependencies(fileName, true)
            };
        }
        
        // 保存版本文件
        string json = JsonUtility.ToJson(bundleInfos);
        File.WriteAllText("version.json", json);
    }
    
    private string CalculateMD5(string filePath)
    {
        using (var md5 = MD5.Create())
        {
            using (var stream = File.OpenRead(filePath))
            {
                byte[] hash = md5.ComputeHash(stream);
                return BitConverter.ToString(hash).Replace("-", "").ToLowerInvariant();
            }
        }
    }
}
```

## 3. 代码热更新

### 3.1 ILRuntime实现

```csharp
public class ILRuntimeManager
{
    private ILRuntime.Runtime.Enviorment.AppDomain appDomain;
    private MemoryStream dllStream;
    private MemoryStream pdbStream;
    
    public async Task LoadHotfixAssembly()
    {
        appDomain = new ILRuntime.Runtime.Enviorment.AppDomain();
        
        // 加载热更新DLL
        byte[] dllBytes = await LoadDLLBytes();
        byte[] pdbBytes = await LoadPDBBytes();
        
        dllStream = new MemoryStream(dllBytes);
        pdbStream = new MemoryStream(pdbBytes);
        
        appDomain.LoadAssembly(dllStream, pdbStream, new ILRuntime.Mono.Cecil.Pdb.PdbReaderProvider());
        
        // 注册CLR绑定
        RegisterCLRBindings();
    }
    
    private void RegisterCLRBindings()
    {
        // 注册跨域继承适配器
        appDomain.RegisterCrossBindingAdaptor(new MonoBehaviourAdapter());
        
        // 注册值类型绑定
        appDomain.RegisterValueTypeBinder(typeof(Vector3), new Vector3Binder());
        
        // 注册委托转换器
        appDomain.DelegateManager.RegisterMethodDelegate<void>();
    }
    
    public void InvokeHotfixMethod(string typeName, string methodName, params object[] args)
    {
        appDomain.Invoke(typeName, methodName, null, args);
    }
}
```

### 3.2 Lua热更新

```csharp
public class LuaManager : MonoBehaviour
{
    private LuaEnv luaEnv;
    private Dictionary<string, byte[]> scriptCache = new Dictionary<string, byte[]>();
    
    void Start()
    {
        InitLuaEnvironment();
    }
    
    private void InitLuaEnvironment()
    {
        luaEnv = new LuaEnv();
        
        // 注册自定义加载器
        luaEnv.AddLoader(CustomLoader);
        
        // 加载入口脚本
        luaEnv.DoString("require 'Main'");
    }
    
    private byte[] CustomLoader(ref string filePath)
    {
        // 先从缓存中查找
        if (scriptCache.TryGetValue(filePath, out byte[] bytes))
        {
            return bytes;
        }
        
        // 从资源中加载
        string path = $"Lua/{filePath}.lua.txt";
        TextAsset textAsset = Resources.Load<TextAsset>(path);
        
        if (textAsset != null)
        {
            scriptCache[filePath] = textAsset.bytes;
            return textAsset.bytes;
        }
        
        return null;
    }
    
    public void CallLuaFunction(string funcName, params object[] args)
    {
        LuaTable global = luaEnv.Global;
        LuaFunction func = global.Get<LuaFunction>(funcName);
        
        if (func != null)
        {
            func.Call(args);
        }
    }
    
    void OnDestroy()
    {
        luaEnv?.Dispose();
    }
}
```

## 4. 配置热更新

### 4.1 配置管理器

```csharp
public class ConfigManager : MonoBehaviour
{
    private Dictionary<string, object> configCache = new Dictionary<string, object>();
    private string configServerUrl = "http://game-server.com/configs/";
    
    public async Task UpdateConfigs()
    {
        try
        {
            // 获取配置列表
            string[] configFiles = await GetConfigFileList();
            
            foreach (string file in configFiles)
            {
                await UpdateConfig(file);
            }
        }
        catch (Exception e)
        {
            Debug.LogError($"更新配置失败: {e.Message}");
        }
    }
    
    private async Task UpdateConfig(string configName)
    {
        using (UnityWebRequest www = UnityWebRequest.Get($"{configServerUrl}{configName}"))
        {
            await www.SendWebRequest();
            
            if (www.result == UnityWebRequest.Result.Success)
            {
                string json = www.downloadHandler.text;
                
                // 解析并缓存配置
                object config = JsonUtility.FromJson(json, GetConfigType(configName));
                configCache[configName] = config;
                
                // 保存到本地
                SaveConfigToLocal(configName, json);
            }
        }
    }
    
    private void SaveConfigToLocal(string configName, string content)
    {
        string path = Path.Combine(Application.persistentDataPath, "Configs", configName);
        Directory.CreateDirectory(Path.GetDirectoryName(path));
        File.WriteAllText(path, content);
    }
    
    public T GetConfig<T>(string configName) where T : class
    {
        if (configCache.TryGetValue(configName, out object config))
        {
            return config as T;
        }
        return null;
    }
}
```

### 4.2 配置加密

```csharp
public class ConfigEncryption
{
    private readonly byte[] key;
    private readonly byte[] iv;
    
    public ConfigEncryption(string key, string iv)
    {
        this.key = Convert.FromBase64String(key);
        this.iv = Convert.FromBase64String(iv);
    }
    
    public string Encrypt(string content)
    {
        using (Aes aes = Aes.Create())
        {
            aes.Key = key;
            aes.IV = iv;
            
            using (MemoryStream ms = new MemoryStream())
            {
                using (CryptoStream cs = new CryptoStream(ms, aes.CreateEncryptor(), CryptoStreamMode.Write))
                using (StreamWriter sw = new StreamWriter(cs))
                {
                    sw.Write(content);
                }
                
                return Convert.ToBase64String(ms.ToArray());
            }
        }
    }
    
    public string Decrypt(string encryptedContent)
    {
        byte[] buffer = Convert.FromBase64String(encryptedContent);
        
        using (Aes aes = Aes.Create())
        {
            aes.Key = key;
            aes.IV = iv;
            
            using (MemoryStream ms = new MemoryStream(buffer))
            using (CryptoStream cs = new CryptoStream(ms, aes.CreateDecryptor(), CryptoStreamMode.Read))
            using (StreamReader sr = new StreamReader(cs))
            {
                return sr.ReadToEnd();
            }
        }
    }
}
```

## 5. 场景热更新

### 5.1 场景管理器

```csharp
public class SceneUpdateManager : MonoBehaviour
{
    private Dictionary<string, string> sceneVersions = new Dictionary<string, string>();
    private string sceneServerUrl = "http://game-server.com/scenes/";
    
    public async Task UpdateScene(string sceneName)
    {
        try
        {
            // 检查场景版本
            string serverVersion = await GetSceneVersion(sceneName);
            string localVersion = PlayerPrefs.GetString($"Scene_{sceneName}_Version", "1.0.0");
            
            if (serverVersion != localVersion)
            {
                // 下载新场景
                await DownloadScene(sceneName);
                
                // 更新版本号
                PlayerPrefs.SetString($"Scene_{sceneName}_Version", serverVersion);
                PlayerPrefs.Save();
            }
        }
        catch (Exception e)
        {
            Debug.LogError($"更新场景失败: {e.Message}");
        }
    }
    
    private async Task DownloadScene(string sceneName)
    {
        using (UnityWebRequest www = UnityWebRequest.Get($"{sceneServerUrl}{sceneName}"))
        {
            await www.SendWebRequest();
            
            if (www.result == UnityWebRequest.Result.Success)
            {
                // 保存场景文件
                string path = Path.Combine(Application.persistentDataPath, "Scenes", sceneName);
                Directory.CreateDirectory(Path.GetDirectoryName(path));
                File.WriteAllBytes(path, www.downloadHandler.data);
            }
        }
    }
    
    public async Task LoadUpdatedScene(string sceneName)
    {
        // 先更新场景
        await UpdateScene(sceneName);
        
        // 加载场景
        string scenePath = Path.Combine(Application.persistentDataPath, "Scenes", sceneName);
        if (File.Exists(scenePath))
        {
            SceneManager.LoadScene(sceneName);
        }
    }
}
```

### 5.2 场景补丁

```csharp
public class ScenePatchManager
{
    [System.Serializable]
    public class ScenePatch
    {
        public string version;
        public List<PatchItem> items;
    }
    
    [System.Serializable]
    public class PatchItem
    {
        public string type; // Add, Modify, Delete
        public string path;
        public string data;
    }
    
    public async Task ApplyScenePatch(string sceneName, ScenePatch patch)
    {
        foreach (var item in patch.items)
        {
            switch (item.type)
            {
                case "Add":
                    await AddSceneObject(sceneName, item);
                    break;
                case "Modify":
                    await ModifySceneObject(sceneName, item);
                    break;
                case "Delete":
                    DeleteSceneObject(sceneName, item);
                    break;
            }
        }
        
        // 保存场景
        EditorSceneManager.SaveScene(SceneManager.GetActiveScene());
    }
    
    private async Task AddSceneObject(string sceneName, PatchItem item)
    {
        // 从补丁数据创建对象
        GameObject obj = JsonUtility.FromJson<GameObject>(item.data);
        if (obj != null)
        {
            // 设置对象位置等属性
            obj.transform.position = JsonUtility.FromJson<Vector3>(item.data);
            // 其他设置...
        }
    }
    
    private async Task ModifySceneObject(string sceneName, PatchItem item)
    {
        // 查找并修改对象
        GameObject obj = GameObject.Find(item.path);
        if (obj != null)
        {
            // 应用修改
            JsonUtility.FromJsonOverwrite(item.data, obj);
        }
    }
    
    private void DeleteSceneObject(string sceneName, PatchItem item)
    {
        // 查找并删除对象
        GameObject obj = GameObject.Find(item.path);
        if (obj != null)
        {
            DestroyImmediate(obj);
        }
    }
}
```

## 6. 热更新管理器

### 6.1 更新流程控制

```csharp
public class HotUpdateManager : MonoBehaviour
{
    private ResourceUpdateManager resourceManager;
    private ILRuntimeManager codeManager;
    private ConfigManager configManager;
    private SceneUpdateManager sceneManager;
    
    public async Task StartUpdate()
    {
        try
        {
            // 检查网络
            if (!await CheckNetwork())
            {
                throw new Exception("网络连接失败");
            }
            
            // 更新资源
            await resourceManager.CheckForUpdates();
            
            // 更新代码
            await codeManager.LoadHotfixAssembly();
            
            // 更新配置
            await configManager.UpdateConfigs();
            
            // 更新场景
            await UpdateRequiredScenes();
            
            // 更新完成
            OnUpdateComplete();
        }
        catch (Exception e)
        {
            OnUpdateFailed(e);
        }
    }
    
    private async Task<bool> CheckNetwork()
    {
        using (UnityWebRequest www = UnityWebRequest.Get("http://game-server.com/heartbeat"))
        {
            await www.SendWebRequest();
            return www.result == UnityWebRequest.Result.Success;
        }
    }
    
    private void OnUpdateComplete()
    {
        Debug.Log("热更新完成");
        // 通知其他系统
        EventManager.Dispatch("HotUpdateComplete");
    }
    
    private void OnUpdateFailed(Exception e)
    {
        Debug.LogError($"热更新失败: {e.Message}");
        // 通知其他系统
        EventManager.Dispatch("HotUpdateFailed", e);
    }
}
```

### 6.2 更新UI

```csharp
public class HotUpdateUI : MonoBehaviour
{
    public Text progressText;
    public Slider progressBar;
    public Text statusText;
    public Button retryButton;
    
    private HotUpdateManager updateManager;
    
    void Start()
    {
        updateManager = GetComponent<HotUpdateManager>();
        retryButton.onClick.AddListener(OnRetryClick);
        
        // 注册事件
        EventManager.AddListener("UpdateProgress", OnUpdateProgress);
        EventManager.AddListener("HotUpdateComplete", OnUpdateComplete);
        EventManager.AddListener("HotUpdateFailed", OnUpdateFailed);
        
        // 开始更新
        StartUpdate();
    }
    
    private async void StartUpdate()
    {
        ShowProgress(0, "正在检查更新...");
        await updateManager.StartUpdate();
    }
    
    private void OnUpdateProgress(object data)
    {
        UpdateProgress progress = data as UpdateProgress;
        if (progress != null)
        {
            ShowProgress(progress.progress, progress.description);
        }
    }
    
    private void ShowProgress(float progress, string status)
    {
        progressBar.value = progress;
        progressText.text = $"{(progress * 100):F1}%";
        statusText.text = status;
    }
    
    private void OnUpdateComplete()
    {
        ShowProgress(1, "更新完成");
        // 进入游戏
        StartCoroutine(EnterGame());
    }
    
    private void OnUpdateFailed(object error)
    {
        ShowProgress(0, $"更新失败: {error}");
        retryButton.gameObject.SetActive(true);
    }
    
    private void OnRetryClick()
    {
        retryButton.gameObject.SetActive(false);
        StartUpdate();
    }
    
    private IEnumerator EnterGame()
    {
        yield return new WaitForSeconds(1f);
        SceneManager.LoadScene("Game");
    }
}
```

```csharp
public class HotUpdateTester : EditorWindow
{
    [MenuItem("Tools/热更新测试工具")]

        GetWindow<HotUpdateTester>("热更新测试");

        StartCoroutine(SimulateUpdateProcess());
    }

    {
        // 模拟下载延迟
        yield return new WaitForSeconds(1f);

            EditorUtility.DisplayProgressBar("模拟更新", 
                $"正在更新... {progress * 100}%", progress);
            yield return new WaitForSeconds(0.5f);
        }
        
        EditorUtility.ClearProgressBar();
    }
```

```csharp
public class HotUpdateTests

    [Test]
```
