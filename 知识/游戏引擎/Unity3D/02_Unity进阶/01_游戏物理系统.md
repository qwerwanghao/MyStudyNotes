# 游戏物理系统

<!-- DEEP-TOPIC-GUIDE-START -->
> 物理系统决定你的对象在“运动、碰撞、约束、检测”层面是否自然可控。它既是玩法表现的基础，也是性能和稳定性的大坑来源。

## 先读这篇时，建议带着这些问题

- Rigidbody、Collider、Trigger 各自该什么时候用
- 为什么 FixedUpdate 和 Update 不应该混着乱写
- 关节、射线、碰撞回调如何协作
- 物理系统为什么经常和性能、角色控制、动画系统纠缠在一起

## 推荐阅读顺序

1. 刚体系统
2. 碰撞检测
3. 物理关节
4. 射线检测
5. 常见交互模式
6. 稳定性与性能注意点

## 这一页最值得重点抓住的概念

- **物理更新节奏和普通帧更新节奏不是一回事**
- **Trigger 和 Collision 适合解决不同类型的问题**
- **关节和射线通常是构建复杂交互的关键工具**
- **物理逻辑写法不当，最容易带来手感和性能双重问题**

## 刚体系统

### 基础刚体操作
```csharp
public class RigidbodyController : MonoBehaviour
{
    private Rigidbody rb;
    
    void Start()
    {
        rb = GetComponent<Rigidbody>();
        
        // 基本属性设置
        rb.mass = 1.0f;              // 质量
        rb.drag = 0.5f;              // 空气阻力
        rb.angularDrag = 0.5f;       // 角阻力
        rb.useGravity = true;        // 是否使用重力
        rb.isKinematic = false;      // 是否由物理引擎控制
    }
    
    void FixedUpdate()
    {
        // 添加力
        rb.AddForce(Vector3.forward * 10f);
        
        // 添加扭矩
        rb.AddTorque(Vector3.up * 5f);
        
        // 直接设置速度
        rb.velocity = new Vector3(1f, 0f, 0f);
        
        // 设置角速度
        rb.angularVelocity = new Vector3(0f, 1f, 0f);
    }
    
    // 施加爆炸力
    public void ApplyExplosionForce(Vector3 explosionPosition)
    {
        rb.AddExplosionForce(1000f, explosionPosition, 5f, 1f, 
            ForceMode.Impulse);
    }
}
```

### 碰撞检测
```csharp
public class CollisionHandler : MonoBehaviour
{
    void OnCollisionEnter(Collision collision)
    {
        // 获取碰撞信息
        foreach (ContactPoint contact in collision.contacts)
        {
            Debug.DrawRay(contact.point, contact.normal, Color.red, 2f);
            
            // 处理碰撞点
            HandleCollisionPoint(contact.point, contact.normal);
        }
        
        // 计算碰撞力
        float collisionForce = collision.impulse.magnitude / 
            Time.fixedDeltaTime;
        
        // 处理碰撞响应
        if (collisionForce > 10f)
        {
            TakeDamage(collisionForce);
        }
    }
    
    void OnTriggerEnter(Collider other)
    {
        // 触发器检测
        if (other.CompareTag("Item"))
        {
            CollectItem(other.gameObject);
        }
    }
    
    // 持续碰撞检测
    void OnCollisionStay(Collision collision)
    {
        // 处理持续接触
        HandleContinuousContact(collision);
    }
}
```

## 物理关节

### 铰链关节
```csharp
public class HingeJointExample : MonoBehaviour
{
    private HingeJoint hinge;
    
    void Start()
    {
        hinge = gameObject.AddComponent<HingeJoint>();
        
        // 设置铰链属性
        hinge.axis = Vector3.up;
        hinge.anchor = Vector3.zero;
        
        // 配置限制
        var limits = hinge.limits;
        limits.min = -45f;
        limits.max = 45f;
        limits.bounciness = 0.5f;
        hinge.limits = limits;
        
        // 设置弹簧
        var spring = hinge.spring;
        spring.spring = 10f;
        spring.damper = 0.2f;
        spring.targetPosition = 0f;
        hinge.spring = spring;
    }
    
    // 控制铰链
    public void RotateHinge(float targetAngle)
    {
        var motor = hinge.motor;
        motor.targetVelocity = targetAngle;
        motor.force = 50f;
        hinge.motor = motor;
        hinge.useMotor = true;
    }
}
```

## 射线检测

### 基础射线检测
```csharp
public class RaycastController : MonoBehaviour
{
    // 单次射线检测
    public bool SingleRaycast(Vector3 origin, Vector3 direction, 
        float distance)
    {
        RaycastHit hit;
        if (Physics.Raycast(origin, direction, out hit, distance))
        {
            Debug.Log($"Hit object: {hit.collider.name} at distance: " +
                     $"{hit.distance}");
            return true;
        }
        return false;
    }
    
    // 多重射线检测
    public void MultipleRaycast()
    {
        RaycastHit[] hits = Physics.RaycastAll(
            transform.position, 
            transform.forward, 
            100f
        );
        
        foreach (RaycastHit hit in hits)
        {
            ProcessHit(hit);
        }
    }
    
    // 球形射线检测
    public void SphereCast()
    {
        RaycastHit hit;
        if (Physics.SphereCast(transform.position, 1f, transform.forward, 
            out hit, 10f))
        {
            Debug.Log($"Sphere hit: {hit.collider.name}");
        }
    }
}
```

### 射线检测应用
```csharp
public class RaycastApplications : MonoBehaviour
{
    // 地面检测
    public bool IsGrounded()
    {
        return Physics.Raycast(
            transform.position,
            Vector3.down,
            1.1f,
            LayerMask.GetMask("Ground")
        );
    }
    
    // 视线检测
    public bool HasLineOfSight(Transform target)
    {
        RaycastHit hit;
        Vector3 direction = (target.position - transform.position)
            .normalized;
            
        if (Physics.Raycast(transform.position, direction, out hit))
        {
            return hit.transform == target;
        }
        return false;
    }
    
    // 武器射线
    public void FireWeapon()
    {
        Ray ray = Camera.main.ScreenPointToRay(Input.mousePosition);
        RaycastHit hit;
        
        if (Physics.Raycast(ray, out hit))
        {
            var damageable = hit.collider.GetComponent<IDamageable>();
            if (damageable != null)
            {
                damageable.TakeDamage(10f);
            }
            
            // 生成击中特效
            CreateHitEffect(hit.point, hit.normal);
        }
    }
}
```

## 物理优化

### 性能调优
```csharp
public class PhysicsOptimization : MonoBehaviour
{
    void Start()
    {
        // 配置物理设置
        Physics.defaultSolverIterations = 6;
        Physics.defaultSolverVelocityIterations = 1;
        
        // 设置分层碰撞
        Physics.IgnoreLayerCollision(
            LayerMask.NameToLayer("Player"),
            LayerMask.NameToLayer("PlayerProjectile")
        );
    }
    
    // 使用物理材质池
    private PhysicMaterial CreatePhysicMaterial(float friction, 
        float bounciness)
    {
        var material = new PhysicMaterial()
        {
            dynamicFriction = friction,
            staticFriction = friction,
            bounciness = bounciness,
            frictionCombine = PhysicMaterialCombine.Average,
            bounceCombine = PhysicMaterialCombine.Maximum
        };
        return material;
    }
    
    // 优化碰撞体
    public void OptimizeColliders()
    {
        var colliders = GetComponentsInChildren<Collider>();
        foreach (var collider in colliders)
        {
            // 使用简化的碰撞体
            if (collider is MeshCollider meshCollider)
            {
                meshCollider.convex = true;
                meshCollider.cookingOptions = 
                    MeshColliderCookingOptions.UseFastMidphase;
            }
        }
    }
}
```

### 物理模拟优化
```csharp
public class PhysicsSimulationOptimizer : MonoBehaviour
{
    // 使用自定义物理更新
    public void CustomPhysicsUpdate()
    {
        Physics.autoSimulation = false;
        
        // 手动控制物理模拟步骤
        Physics.Simulate(Time.fixedDeltaTime);
    }
    
    // 使用物理分组
    public void SetupPhysicsGroups()
    {
        // 创建物理分组
        PhysicsScene mainScene = gameObject.scene.GetPhysicsScene();
        
        // 在独立的物理场景中模拟
        if (mainScene.IsValid())
        {
            mainScene.Simulate(Time.fixedDeltaTime);
        }
    }
    
    // 优化刚体休眠
    public void OptimizeRigidbodySleep()
    {
        var rigidbodies = GetComponentsInChildren<Rigidbody>();
        foreach (var rb in rigidbodies)
        {
            rb.sleepThreshold = 0.005f;
            rb.solverIterations = 4;
            rb.maxAngularVelocity = 7f;
        }
    }
}
